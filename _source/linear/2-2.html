        const rotatePoint = (x, y, z, angleX, angleY) => {

            let cosY = Math.cos(angleY);

            let sinY = Math.sin(angleY);

            let x1 = x * cosY - z * sinY;

            let z1 = x * sinY + z * cosY;


            let cosX = Math.cos(angleX);

            let sinX = Math.sin(angleX);

            let y2 = y * cosX - z1 * sinX;

            let z2 = y * sinX + z1 * cosX;


            return { x: x1, y: y2, z: z2 };

        };


        const project = (visX, visY, visZ, width, height, scale, angleX, angleY) => {

            const rotated = rotatePoint(visX, visY, visZ, angleX, angleY);

            return {

                x: width / 2 + rotated.x * scale,

                y: height / 2 - rotated.y * scale,

                z: rotated.z 

            };

        };


        // Matrix Multiplication (3x3 * 3x3)

        const multiplyMatrices = (m1, m2) => {

            const res = {};

            for (let i = 1; i <= 3; i++) {

                for (let j = 1; j <= 3; j++) {

                    let sum = 0;

                    for (let k = 1; k <= 3; k++) {

                        sum += m1[`a${i}${k}`] * m2[`a${k}${j}`];

                    }

                    res[`a${i}${j}`] = sum;

                }

            }

            return res;

        };

            '3-2-3': { mode: 'product', type: 'prod_323', label: '3D → 2D → 3D', dims: { in: 3, mid: 2, out: 3 } }, // B(2x3), A(3x2)

        const App = () => {

            const [configKey, setConfigKey] = useState('3-3');

            const config = CONFIGS[configKey];

            const { mode, dims } = config;


            // Matrix A & B (Always maintain 3x3 internal state, mask externally)

            const [matrixA, setMatrixA] = useState(identityMatrix);

            const [matrixB, setMatrixB] = useState(identityMatrix);

            const [userVector, setUserVector] = useState({ x: 1, y: 1, z: 1 });

            

            const [t, setT] = useState(0);

            // Reset state when config changes

            useEffect(() => {

                resetViewAndValues();

            // Reset Function

            const resetViewAndValues = () => {

                setT(0);

                setIsAnimating(false);

                setMatrixA(identityMatrix);

                setMatrixB(identityMatrix);

                setZoom(1.0);

                

                // For 2D input modes, reset Z vector to 0

                const defaultV = { x: 1, y: 1, z: 1 };

                if (CONFIGS[configKey].dims.in === 2) {

                    defaultV.z = 0;

                }

                setUserVector(defaultV);

            };


            // Calculate Effective Current Matrix (Interpolated)

            const currentMatrix = useMemo(() => {

                const interpolate = (startM, endM, factor) => {

                    const res = {};

                    for (let key in startM) {

                        res[key] = startM[key] + (endM[key] - startM[key]) * factor;

                    }

                    return res;

                };


                let effA = {...matrixA};

                let effB = {...matrixB};


                // Apply constraints to A and B based on dimensions

                // If single mode (A):

                if (mode === 'single') {

                    // dims.in (cols), dims.out (rows)

                    if (dims.out === 2) { effA.a31=0; effA.a32=0; effA.a33=0; } // Row 3 = 0

                    if (dims.in === 2) { effA.a13=0; effA.a23=0; effA.a33=0; } // Col 3 = 0 (Effectively)

                } else {

                    // Product mode: B is first (in -> mid), A is second (mid -> out)

                    // Matrix B constraints

                    if (dims.mid === 2) { effB.a31=0; effB.a32=0; effB.a33=0; } // B Row 3 = 0

                    if (dims.in === 2) { effB.a13=0; effB.a23=0; effB.a33=0; } // B Col 3 = 0


                    // Matrix A constraints

                    if (dims.out === 2) { effA.a31=0; effA.a32=0; effA.a33=0; } // A Row 3 = 0

                    if (dims.mid === 2) { effA.a13=0; effA.a23=0; effA.a33=0; } // A Col 3 = 0

                }


                if (mode === 'single') {

                    return interpolate(identityMatrix, effA, t);

                } else {

                    // Phase 1 (0 <= t < 0.5): I -> B

                    if (t <= 0.5) {

                        const factor = t * 2;

                        return interpolate(identityMatrix, effB, factor);

                    } else {

                        // Phase 2 (0.5 <= t <= 1): B -> AB

                        const factor = (t - 0.5) * 2;

                        const partialA = interpolate(identityMatrix, effA, factor);

                        return multiplyMatrices(partialA, effB);

                    }

                }

            }, [matrixA, matrixB, t, configKey]);


            // Final Result Vector Calculation (for display)

            const resultVector = useMemo(() => {

                const m = currentMatrix;

                // Force input vector z=0 if dims.in is 2

                const v = dims.in === 2 ? { ...userVector, z: 0 } : userVector;

                

                const res = {

                    x: m.a11 * v.x + m.a12 * v.y + m.a13 * v.z,

                    y: m.a21 * v.x + m.a22 * v.y + m.a23 * v.z,

                    z: m.a31 * v.x + m.a32 * v.y + m.a33 * v.z

                };

                

                // Force output vector z=0 if dims.out is 2 (just to be clean)

                if (dims.out === 2) res.z = 0;

                

                return res;

            }, [currentMatrix, userVector, dims]);


            useEffect(() => {

                let animationFrameId;

                if (isAnimating) {

                    const step = 0.005;

                    const animate = () => {

                        setT(prev => {

                            if (prev >= 1) {

                                setIsAnimating(false);

                                return 1;

                            }

                            return prev + step;

                        });

                        animationFrameId = requestAnimationFrame(animate);

                    };

                    animationFrameId = requestAnimationFrame(animate);

                }

                return () => cancelAnimationFrame(animationFrameId);

            }, [isAnimating]);


            useEffect(() => { lucide.createIcons(); }, []);


            // --- Drawing Logic ---

            useEffect(() => {

                const canvas = canvasRef.current;

                const ctx = canvas.getContext('2d');

                const container = containerRef.current;

                

                const w = container.clientWidth;

                const h = container.clientHeight;

                if (w === 0 || h === 0) return;

                canvas.width = w; canvas.height = h;


                const scale = (Math.min(w, h) / 6) * zoom; 


                ctx.fillStyle = "#0f172a";

                ctx.fillRect(0, 0, w, h);


                // Helper to enforce 2D input

                const enforceInputDim = (p) => dims.in === 2 ? { ...p, z: 0 } : p;


                const transform = (p) => {

                    // Apply dimension constraint to input point

                    const pIn = enforceInputDim(p);

                    const m = currentMatrix;

                    return {

                        x: m.a11 * pIn.x + m.a12 * pIn.y + m.a13 * pIn.z,

                        y: m.a21 * pIn.x + m.a22 * pIn.y + m.a23 * pIn.z,

                        z: m.a31 * pIn.x + m.a32 * pIn.y + m.a33 * pIn.z

                    };

                };


                const toScreen = (p) => project(p.x, p.z, p.y, w, h, scale, camera.x, camera.y);


                const drawLine = (p1, p2, color, width = 1, dashed = false) => {

                    const s1 = toScreen(p1);

                    const s2 = toScreen(p2);

                    ctx.beginPath();

                    ctx.moveTo(s1.x, s1.y);

                    ctx.lineTo(s2.x, s2.y);

                    ctx.strokeStyle = color;

                    ctx.lineWidth = width;

                    if (dashed) ctx.setLineDash([4, 4]); else ctx.setLineDash([]);

                    ctx.stroke();

                    ctx.setLineDash([]);

                };


                const drawDot = (p, color, radius = 3) => {

                    const s = toScreen(p);

                    ctx.beginPath();

                    ctx.arc(s.x, s.y, radius, 0, Math.PI * 2);

                    ctx.fillStyle = color;

                    ctx.fill();

                };


                const origin = { x: 0, y: 0, z: 0 };


                // Grid (Floor)

                const gridSize = 4;

                ctx.lineWidth = 1;

                for (let i = -gridSize; i <= gridSize; i++) {

                    drawLine({x: i, y: -gridSize, z: 0}, {x: i, y: gridSize, z: 0}, "rgba(255, 255, 255, 0.05)", 1);

                    drawLine({x: -gridSize, y: i, z: 0}, {x: gridSize, y: i, z: 0}, "rgba(255, 255, 255, 0.05)", 1);

                }

                drawLine({x:0, y:0, z:0}, {x:0, y:5, z:0}, "rgba(16, 185, 129, 0.2)", 1);

                drawLine({x:0, y:0, z:0}, {x:0, y:0, z:5}, "rgba(59, 130, 246, 0.2)", 1);

                        }

                        ctx.fill();

                    ctx.lineTo(sTo.x, sTo.y);

                    ctx.strokeStyle = color;

                    ctx.lineWidth = 3;

                    ctx.stroke();

                    ctx.fillStyle = color;

                    ctx.font = "bold 14px Arial";

                    ctx.fillText(label, sTo.x + 5, sTo.y - 5);

                    drawDot(to, color, 4);

                };

                // ... omitted for clarity in mixed dimensions ...
