        document.addEventListener('DOMContentLoaded', () => {

            // --- Shared Utility Functions ---

            const factorial = (n) => {

                if (n < 0) return Infinity;

                if (n === 0) return 1;

                let result = 1;

                for (let i = 2; i <= n; i++) result *= i;

                return result;

            };


            const setupCanvas = (canvas, ctx) => {

                const dpr = window.devicePixelRatio || 1;

                const rect = canvas.parentElement.getBoundingClientRect();

                if (rect.width === 0 || rect.height === 0) return; // Don't setup hidden canvases

                canvas.width = rect.width * dpr;

                canvas.height = rect.height * dpr;

                canvas.style.width = `${rect.width}px`;

                canvas.style.height = `${rect.height}px`;

                ctx.scale(dpr, dpr);

            };


            const drawGridAndAxes = (ctx, width, height, xRange, yRange, showDetails = true, axisColor, gridColor, textColor) => {

                ctx.clearRect(0, 0, width, height);

                ctx.fillStyle = '#FFFFFF';

                ctx.fillRect(0, 0, width, height);


                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const toCanvasX = (x) => (x - xMin) / (xMax - xMin) * width;

                const toCanvasY = (y) => height - (y - yMin) / (yMax - yMin) * height;


                if (showDetails) {

                    ctx.strokeStyle = gridColor;

                    ctx.lineWidth = 1;

                    ctx.beginPath();

                    const xStep = (xMax - xMin) / 10;

                    for (let x = xMin; x <= xMax; x += xStep) { const cx = toCanvasX(x); ctx.moveTo(cx, 0); ctx.lineTo(cx, height); }

                    const yStep = (yMax - yMin) / 10;

                    for (let y = yMin; y <= yMax; y += yStep) { const cy = toCanvasY(y); ctx.moveTo(0, cy); ctx.lineTo(width, cy); }

                    ctx.stroke();

                }


                ctx.strokeStyle = axisColor;

                ctx.lineWidth = showDetails ? 2 : 1.5;

                ctx.beginPath();

                const yZero = toCanvasY(0);

                if (yZero >= 0 && yZero <= height) { ctx.moveTo(0, yZero); ctx.lineTo(width, yZero); }

                const xZero = toCanvasX(0);

                if (xZero >= 0 && xZero <= width) { ctx.moveTo(xZero, 0); ctx.lineTo(xZero, height); }

                ctx.stroke();


                if (showDetails) {

                    ctx.fillStyle = textColor;

                    ctx.font = '12px sans-serif';

                    ctx.textAlign = 'center'; ctx.textBaseline = 'top';

                    const xStep = (xMax - xMin) / 5;

                    for (let i = 1; i < 5; i++) { const x = xMin + i * xStep; if(Math.abs(x) > 1e-9) ctx.fillText(x.toFixed(2), toCanvasX(x), yZero + 5); }

                    ctx.textAlign = 'right'; ctx.textBaseline = 'middle';

                    const yStep = (yMax - yMin) / 5;

                    for (let i = 1; i < 5; i++) { const y = yMin + i * yStep; if(Math.abs(y) > 1e-9) ctx.fillText(y.toFixed(2), xZero - 5, toCanvasY(y)); }

                }

            };


            const drawFunction = (ctx, width, height, xRange, yRange, func, color, lineWidth = 2) => {

                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const toCanvasX = (x) => (x - xMin) / (xMax - xMin) * width;

                const toCanvasY = (y) => height - (y - yMin) / (yMax - yMin) * height;

                

                ctx.strokeStyle = color;

                ctx.lineWidth = lineWidth;

                ctx.beginPath();

                let firstPoint = true;

                for (let px = 0; px <= width; px++) {

                    const x = xMin + (px / width) * (xMax - xMin);

                    const y = func(x);

                    const cx = toCanvasX(x);

                    const cy = toCanvasY(y);

                    if (isFinite(cy) && cy > -1e4 && cy < 1e4) {

                        if (firstPoint) { ctx.moveTo(cx, cy); firstPoint = false; } 

                        else { ctx.lineTo(cx, cy); }

                    } else {

                        firstPoint = true;

                    }

                }

                ctx.stroke();

            };

            

            const drawPoint = (ctx, x, y, xRange, yRange, width, height, color, radius = 5) => {

                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const canvasX = (x - xMin) / (xMax - xMin) * width;

                const canvasY = height - (y - yMin) / (yMax - yMin) * height;


                if (isFinite(canvasX) && isFinite(canvasY)) {

                    ctx.fillStyle = color;

                    ctx.beginPath();

                    ctx.arc(canvasX, canvasY, radius, 0, 2 * Math.PI);

                    ctx.fill();

                }

            };


            const drawErrorLine = (ctx, x, y1, y2, xRange, yRange, width, height, color) => {

                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const toCanvasX = (val) => (val - xMin) / (xMax - xMin) * width;

                const toCanvasY = (val) => height - (val - yMin) / (yMax - yMin) * height;


                const canvasX = toCanvasX(x);

                const canvasY1 = toCanvasY(y1);

                const canvasY2 = toCanvasY(y2);


                if (isFinite(canvasX) && isFinite(canvasY1) && isFinite(canvasY2)) {

                    ctx.strokeStyle = color;

                    ctx.lineWidth = 1.5;

                    ctx.setLineDash([4, 4]);

                    ctx.beginPath();

                    ctx.moveTo(canvasX, canvasY1);

                    ctx.lineTo(canvasX, canvasY2);

                    ctx.stroke();

                    ctx.setLineDash([]);

                }

            };

                displayEl.innerHTML = formula;

            };

                const focusX_Y = sin.originalFunc(sin.FOCUS_X);

                

                const gw = sin.globalCanvas.clientWidth, gh = sin.globalCanvas.clientHeight; const gxRange = [-2 * Math.PI, 2 * Math.PI], gyRange = [-3, 3];

                drawGridAndAxes(sin.gCtx, gw, gh, gxRange, gyRange, false, sin.COLORS.axis, sin.COLORS.grid, sin.COLORS.text);

                drawFunction(sin.gCtx, gw, gh, gxRange, gyRange, sin.originalFunc, sin.COLORS.original, 3);

                drawFunction(sin.gCtx, gw, gh, gxRange, gyRange, (x) => sin.taylorApprox(x, degree, sin.CENTER_A), sin.COLORS.approx, 3);

                drawPoint(sin.gCtx, sin.CENTER_A, centerA_Y, gxRange, gyRange, gw, gh, sin.COLORS.approx);

                drawPoint(sin.gCtx, sin.FOCUS_X, focusX_Y, gxRange, gyRange, gw, gh, sin.COLORS.original);


                const zw = sin.zoomedCanvas.clientWidth, zh = sin.zoomedCanvas.clientHeight; const zoomFactor = Math.max(0.2, 1.5 - degree * 0.05); const zxRange = [sin.FOCUS_X - zoomFactor, sin.FOCUS_X + zoomFactor]; const zyRange = [focusX_Y - zoomFactor, focusX_Y + zoomFactor];

                drawGridAndAxes(sin.zCtx, zw, zh, zxRange, zyRange, true, sin.COLORS.axis, sin.COLORS.grid, sin.COLORS.text);

                drawFunction(sin.zCtx, zw, zh, zxRange, zyRange, sin.originalFunc, sin.COLORS.original, 3);

                drawFunction(sin.zCtx, zw, zh, zxRange, zyRange, (x) => sin.taylorApprox(x, degree, sin.CENTER_A), sin.COLORS.approx, 3);

                

                drawErrorLine(sin.zCtx, sin.FOCUS_X, focusX_Y, sin.approxValueAtFocus, zxRange, zyRange, zw, zh, sin.COLORS.text);

                drawPoint(sin.zCtx, sin.CENTER_A, centerA_Y, zxRange, zyRange, zw, zh, sin.COLORS.approx);

                drawPoint(sin.zCtx, sin.FOCUS_X, focusX_Y, zxRange, zyRange, zw, zh, sin.COLORS.original);

                drawPoint(sin.zCtx, sin.FOCUS_X, sin.approxValueAtFocus, zxRange, zyRange, zw, zh, sin.COLORS.approx);

            }


            // --- Tab and Initializer Logic ---

            function setupTabs() {

                const tabButtons = document.querySelectorAll('.tab-button');

                tabButtons.forEach(button => {

                    button.addEventListener('click', () => {

                        const targetPanel = document.querySelector(button.dataset.tabTarget);


                        // Deactivate all buttons and hide all panels

                        document.querySelectorAll('.tab-button').forEach(btn => btn.classList.remove('active-tab'));

                        document.querySelectorAll('.tab-panel').forEach(panel => panel.classList.add('hidden'));


                        // Activate the clicked button and show its panel

                        button.classList.add('active-tab');

                        targetPanel.classList.remove('hidden');


                        // Redraw canvases in the now-visible panel

                        handleResize();

                    });

                });

            }
