    <script>

        document.addEventListener('DOMContentLoaded', () => {

            // KaTeXによる数式描画

            const katexElements = [

                { id: 'func-f', expression: 'y = f(x) = 0.1x^3 + 0.5x' },

                { id: 'func-g', expression: 'x = f^{-1}(y)' },

                { id: 'explanation-f', expression: 'f(x)' },

                { id: 'explanation-g', expression: 'f^{-1}(y)' },

                { id: 'explanation-m2', expression: 'm_2 = 1 / m_1' },

                { id: 'formula1', expression: `\\frac{dx}{dy} = \\frac{1}{\\frac{dy}{dx}}` },

                { id: 'formula2', expression: `(f^{-1})'(y) = \\frac{1}{f'(x)}` },

            ];

            katexElements.forEach(item => {

                try {

                    katex.render(item.expression, document.getElementById(item.id), {

                        throwOnError: false

                    });

                } catch (e) {

                    console.error(`KaTeX rendering failed for ID ${item.id}:`, e);

                    document.getElementById(item.id).textContent = item.expression;

                }

            });


            const canvas = document.getElementById('graphCanvas');

            const ctx = canvas.getContext('2d');

            const slider = document.getElementById('xSlider');

            

            const pointPElement = document.getElementById('pointP');

            const slopePElement = document.getElementById('slopeP');

            const pointQElement = document.getElementById('pointQ');

            const slopeQElement = document.getElementById('slopeQ');

            const slopeRelationElement = document.getElementById('slope-relation');


            let scale = 50;

            let originX, originY;


            // 元の関数とその導関数

            const f = x => 0.1 * Math.pow(x, 3) + 0.5 * x;

            const f_prime = x => 0.3 * Math.pow(x, 2) + 0.5;


            function setupCanvas() {

                const container = canvas.parentElement;

                const size = Math.min(container.clientWidth, 600);

                canvas.width = size;

                canvas.height = size;

                originX = canvas.width * 0.1;

                originY = canvas.height * 0.9;

                scale = Math.min(canvas.width, canvas.height) / 10;

                draw();

            }


            function toCanvasX(x) { return originX + x * scale; }

            function toCanvasY(y) { return originY - y * scale; }

            function fromCanvasX(cx) { return (cx - originX) / scale; }


            function drawAxes() {

                ctx.beginPath();

                ctx.strokeStyle = '#9ca3af'; // gray-400

                ctx.lineWidth = 1;

                // X axis

                ctx.moveTo(0, originY);

                ctx.lineTo(canvas.width, originY);

                // Y axis

                ctx.moveTo(originX, 0);

                ctx.lineTo(originX, canvas.height);

                ctx.stroke();


                // Axis labels

                ctx.fillStyle = '#4b5563'; // gray-600

                ctx.font = '14px Inter';

                ctx.fillText('x', canvas.width - 15, originY - 10);

                ctx.fillText('y', originX + 10, 15);

                

                // Ticks

                for(let i = 1; i * scale < canvas.width - originX; i++) {

                    ctx.fillText(i, toCanvasX(i), originY + 15);

                    ctx.beginPath();

                    ctx.moveTo(toCanvasX(i), originY - 3);

                    ctx.lineTo(toCanvasX(i), originY + 3);

                    ctx.stroke();

                }

                for(let i = 1; i * scale < originY; i++) {

                    ctx.fillText(i, originX - 20, toCanvasY(i) + 5);

                    ctx.beginPath();

                    ctx.moveTo(originX - 3, toCanvasY(i));

                    ctx.lineTo(originX + 3, toCanvasY(i));

                    ctx.stroke();

                }

            }


            function drawFunction(func, color) {

                ctx.beginPath();

                ctx.strokeStyle = color;

                ctx.lineWidth = 3;

                let firstPoint = true;

                for (let cx = 0; cx < canvas.width; cx++) {

                    const x = fromCanvasX(cx);

                    if (x >= 0) {

                        const y = func(x);

                        const cy = toCanvasY(y);

                        if (firstPoint) {

                            ctx.moveTo(cx, cy);

                            firstPoint = false;

                        } else {

                            ctx.lineTo(cx, cy);

                        }

                    }

                }

                ctx.stroke();

            }


            function drawInverse(func, color) {

                ctx.beginPath();

                ctx.strokeStyle = color;

                ctx.lineWidth = 3;

                let firstPoint = true;

                for (let cx = 0; cx < canvas.width; cx++) {

                    const x = fromCanvasX(cx);

                     if (x >= 0) {

                        const y = func(x);

                        // Swap x and y to draw the inverse

                        const inv_cx = toCanvasX(y);

                        const inv_cy = toCanvasY(x);

                        if (inv_cx > originX) {

                            if (firstPoint) {

                                ctx.moveTo(inv_cx, inv_cy);

                                firstPoint = false;

                            } else {

                                ctx.lineTo(inv_cx, inv_cy);

                            }

                        }

                    }

                }

                ctx.stroke();

            }

            

            function drawLine(y_intercept, slope, color, dashed = false) {

                ctx.beginPath();

                ctx.strokeStyle = color;

                ctx.lineWidth = 2;

                if (dashed) {

                    ctx.setLineDash([5, 5]);

                }

                const x0 = 0;

                const y0 = slope * (x0 - fromCanvasX(0)) + y_intercept;

                const x1 = fromCanvasX(canvas.width);

                const y1 = slope * (x1 - fromCanvasX(0)) + y_intercept;


                ctx.moveTo(toCanvasX(x0), toCanvasY(y0));

                ctx.lineTo(toCanvasX(x1), toCanvasY(y1));

                ctx.stroke();

                ctx.setLineDash([]);

            }


            function drawTangent(px, py, slope, color) {

                ctx.beginPath();

                ctx.strokeStyle = color;

                ctx.lineWidth = 2;

                ctx.setLineDash([4, 4]);


                const length = 1.5;

                const angle = Math.atan(slope);

                const dx = length * Math.cos(angle);

                const dy = length * Math.sin(angle);


                ctx.moveTo(toCanvasX(px - dx), toCanvasY(py - dy));

                ctx.lineTo(toCanvasX(px + dx), toCanvasY(py + dy));

                

                ctx.stroke();

                ctx.setLineDash([]);

            }


            function drawPoint(x, y, color) {

                ctx.beginPath();

                ctx.fillStyle = color;

                ctx.arc(toCanvasX(x), toCanvasY(y), 6, 0, 2 * Math.PI);

                ctx.fill();

                ctx.strokeStyle = 'white';

                ctx.lineWidth = 2;

                ctx.stroke();

            }


            function draw() {

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                drawAxes();

                

                // Draw y = x

                ctx.save();

                ctx.beginPath();

                ctx.strokeStyle = '#9ca3af'; // gray-400

                ctx.lineWidth = 1.5;

                ctx.setLineDash([3, 3]);

                ctx.moveTo(0, canvas.height);

                ctx.lineTo(canvas.width, 0);

                ctx.stroke();

                ctx.restore();


                // Draw functions

                drawFunction(f, '#3b82f6'); // blue-500

                drawInverse(f, '#22c55e'); // green-500


                // Get current point

                const a = parseFloat(slider.value);

                const b = f(a);


                // Point P on f(x)

                const px = a;

                const py = b;

                const slope_p = f_prime(a);

                drawTangent(px, py, slope_p, '#60a5fa'); // blue-400

                drawPoint(px, py, '#1d4ed8'); // blue-700


                // Point Q on f^-1(y)

                const qx = b;

                const qy = a;

                const slope_q = 1 / slope_p;

                drawTangent(qx, qy, slope_q, '#4ade80'); // green-400

                drawPoint(qx, qy, '#15803d'); // green-700

                

                // Update info text

                pointPElement.textContent = `P(a, b) = (${px.toFixed(2)}, ${py.toFixed(2)})`;

                slopePElement.textContent = `傾き f'(a) = ${slope_p.toFixed(3)}`;

                pointQElement.textContent = `Q(b, a) = (${qx.toFixed(2)}, ${qy.toFixed(2)})`;

                slopeQElement.textContent = `傾き (f⁻¹)'(b) = ${slope_q.toFixed(3)}`;

                

                slopeRelationElement.textContent = `${slope_p.toFixed(3)} × ${slope_q.toFixed(3)} ≈ 1`;

            }


            slider.addEventListener('input', draw);

            window.addEventListener('resize', setupCanvas);

            

            setupCanvas();

        });
