document.addEventListener('DOMContentLoaded', (event) => {

    // --- Canvas and Visualization Setup ---

    const canvas = document.getElementById('heatmapCanvas');

    const ctx = canvas.getContext('2d');

    const marker = document.getElementById('selectionMarker');

    const colorbar = document.getElementById('colorbar');


    let width, height;

    let animationFrameId;

    let selectedPixel = null;

    let history = [];

    const HISTORY_LENGTH = 200;


    // --- Chart.js Setup ---

    const chartCanvas = document.getElementById('timeChart');

    const chartPlaceholder = document.getElementById('chart-placeholder');

    const chartDataDisplay = document.getElementById('chart-data');

    const selectedPointEl = document.getElementById('selectedPoint');

    const derivativeValueEl = document.getElementById('derivativeValue');

    let timeChart;


    function initChart() {

        if (timeChart) {

            timeChart.destroy();

        }

        const chartCtx = chartCanvas.getContext('2d');

        timeChart = new Chart(chartCtx, {

            type: 'line',

            data: {

                labels: [],

                datasets: [{

                    label: '温度',

                    data: [],

                    borderColor: 'rgb(56, 189, 248)',

                    backgroundColor: 'rgba(56, 189, 248, 0.2)',

                    borderWidth: 2,

                    tension: 0.3,

                    pointRadius: 0

                },

                {

                    label: '接線',

                    data: [],

                    borderColor: 'rgb(74, 222, 128)',

                    borderWidth: 1.5,

                    pointRadius: 0,

                    borderDash: [5, 5],

                }]

            },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                scales: {

                    x: {

                        display: false,

                    },

                    y: {

                        beginAtZero: false,

                        max: 1,

                        min: -1,

                        ticks: { color: '#9ca3af' },

                        grid: { color: '#374151' }

                    }

                },

                plugins: {

                    legend: {

                        display: false

                    }

                },

                animation: {

                    duration: 0

                }

            }

        });

    }


    // --- Simulation Logic ---

    // 空間と時間によって変化する温度を定義する関数

    function getTemperature(x, y, t) {

        // x, y は 0-1 の範囲

        // t は秒単位

        const term1 = Math.sin(x * 6 + t) * Math.cos(y * 6 - t) * 0.5;

        

        // 時間で移動する熱源

        const heatSourceX = 0.5 + 0.3 * Math.sin(t * 0.7);

        const heatSourceY = 0.5 + 0.3 * Math.cos(t * 1.1);

        const distSq = (x - heatSourceX)**2 + (y - heatSourceY)**2;

        const term2 = Math.exp(-distSq * 15) * (0.5 + 0.5 * Math.sin(t*2));


        return term1 + term2 - 0.5; // -1から1の範囲に正規化

    }

    

    // getTemperature の時間微分（解析的）

    function getTemperatureDerivative(x, y, t) {

        // term1'

        const term1_dt = (Math.cos(x * 6 + t) * Math.cos(y * 6 - t) + Math.sin(x * 6 + t) * Math.sin(y * 6 - t)) * 0.5;

        

        // term2' (product rule)

        const heatSourceX = 0.5 + 0.3 * Math.sin(t * 0.7);

        const heatSourceY = 0.5 + 0.3 * Math.cos(t * 1.1);

        const dx_dt = 0.3 * 0.7 * Math.cos(t*0.7);

        const dy_dt = -0.3 * 1.1 * Math.sin(t*1.1);

        const distSq = (x - heatSourceX)**2 + (y - heatSourceY)**2;

        const expTerm = Math.exp(-distSq * 15);

        

        const d_distSq_dt = -2 * (x - heatSourceX) * dx_dt - 2 * (y - heatSourceY) * dy_dt;

        const d_expTerm_dt = expTerm * (-15 * d_distSq_dt);

        

        const sinTerm = (0.5 + 0.5 * Math.sin(t*2));

        const d_sinTerm_dt = 0.5 * 2 * Math.cos(t*2);


        const term2_dt = d_expTerm_dt * sinTerm + expTerm * d_sinTerm_dt;


        return term1_dt + term2_dt;

    }



    function tempToColor(value) {

        // -1 (blue) to 1 (red)

        const r = 255 * Math.max(0, value);

        const b = 255 * Math.max(0, -value);

        const g = 255 - r - b;

        return `rgb(${r},${g},${b})`;

    }

    

    function drawColorbar() {

        const gradient = ctx.createLinearGradient(0, 0, colorbar.clientWidth, 0);

        for(let i=0; i<=10; i++){

            const value = (i/5.0) - 1.0;

            const color = tempToColor(value);

            gradient.addColorStop(i/10, color);

        }

        colorbar.style.background = `linear-gradient(to right, ${tempToColor(-1)}, ${tempToColor(0)}, ${tempToColor(1)})`;

    }


    function resizeCanvas() {

        const dpr = window.devicePixelRatio || 1;

        const rect = canvas.getBoundingClientRect();

        width = rect.width;

        height = rect.height;

        canvas.width = width * dpr;

        canvas.height = height * dpr;

        ctx.scale(dpr, dpr);

        if (selectedPixel) {

            updateMarkerPosition(selectedPixel.x, selectedPixel.y);

        }

    }

    

    function drawHeatmap(time) {

        const imageData = ctx.createImageData(canvas.width, canvas.height);

        const data = imageData.data;

        const t = time / 1000;


        for (let y = 0; y < canvas.height; y++) {

            for (let x = 0; x < canvas.width; x++) {

                const normX = x / canvas.width;

                const normY = y / canvas.height;

                const temp = getTemperature(normX, normY, t);

                const r = 255 * Math.max(0, temp);

                const b = 255 * Math.max(0, -temp);

                const g = 255 - r - b; // Simplified green component for speed


                const index = (y * canvas.width + x) * 4;

                data[index] = r;

                data[index + 1] = g;

                data[index + 2] = b;

                data[index + 3] = 255;

            }

        }

        ctx.putImageData(imageData, 0, 0);

    }

    

    function updateMarkerPosition(x, y) {

        marker.style.left = `${x * 100}%`;

        marker.style.top = `${y * 100}%`;

        marker.classList.remove('hidden');

    }


    function updateChart(time) {

        if (!selectedPixel || !timeChart) return;

        

        const t = time / 1000;

        const temp = getTemperature(selectedPixel.x, selectedPixel.y, t);


        history.push({t, temp});

        if (history.length > HISTORY_LENGTH) {

            history.shift();

        }


        const labels = history.map(p => p.t);

        const data = history.map(p => p.temp);

        

        const derivative = getTemperatureDerivative(selectedPixel.x, selectedPixel.y, t);

        derivativeValueEl.textContent = derivative.toFixed(3);


        // Update tangent line

        const tangentData = labels.map((label_t, index) => {

             // Show tangent only around the current point

             if (index > history.length - 20 && index < history.length) {

                 return temp + derivative * (label_t - t);

             }

             return null;

        });


        timeChart.data.labels = labels;

        timeChart.data.datasets[0].data = data;

        timeChart.data.datasets[1].data = tangentData;

        timeChart.update();

    }


    function animate(time) {

        drawHeatmap(time);

        if (selectedPixel) {

            updateChart(time);

        }

        animationFrameId = requestAnimationFrame(animate);

    }


    // --- Event Listeners ---

    canvas.addEventListener('click', (e) => {

        const rect = canvas.getBoundingClientRect();

        const x = (e.clientX - rect.left) / rect.width;

        const y = (e.clientY - rect.top) / rect.height;

        selectedPixel = { x, y };


        updateMarkerPosition(x, y);


        chartPlaceholder.classList.add('hidden');

        chartDataDisplay.classList.remove('hidden');

        selectedPointEl.textContent = `(${x.toFixed(2)}, ${y.toFixed(2)})`;


        // Reset and start chart

        history = [];

        initChart();

    });


    window.addEventListener('resize', resizeCanvas);


    // --- Initialization ---

    initChart();

    resizeCanvas();

    drawColorbar();

    animationFrameId = requestAnimationFrame(animate);

});
