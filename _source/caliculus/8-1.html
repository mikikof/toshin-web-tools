    <script type="module">

        import * as THREE from 'https://cdn.skypack.dev/three@0.132.2';

        import { OrbitControls } from 'https://cdn.skypack.dev/three@0.132.2/examples/jsm/controls/OrbitControls.js';


        let scene, camera, renderer, controls, gridHelper;

        let surfaceMesh, pointMarker;

        let tangentMeshX, tangentMeshY;

        let sliceTubeX, sliceTubeY;

        let slicePlaneX, slicePlaneY;


        const xSlider = document.getElementById('x-slider');

        const ySlider = document.getElementById('y-slider');

        const xValueSpan = document.getElementById('x-value');

        const yValueSpan = document.getElementById('y-value');

        const dfdxValueSpan = document.getElementById('dfdx-value');

        const dfdyValueSpan = document.getElementById('dfdy-value');

        const showXSlicePlaneCheckbox = document.getElementById('show-x-slice-plane');

        const showYSlicePlaneCheckbox = document.getElementById('show-y-slice-plane');


        const f = (x, y) => 1.5 - (x * x + y * y) / 4;

        const dfdx = (x, y) => -x / 2;

        const dfdy = (x, y) => -y / 2;


        function init() {

            const isDarkMode = () => document.documentElement.classList.contains('dark');


            scene = new THREE.Scene();

            scene.fog = new THREE.Fog(isDarkMode() ? 0x111827 : 0xf1f5f9, 10, 30);


            const container = document.getElementById('canvas-container');

            camera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);

            camera.position.set(4, 3, 4);


            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

            renderer.setPixelRatio(window.devicePixelRatio);

            renderer.setSize(container.clientWidth, container.clientHeight);

            renderer.setClearColor(isDarkMode() ? 0x111827 : 0xf1f5f9, 1);

            container.appendChild(renderer.domElement);

            

            const ambientLight = new THREE.AmbientLight(0xcccccc, 0.5);

            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);

            directionalLight.position.set(5, 10, 7.5);

            scene.add(directionalLight);


            controls = new OrbitControls(camera, renderer.domElement);

            controls.enableDamping = true;

            controls.dampingFactor = 0.05;


            createSurface();

            createThickAxes();

            createSlicePlanes();

            

            pointMarker = new THREE.Mesh(

                new THREE.SphereGeometry(0.08, 32, 32),

                new THREE.MeshStandardMaterial({ color: 0xffff00, metalness: 0.5, roughness: 0.2 })

            );

            scene.add(pointMarker);


            const tangentMaterialX = new THREE.MeshStandardMaterial({ color: 0x60a5fa, metalness: 0.3, roughness: 0.5 }); // Blue for X

            const tangentMaterialY = new THREE.MeshStandardMaterial({ color: 0xf87171, metalness: 0.3, roughness: 0.5 }); // Red for Y

            const tubeRadius = 0.02;


            tangentMeshX = new THREE.Mesh(new THREE.CylinderGeometry(tubeRadius, tubeRadius, 1, 8), tangentMaterialX);

            tangentMeshY = new THREE.Mesh(new THREE.CylinderGeometry(tubeRadius, tubeRadius, 1, 8), tangentMaterialY);

            scene.add(tangentMeshX, tangentMeshY);


            sliceTubeX = new THREE.Mesh(new THREE.BufferGeometry(), tangentMaterialX);

            sliceTubeY = new THREE.Mesh(new THREE.BufferGeometry(), tangentMaterialY);

            scene.add(sliceTubeX, sliceTubeY);


            setupEventListeners();

            updateVisualization();

            animate();

        }

        

        function createAxisLabel(text, position, color, background) {

            const canvas = document.createElement('canvas');

            const context = canvas.getContext('2d');

            const size = 256;

            canvas.width = size;

            canvas.height = size;


            context.font = `Bold 96px Arial`;

            context.textAlign = 'center';

            context.textBaseline = 'middle';

            

            context.fillStyle = background;

            context.arc(size / 2, size / 2, size / 2.2, 0, 2 * Math.PI);

            context.fill();


            context.fillStyle = color;

            context.fillText(text, size / 2, size / 2);


            const texture = new THREE.CanvasTexture(canvas);

            const material = new THREE.SpriteMaterial({ map: texture, depthTest: false });

            const sprite = new THREE.Sprite(material);

            sprite.scale.set(0.5, 0.5, 0.5);

            sprite.position.copy(position);

            scene.add(sprite);

        }


        function createGrid() {

            if(gridHelper){

                scene.remove(gridHelper);

                gridHelper.geometry.dispose();

                gridHelper.material.dispose();

            }

            const range = 3.5;

            const isDarkMode = document.documentElement.classList.contains('dark');

            const color1 = isDarkMode ? 0x4b5563 : 0x94a3b8;

            const color2 = isDarkMode ? 0x374151 : 0xe2e8f0;

            gridHelper = new THREE.GridHelper(range * 2, 10, color1, color2);

            scene.add(gridHelper);

        }


        function createSurface() {

            const range = 3.5;

            const segments = 50;

            const geometry = new THREE.PlaneGeometry(range * 2, range * 2, segments, segments);

            geometry.rotateX(-Math.PI / 2);

            

            const vertices = geometry.attributes.position.array;

            for (let i = 0; i < vertices.length; i += 3) {

                vertices[i+1] = f(vertices[i], vertices[i+2]);

            }

            geometry.computeVertexNormals();

            

            const material = new THREE.MeshStandardMaterial({

                color: 0x818cf8, side: THREE.DoubleSide, transparent: true,

                opacity: 0.7, metalness: 0.2, roughness: 0.6

            });

            surfaceMesh = new THREE.Mesh(geometry, material);

            scene.add(surfaceMesh);

            

            createGrid();

        }


        function createThickAxes() {

            const axisRadius = 0.015, axisLength = 4, arrowRadius = 0.05, arrowHeight = 0.2;

            const createAxis = (color, direction) => {

                const material = new THREE.MeshStandardMaterial({ color, metalness: 0.3, roughness: 0.5, depthTest: false });

                const group = new THREE.Group();

                const cylinder = new THREE.Mesh(new THREE.CylinderGeometry(axisRadius, axisRadius, axisLength, 12), material);

                const cone = new THREE.Mesh(new THREE.ConeGeometry(arrowRadius, arrowHeight, 12), material);

                group.add(cylinder, cone);

                cone.position.y = axisLength / 2;

                if (!direction.y) group.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), direction);

                scene.add(group);

            };

            createAxis(0x60a5fa, new THREE.Vector3(1, 0, 0)); // x-axis: Blue

            createAxis(0x4ade80, new THREE.Vector3(0, 1, 0)); // z-axis (height): Green

            createAxis(0xf87171, new THREE.Vector3(0, 0, 1)); // y-axis: Red


            const labelPos = axisLength / 2 + 0.6; // Increased distance

            createAxisLabel('x', new THREE.Vector3(labelPos, 0, 0), '#fff', '#60a5fa'); // Blue

            createAxisLabel('y', new THREE.Vector3(0, 0, labelPos), '#fff', '#f87171'); // Red

            createAxisLabel('z', new THREE.Vector3(0, labelPos, 0), '#fff', '#4ade80'); // Green

        }


        function createSlicePlanes() {

            const planeSize = 7;

            const planeGeom = new THREE.PlaneGeometry(planeSize, planeSize);

            slicePlaneX = new THREE.Mesh(planeGeom.clone(), new THREE.MeshBasicMaterial({ color: 0x60a5fa, side: THREE.DoubleSide, transparent: true, opacity: 0.2 })); // Blue for x-slice

            slicePlaneY = new THREE.Mesh(planeGeom.clone(), new THREE.MeshBasicMaterial({ color: 0xf87171, side: THREE.DoubleSide, transparent: true, opacity: 0.2 })); // Red for y-slice

            slicePlaneY.rotation.y = Math.PI / 2;

            scene.add(slicePlaneX, slicePlaneY);

        }


        function updateVisualization() {

            const x = parseFloat(xSlider.value), y = parseFloat(ySlider.value);

            const z_val = f(x, y);


            xValueSpan.textContent = x.toFixed(2);

            yValueSpan.textContent = y.toFixed(2);

            dfdxValueSpan.textContent = dfdx(x, y).toFixed(2);

            dfdyValueSpan.textContent = dfdy(x, y).toFixed(2);


            pointMarker.position.set(x, z_val, y);

            updateTangentLines(x, y, z_val);

            updateSliceCurves(x, y);


            slicePlaneX.visible = showXSlicePlaneCheckbox.checked;

            slicePlaneX.position.set(0, 0, y);

            slicePlaneY.visible = showYSlicePlaneCheckbox.checked;

            slicePlaneY.position.set(x, 0, 0);

        }

        

        function updateTangentLines(x, y, z) {

            const len = 2.0;

            const updateCylinder = (mesh, start, end) => {

                const dir = new THREE.Vector3().subVectors(end, start);

                mesh.position.copy(start).add(dir.clone().multiplyScalar(0.5));

                mesh.scale.y = dir.length();

                mesh.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), dir.normalize());

            };

            const slopeX = dfdx(x,y), dirX = new THREE.Vector3(1,slopeX,0).normalize();

            updateCylinder(tangentMeshX, new THREE.Vector3(x,z,y).sub(dirX.clone().multiplyScalar(len/2)), new THREE.Vector3(x,z,y).add(dirX.clone().multiplyScalar(len/2)));

            const slopeY = dfdy(x,y), dirY = new THREE.Vector3(0,slopeY,1).normalize();

            updateCylinder(tangentMeshY, new THREE.Vector3(x,z,y).sub(dirY.clone().multiplyScalar(len/2)), new THREE.Vector3(x,z,y).add(dirY.clone().multiplyScalar(len/2)));

        }

        

        function updateSliceCurves(x_val, y_val) {

            const range = 3.5, segments = 100, tubeRadius = 0.02;

            const createTube = (points) => new THREE.TubeGeometry(new THREE.CatmullRomCurve3(points), segments, tubeRadius, 8, false);

            

            let pointsX = [];

            for (let i = 0; i <= segments; i++) { const x = -range + (2 * range * i) / segments; pointsX.push(new THREE.Vector3(x, f(x, y_val), y_val)); }

            sliceTubeX.geometry.dispose();

            sliceTubeX.geometry = createTube(pointsX);


            let pointsY = [];

            for (let i = 0; i <= segments; i++) { const y = -range + (2 * range * i) / segments; pointsY.push(new THREE.Vector3(x_val, f(x_val, y), y)); }

            sliceTubeY.geometry.dispose();

            sliceTubeY.geometry = createTube(pointsY);

        }

        }
