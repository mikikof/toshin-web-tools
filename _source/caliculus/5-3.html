            <h2 class="text-2xl mb-4 text-slate-800">1. なぜ「誤差の評価」が必要なのか？</h2>

            <p class="mb-4">数学や物理では、複雑な関数をシンプルな多項式で近似することがよくあります。例えば、<span class="latex">sin(x)</span> は <span class="latex">x</span> が0に近いとき、<span class="latex">x</span> でとてもよく近似できます。</p>

            <div class="p-4 bg-slate-100 rounded-lg text-center text-xl md:text-2xl font-mono">

                <span class="latex">sin(x) &approx; x</span>

            </div>

            <p class="mt-4">しかし、この近似には必ず「誤差」が伴います。その誤差はどれくらい小さいのでしょうか？「すごく小さい」だけでは不十分です。誤差が<strong class="text-indigo-600">どれくらいの速さで0に近づくか</strong>を数学的に表現するために、ランダウの記号が使われます。</p>

        </div>


        <!-- Section 2: Interactive Demo -->

            <h2 class="text-2xl mb-4 text-slate-800">2. 「はるかに小さい」を体験する</h2>

            <p class="mb-4">ランダウ記号 <span class="latex">f(x) = o(g(x))</span> は、「<span class="latex">x</span> がある値に近づくとき、<span class="latex">f(x)</span> は <span class="latex">g(x)</span> よりも圧倒的に速く0に近づく」ことを意味します。ここでは <span class="latex">f(x) = x³</span> と <span class="latex">g(x) = x²</span> で、<span class="latex">x &rarr; 0</span> の場合を体験してみましょう。</p>

            

            <div class="mb-6">

        // --- Taylor Chart Script (Section 3) ---

        const taylorCanvas = document.getElementById('taylorChart');

        let taylorChart;

        const labels = Array.from({length: 201}, (_, i) => (i - 100) * Math.PI / 100);

        const sinData = labels.map(x => Math.sin(x));


        const datasets = {

            approx3: { label: '近似: x - x³/6', data: labels.map(x => x - Math.pow(x, 3) / 6), borderColor: 'rgb(168, 85, 247)', borderDash: [5, 5], borderWidth: 2, pointRadius: 0 },

            approx5: { label: '近似: x - x³/6 + x⁵/120', data: labels.map(x => x - Math.pow(x, 5) / 120), borderColor: 'rgb(236, 72, 153)', borderDash: [5, 5], borderWidth: 2, pointRadius: 0 }

        function createChart(approxOrder) {

            taylorChart = new Chart(taylorCanvas, {

        }

        // --- Pendulum Script (Section 4, Revised) ---

        const thetaSlider = document.getElementById('theta-slider');

        const thetaValue = document.getElementById('theta-value');

        const pendulumComment = document.getElementById('pendulum-comment');

        const pendulumCanvas = document.getElementById('pendulum-canvas');

        const p_ctx = pendulumCanvas.getContext('2d');

        const graphCanvas = document.getElementById('pendulum-graph');

        let pendulumGraph;


        let animationFrameId;

        let time = 0;

        const GRAPH_DURATION = 10; // The total time duration the graph will show (in seconds)


        class PendulumModel {

            constructor(initialTheta, useApprox) {

                this.theta = initialTheta * Math.PI / 180;

                this.omega = 0; // angular velocity

                this.useApprox = useApprox;

                this.g = 9.8 * 5; // Increased speed for better visualization

                this.length = 150;

                this.dt = 1 / 60;

            }


            update() {

                const alpha = this.useApprox ?

                    -(this.g / this.length) * this.theta :

                    -(this.g / this.length) * Math.sin(this.theta);

                this.omega += alpha * this.dt;

                this.omega *= 0.999; // A bit of damping

                this.theta += this.omega * this.dt;

            }

        }


        function drawPendulum(ctx, model, isGhost = false) {

            const w = ctx.canvas.width;

            const origin = { x: w / 2, y: 50 };

            const bobX = origin.x + model.length * Math.sin(model.theta);

            const bobY = origin.y + model.length * Math.cos(model.theta);


            ctx.globalAlpha = isGhost ? 0.4 : 1.0;


            // Rod

            ctx.beginPath();

            ctx.moveTo(origin.x, origin.y);

            ctx.lineTo(bobX, bobY);

            ctx.strokeStyle = '#64748b';

            ctx.lineWidth = 2;

            ctx.stroke();


            // Bob

            ctx.beginPath();

            ctx.arc(bobX, bobY, 15, 0, 2 * Math.PI);

            ctx.fillStyle = isGhost ? '#a855f7' : '#1d4ed8'; // Approx: purple, Exact: blue

            ctx.fill();


            ctx.globalAlpha = 1.0;

        }


        function setupPendulumGraph() {

            if (pendulumGraph) pendulumGraph.destroy();

            const initialThetaRad = parseFloat(thetaSlider.value) * Math.PI / 180;

            

            pendulumGraph = new Chart(graphCanvas, {

                type: 'line',

                data: {

                    datasets: [{

                        label: '正確な計算 sin(θ)',

                        borderColor: '#1d4ed8',

                        borderWidth: 2,

                        data: [],

                        pointRadius: 0,

                        tension: 0.2

                    }, {

                        label: '近似計算 θ',

                        borderColor: '#a855f7',

                        borderWidth: 2,

                        borderDash: [5, 5],

                        data: [],

                        pointRadius: 0,

                        tension: 0.2

                    }]

                },

                options: {

                    responsive: true,

                    scales: {

                        x: { 

                            type: 'linear', 

                            title: { display: true, text: '時間 (s)' },

                            min: 0,

                            max: GRAPH_DURATION // Fix the x-axis

                        },

                        y: { 

                            title: { display: true, text: '角度 (rad)' },

                            min: -initialThetaRad * 1.1, // Set y-axis based on initial angle

                            max: initialThetaRad * 1.1

                        }

                    },

                    animation: {

                        duration: 0 // No animation for drawing

                    },

                    plugins: {

                        legend: {

                            position: 'top',

                        },

                    }

                }

            });

        }


        let pendulumExact, pendulumApprox;


        function startPendulumAnimation() {

            if (animationFrameId) cancelAnimationFrame(animationFrameId);

            

            const initialTheta = parseFloat(thetaSlider.value);

            thetaValue.textContent = initialTheta;

            

            const containerWidth = pendulumCanvas.parentElement.clientWidth;

            pendulumCanvas.width = containerWidth;

            pendulumCanvas.height = containerWidth * 0.6;


            pendulumExact = new PendulumModel(initialTheta, false);

            pendulumApprox = new PendulumModel(initialTheta, true);


            if (initialTheta <= 20) {

                pendulumComment.textContent = "角度が小さいと、両者の動きと周期はほぼ一致しますね！";

                pendulumComment.className = "mt-6 text-center font-medium text-green-700";

            } else {

                pendulumComment.textContent = "角度が大きくなると、ゴーストとのズレが大きくなり、周期も変わってきます。";

                pendulumComment.className = "mt-6 text-center font-medium text-red-700";

            }

            

            time = 0;

            setupPendulumGraph();

            animatePendulums();

        }


        function animatePendulums() {

            pendulumExact.update();

            pendulumApprox.update();

            

            p_ctx.clearRect(0, 0, p_ctx.canvas.width, p_ctx.canvas.height);

            drawPendulum(p_ctx, pendulumApprox, true); // Draw ghost first

            drawPendulum(p_ctx, pendulumExact, false); // Draw solid on top


            // Update graph data only within the defined duration

            time += pendulumExact.dt;

            if (time <= GRAPH_DURATION) {

                const datasets = pendulumGraph.data.datasets;

                datasets[0].data.push({x: time, y: pendulumExact.theta});

                datasets[1].data.push({x: time, y: pendulumApprox.theta});

                pendulumGraph.update('none'); // Update without any animation

            }


            animationFrameId = requestAnimationFrame(animatePendulums);

        }


        thetaSlider.addEventListener('input', startPendulumAnimation);

        

        const resizeObserver = new ResizeObserver(entries => {

            // Use a timeout to avoid rapid re-renders during resize

            if (this.resizeTimeout) clearTimeout(this.resizeTimeout);

            this.resizeTimeout = setTimeout(startPendulumAnimation, 100);

        });

        resizeObserver.observe(pendulumCanvas.parentElement);


        window.addEventListener('load', startPendulumAnimation);
