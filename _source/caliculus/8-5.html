        // --- Visualization 1: Gradient Descent ---

        function setupGradientDescent() {

            const canvas = document.getElementById('grad-canvas');

            const ctx = canvas.getContext('2d');

            const resetButton = document.getElementById('grad-reset');

            const w = canvas.width;

            const h = canvas.height;


            let animationId;

            let path = [];

            let pos = { x: w * 0.2, y: h * 0.2 };

            let running = false;

            

            // The loss function: f(x, y) = a*x^2 + b*y^2

            const lossFn = (x, y) => {

                const normX = (x - w / 2) / (w / 2);

                const normY = (y - h / 2) / (h / 2);

                return 0.5 * normX**2 + 1 * normY**2;

            };

            

            // The gradient of the loss function

            const gradient = (x, y) => {

                const normX = (x - w / 2) / (w / 2);

                const normY = (y - h / 2) / (h / 2);

                return {

                    dx: (1 / (w / 2)) * normX,

                    dy: (2 / (h / 2)) * normY

                };

            };

            

            const drawBackground = () => {

                const imageData = ctx.createImageData(w, h);

                let maxVal = -Infinity;

                let minVal = Infinity;

                const values = [];


                for (let i = 0; i < w; i++) {

                    for (let j = 0; j < h; j++) {

                        const val = lossFn(i, j);

                        values.push(val);

                        if (val > maxVal) maxVal = val;

                        if (val < minVal) minVal = val;

                    }

                }

                

                for (let i = 0; i < values.length; i++) {

                    const normalized = (values[i] - minVal) / (maxVal - minVal);

                    const intensity = 255 * (1 - Math.pow(normalized, 0.5));

                    const [r,g,b] = [31 + intensity * 0.2, 41 + intensity * 0.3, 55 + intensity*0.5];

                    

                    imageData.data[i * 4] = r;

                    imageData.data[i * 4 + 1] = g;

                    imageData.data[i * 4 + 2] = b;

                    imageData.data[i * 4 + 3] = 255;

                }

                ctx.putImageData(imageData, 0, 0);

            };


            const drawPath = () => {

                if (path.length < 2) return;

                ctx.beginPath();

                ctx.moveTo(path[0].x, path[0].y);

                for(let i=1; i<path.length; i++) {

                    ctx.lineTo(path[i].x, path[i].y);

                }

                ctx.strokeStyle = 'rgba(255, 255, 255, 0.5)';

                ctx.lineWidth = 2;

                ctx.stroke();

            };


            const drawPoint = () => {

                ctx.beginPath();

                ctx.arc(pos.x, pos.y, 5, 0, Math.PI * 2);

                ctx.fillStyle = '#a78bfa'; // violet-400

                ctx.fill();

                ctx.strokeStyle = '#fff';

                ctx.lineWidth = 2;

                ctx.stroke();

            };


            const animate = () => {

                const grad = gradient(pos.x, pos.y);

                const learningRate = 50;

                

                pos.x -= learningRate * grad.dx;

                pos.y -= learningRate * grad.dy;

                path.push({x: pos.x, y: pos.y});


                draw();


                if (Math.abs(grad.dx) + Math.abs(grad.dy) > 0.0001 && path.length < 200) {

                    animationId = requestAnimationFrame(animate);

                } else {

                    running = false;

                }

            };

            

            const draw = () => {

                ctx.clearRect(0, 0, w, h);

                drawBackground();

                drawPath();

                drawPoint();

            };


            const start = (event) => {

                if (running) return;

                running = true;

                const rect = canvas.getBoundingClientRect();

                pos = { x: event.clientX - rect.left, y: event.clientY - rect.top };

                path = [{x: pos.x, y: pos.y}];

                cancelAnimationFrame(animationId);

                animate();

            };

            

            const reset = () => {

                running = false;

                cancelAnimationFrame(animationId);

                path = [];

                pos = { x: w * 0.2, y: h * 0.2 };

                draw();

            }


            canvas.addEventListener('click', start);

            resetButton.addEventListener('click', reset);

            reset();

        }


        // --- Visualization 2: Edge Detection ---

        function setupEdgeDetection() {

            const originalCanvas = document.getElementById('edge-original-canvas');

            const octx = originalCanvas.getContext('2d');

            const detectCanvas = document.getElementById('edge-detect-canvas');

            const dctx = detectCanvas.getContext('2d');

            const w = originalCanvas.width;

            const h = originalCanvas.height;


            const drawOriginal = () => {

                octx.fillStyle = '#4b5563'; // bg-gray-600

                octx.fillRect(0, 0, w, h);

                octx.fillStyle = '#e5e7eb'; // text-gray-200

                octx.fillRect(40, 40, 80, 80);

                octx.beginPath();

                octx.arc(140, 140, 40, 0, Math.PI * 2);

                octx.fill();

            };

            

            const sobel = (pixels) => {

                const grayscale = new Uint8ClampedArray(w * h);

                for (let i = 0; i < pixels.data.length; i += 4) {

                    const lightness = (pixels.data[i] + pixels.data[i+1] + pixels.data[i+2]) / 3;

                    grayscale[i/4] = lightness;

                }

                

                const output = new Uint8ClampedArray(pixels.data.length);

                const kx = [[-1, 0, 1], [-2, 0, 2], [-1, 0, 1]];

                const ky = [[-1, -2, -1], [0, 0, 0], [1, 2, 1]];

                

                for (let y = 1; y < h - 1; y++) {

                    for (let x = 1; x < w - 1; x++) {

                        let Gx = 0, Gy = 0;

                        for (let i = -1; i <= 1; i++) {

                            for (let j = -1; j <= 1; j++) {

                                const val = grayscale[(y + j) * w + (x + i)];

                                Gx += val * kx[j + 1][i + 1];

                                Gy += val * ky[j + 1][i + 1];

                            }

                        }

                        const magnitude = Math.sqrt(Gx * Gx + Gy * Gy);

                        const idx = (y * w + x) * 4;

                        output[idx] = output[idx+1] = output[idx+2] = magnitude;

                        output[idx+3] = 255;

                    }

                }

                return new ImageData(output, w, h);

            };

            

            drawOriginal();

            const originalImageData = octx.getImageData(0, 0, w, h);

            const edgeImageData = sobel(originalImageData);

            

            originalCanvas.addEventListener('mousemove', (e) => {

                const rect = originalCanvas.getBoundingClientRect();

                const x = e.clientX - rect.left;

                const y = e.clientY - rect.top;

                

                dctx.clearRect(0,0,w,h);

                dctx.globalAlpha = 0.5;

                dctx.putImageData(edgeImageData, 0, 0);

                dctx.globalAlpha = 1.0;


                const radius = 30;

                dctx.save();

                dctx.beginPath();

                dctx.arc(x, y, radius, 0, Math.PI*2);

                dctx.clip();

                dctx.putImageData(edgeImageData, 0, 0);

                dctx.restore();

            });

            

             originalCanvas.addEventListener('mouseleave', () => {

                dctx.clearRect(0,0,w,h);

            });

        }


        // --- Visualization 3: Logistic Regression ---

        function setupLogisticRegression() {

            const canvas = document.getElementById('logistic-canvas');

            const ctx = canvas.getContext('2d');

            const resetButton = document.getElementById('logistic-reset');

            const w = canvas.width;

            const h = canvas.height;


            let points = [];

            let weights = { w1: 0.1, w2: -0.1, b: 0 }; // w1*x + w2*y + b = 0

            

            const sigmoid = (z) => 1 / (1 + Math.exp(-z));


            const drawBackground = () => {

                 const imageData = ctx.createImageData(w, h);

                 for (let x = 0; x < w; x++) {

                     for (let y = 0; y < h; y++) {

                        const score = weights.w1 * (x - w/2) + weights.w2 * (y - h/2) + weights.b;

                        const prob = sigmoid(score * 0.05); // scale score

                        const idx = (y * w + x) * 4;

                        imageData.data[idx] = 251 * (1 - prob) + 129 * prob; // amber-300 to blue-400

                        imageData.data[idx+1] = 191 * (1 - prob) + 140 * prob;

                        imageData.data[idx+2] = 36 * (1 - prob) + 247 * prob;

                        imageData.data[idx+3] = 255;

                     }

                 }

                 ctx.putImageData(imageData, 0, 0);

            };


            const drawPoints = () => {

                points.forEach(p => {

                    ctx.beginPath();

                    ctx.arc(p.x, p.y, 5, 0, Math.PI * 2);

                    ctx.fillStyle = p.label === 1 ? '#60a5fa' : '#fcd34d'; // blue-400 : amber-300

                    ctx.fill();

                    ctx.strokeStyle = '#1f2937';

                    ctx.lineWidth = 1;

                    ctx.stroke();

                });

            };


            const drawBoundary = () => {

                ctx.beginPath();

                // w1*x + w2*y + b = 0 => y = (-w1*x - b) / w2

                const y1 = (-weights.w1 * (0 - w/2) - weights.b) / weights.w2 + h/2;

                const y2 = (-weights.w1 * (w - w/2) - weights.b) / weights.w2 + h/2;

                ctx.moveTo(0, y1);

                ctx.lineTo(w, y2);

                ctx.strokeStyle = 'rgba(255, 255, 255, 0.8)';

                ctx.lineWidth = 2;

                ctx.stroke();

            };

            

            const updateWeights = (point) => {

                const learningRate = 0.1;

                const score = weights.w1 * (point.x - w/2) + weights.w2 * (point.y - h/2) + weights.b;

                const prediction = sigmoid(score * 0.05);

                const error = point.label - prediction;


                weights.w1 += learningRate * error * (point.x-w/2) * 0.01;

                weights.w2 += learningRate * error * (point.y-h/2) * 0.01;

                weights.b += learningRate * error;

            };


            const addPoint = (e) => {

                const rect = canvas.getBoundingClientRect();

                const x = e.clientX - rect.left;

                const y = e.clientY - rect.top;

                

                const score = weights.w1 * (x - w/2) + weights.w2 * (y - h/2) + weights.b;

                const label = score > 0 ? 1 : 0;

                

                const newPoint = { x, y, label };

                points.push(newPoint);

                updateWeights(newPoint);

                draw();

            };

            

            const draw = () => {

                ctx.clearRect(0,0,w,h);

                drawBackground();

                drawPoints();

                // drawBoundary(); // Boundary is implicit in background

            };

            

            const reset = () => {

                points = [];

                weights = { w1: 0.1, w2: -0.1, b: 0 };

                // Add some initial points for clarity

                for(let i=0; i<10; i++){

                    points.push({ x: w*0.25 + Math.random()*50-25, y: h*0.5 + Math.random()*100-50, label: 0 });

                    points.push({ x: w*0.75 + Math.random()*50-25, y: h*0.5 + Math.random()*100-50, label: 1 });

                }

                draw();

            };

            

            canvas.addEventListener('click', addPoint);

            resetButton.addEventListener('click', reset);

            reset();

        }


        // --- Visualization 4: Diffusion / Noise Removal ---

        function setupDiffusion() {

            const originalCanvas = document.getElementById('noise-original-canvas');

            const octx = originalCanvas.getContext('2d', { willReadFrequently: true });

            const denoisedCanvas = document.getElementById('noise-denoised-canvas');

            const dctx = denoisedCanvas.getContext('2d');

            const slider = document.getElementById('diffusion-slider');

            const w = originalCanvas.width;

            const h = originalCanvas.height;


            let noisyImageData;

            

            const generateBaseImage = (ctx) => {

                ctx.fillStyle = '#4b5563'; // bg-gray-600

                ctx.fillRect(0, 0, w, h);

                ctx.fillStyle = '#e5e7eb'; // text-gray-200

                ctx.fillRect(40, 40, 80, 80);

                ctx.beginPath();

                ctx.arc(140, 140, 40, 0, Math.PI * 2);

                ctx.fill();

            };


            const addNoise = (imageData) => {

                const d = imageData.data;

                for (let i = 0; i < d.length; i += 4) {

                    if (Math.random() < 0.15) { // 15% noise

                        const color = Math.random() > 0.5 ? 255 : 0;

                        d[i] = d[i+1] = d[i+2] = color;

                    }

                }

                return imageData;

            };


            const applyBlur = (srcData, iterations) => {

                let src = new Uint8ClampedArray(srcData.data);

                let dst = new Uint8ClampedArray(srcData.data.length);

                

                for(let iter=0; iter<iterations; iter++) {

                    for(let y=1; y<h-1; y++) {

                        for(let x=1; x<w-1; x++) {

                            for (let c=0; c<3; c++) { // R, G, B channels

                                let sum = 0;

                                for(let j=-1; j<=1; j++) {

                                    for(let i=-1; i<=1; i++) {

                                        sum += src[((y+j) * w + (x+i))*4 + c];

                                    }

                                }

                                dst[((y*w+x)*4)+c] = sum / 9;

                            }

                           dst[((y*w+x)*4)+3] = 255; // Alpha

                        }

                    }

                    src.set(dst);

                }

                return new ImageData(dst, w, h);

            };


            const init = () => {

                generateBaseImage(octx);

                const baseImageData = octx.getImageData(0, 0, w, h);

                noisyImageData = addNoise(baseImageData);

                octx.putImageData(noisyImageData, 0, 0);

                updateDenoised();

            };

            

            const updateDenoised = () => {

                const iterations = parseInt(slider.value);

                if (iterations === 0) {

                    dctx.clearRect(0,0,w,h);

                    dctx.putImageData(noisyImageData, 0,0);

                    return;

                }

                const denoisedData = applyBlur(noisyImageData, iterations);

                dctx.putImageData(denoisedData, 0, 0);

            };


            slider.addEventListener('input', updateDenoised);

            init();

        }


    </script>
