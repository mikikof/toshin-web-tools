    function drawLossChart(){

      const padL=36, padR=12, padT=10, padB=20;

      const w = lossSvg.clientWidth || 760; const h = lossSvg.clientHeight || 280;

      const lMin = data.length? Math.min(...data.map(d=>d.loss)) : 0;

      let svg = '';

    function stepOnce(){

      const g = grad(state.theta.x, state.theta.y);

      const gn = { x: g.x + state.noise*(Math.random()-0.5), y: g.y + state.noise*(Math.random()-0.5) };

      let next = { x: state.theta.x, y: state.theta.y };

      if(state.algo==='SGD'){

        next.x -= state.eta*gn.x; next.y -= state.eta*gn.y;

        if(state.weightDecay>0){ next.x -= state.eta*state.weightDecay*state.theta.x; next.y -= state.eta*state.weightDecay*state.theta.y; }

      }else{

        state.m.x = state.beta1*state.m.x + (1-state.beta1)*gn.x; state.m.y = state.beta1*state.m.y + (1-state.beta1)*gn.y;

        state.v.x = state.beta2*state.v.x + (1-state.beta2)*(gn.x*gn.x); state.v.y = state.beta2*state.v.y + (1-state.beta2)*(gn.y*gn.y);

        state.t += 1; const t=state.t;

        const mHat = { x: state.m.x/(1-Math.pow(state.beta1,t)), y: state.m.y/(1-Math.pow(state.beta1,t)) };

        const vHat = { x: state.v.x/(1-Math.pow(state.beta2,t)), y: state.v.y/(1-Math.pow(state.beta2,t)) };

        const upd  = { x: mHat.x/(Math.sqrt(vHat.x)+state.eps), y: mHat.y/(Math.sqrt(vHat.y)+state.eps) };

        if(state.algo==='AdamW' && state.weightDecay>0){ next.x -= state.eta*(upd.x + state.weightDecay*state.theta.x); next.y -= state.eta*(upd.y + state.weightDecay*state.theta.y); }

        else{ next.x -= state.eta*upd.x; next.y -= state.eta*upd.y; }

      }

      state.theta = next;

      state.path.push(next); if(state.path.length>700) state.path.shift();

      state.step += 1;

      state.lossData.push({ step: state.step, loss: loss(next.x, next.y) }); if(state.lossData.length>800) state.lossData.shift();

    }

    function numericGrad(x,y,h=1e-4){ const L=loss(x,y); return { x:(loss(x+h,y)-L)/h, y:(loss(x,y+h)-L)/h }; }

    function runTests(){

      const out=[]; let pass=true;

      // T1: grad

      const pts=[{x:1,y:1},{x:2,y:-1},{x:-3,y:0.5}]; let t1ok=true, maxRel=0, worst=null; for(const p of pts){ const ga=grad(p.x,p.y), gn=numericGrad(p.x,p.y); const diff=Math.hypot(ga.x-gn.x, ga.y-gn.y); const denom=Math.max(1e-9,Math.hypot(ga.x,ga.y)); const rel=diff/denom; if(rel>maxRel){maxRel=rel; worst={p,rel}} if(rel>1e-3) t1ok=false; }

      out.push({name:'T1: grad 一致（数値微分 vs 解析）', ok:t1ok, detail:`maxRelErr=${maxRel.toExponential(2)} at ${JSON.stringify(worst&&worst.p)}`}); pass=pass&&t1ok;

      // T2: SGD

      const p0={x:2,y:-1}; const g0=grad(p0.x,p0.y); const p1={x:p0.x-0.01*g0.x, y:p0.y-0.01*g0.y}; const t2ok=loss(p1.x,p1.y)<loss(p0.x,p0.y); out.push({name:'T2: SGD で損失降下', ok:t2ok, detail:`L0=${loss(p0.x,p0.y).toFixed(4)}→L1=${loss(p1.x,p1.y).toFixed(4)}`}); pass=pass&&t2ok;

      // T3: Adam g=0

      const z={x:0,y:0}; let m={x:0,y:0}, v={x:0,y:0}, t=0; const gn={x:0,y:0}; function adamPreview(prev, gn, m, v, t, hp){ const {beta1,beta2,eps,eta,weightDecay,algo}=hp; const m2={x:beta1*m.x+(1-beta1)*gn.x, y:beta1*m.y+(1-beta1)*gn.y}; const v2={x:beta2*v.x+(1-beta2)*(gn.x*gn.x), y:beta2*v.y+(1-beta2)*(gn.y*gn.y)}; const t2=t+1; const mHat={x:m2.x/(1-Math.pow(beta1,t2)), y:m2.y/(1-Math.pow(beta1,t2))}; const vHat={x:v2.x/(1-Math.pow(beta2,t2)), y:v2.y/(1-Math.pow(beta2,t2))}; const upd={x:mHat.x/(Math.sqrt(vHat.x)+eps), y:mHat.y/(Math.sqrt(vHat.y)+eps)}; let next={...prev}; if(algo==='AdamW'&&weightDecay>0){ next.x-=eta*(upd.x+weightDecay*prev.x); next.y-=eta*(upd.y+weightDecay*prev.y);} else { next.x-=eta*upd.x; next.y-=eta*upd.y; } return {next,m:m2,v:v2,t:t2}; }

      const a3=adamPreview(z,gn,{x:0,y:0},{x:0,y:0},0,{beta1:0.9,beta2:0.999,eps:1e-8,eta:0.01,weightDecay:0,algo:'Adam'}); const t3ok = Math.hypot(a3.next.x-z.x, a3.next.y-z.y)<1e-12; out.push({name:'T3: Adam 勾配ゼロで不動 (λ=0)', ok:t3ok, detail:`|Δθ|=${Math.hypot(a3.next.x-z.x, a3.next.y-z.y).toExponential(2)}`}); pass=pass&&t3ok;

      // T4: AdamW decay

      const z2={x:1,y:-0.5}; const a4=adamPreview(z2,gn,{x:0,y:0},{x:0,y:0},0,{beta1:0.9,beta2:0.999,eps:1e-8,eta:0.01,weightDecay:0.05,algo:'AdamW'}); const t4ok = Math.hypot(a4.next.x, a4.next.y) < Math.hypot(z2.x, z2.y); out.push({name:'T4: AdamW で重み減衰が原点へ寄せる', ok:t4ok, detail:`|θ'| < |θ| ? ${t4ok}`}); pass=pass&&t4ok;

      // render

      const ul = document.getElementById('testList'); ul.innerHTML = out.map(t=>`<li class="${t.ok?'pass':'fail'}">${t.ok?'✔':'✖'} ${t.name} — <span class="muted">${t.detail}</span></li>`).join('');

      return pass;

    }
