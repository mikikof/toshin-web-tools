    <script>

        document.addEventListener('DOMContentLoaded', () => {

            // --- Shared Functions ---

            const factorial = (n) => {

                if (n < 0) return Infinity;

                if (n === 0) return 1;

                let result = 1;

                for (let i = 2; i <= n; i++) result *= i;

                return result;

            };


            const setupCanvas = (canvas, ctx) => {

                const dpr = window.devicePixelRatio || 1;

                const rect = canvas.parentElement.getBoundingClientRect();

                canvas.width = rect.width * dpr;

                canvas.height = rect.height * dpr;

                canvas.style.width = `${rect.width}px`;

                canvas.style.height = `${rect.height}px`;

                ctx.scale(dpr, dpr);

            };


            const drawGridAndAxes = (ctx, width, height, xRange, yRange, showDetails = true, axisColor, gridColor, textColor) => {

                ctx.clearRect(0, 0, width, height);

                ctx.fillStyle = '#FFFFFF';

                ctx.fillRect(0, 0, width, height);


                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const toCanvasX = (x) => (x - xMin) / (xMax - xMin) * width;

                const toCanvasY = (y) => height - (y - yMin) / (yMax - yMin) * height;


                if (showDetails) {

                    ctx.strokeStyle = gridColor;

                    ctx.lineWidth = 1;

                    ctx.beginPath();

                    for (let i = Math.ceil(xMin); i <= Math.floor(xMax); i++) {

                        const cx = toCanvasX(i); ctx.moveTo(cx, 0); ctx.lineTo(cx, height);

                    }

                    for (let i = Math.ceil(yMin * 2) / 2; i <= Math.floor(yMax * 2) / 2; i+=0.5) {

                        const cy = toCanvasY(i); ctx.moveTo(0, cy); ctx.lineTo(width, cy);

                    }

                    ctx.stroke();

                }


                ctx.strokeStyle = axisColor;

                ctx.lineWidth = showDetails ? 2 : 1.5;

                ctx.beginPath();

                const yZero = toCanvasY(0);

                if (yZero >= 0 && yZero <= height) { ctx.moveTo(0, yZero); ctx.lineTo(width, yZero); }

                const xZero = toCanvasX(0);

                if (xZero >= 0 && xZero <= width) { ctx.moveTo(xZero, 0); ctx.lineTo(xZero, height); }

                ctx.stroke();


                if (showDetails) {

                    ctx.fillStyle = textColor;

                    ctx.font = '12px sans-serif';

                    ctx.textAlign = 'center'; ctx.textBaseline = 'top';

                    for (let i = Math.ceil(xMin); i <= Math.floor(xMax); i++) {

                        if (i !== 0) ctx.fillText(i, toCanvasX(i), yZero + 5);

                    }

                    ctx.textAlign = 'right'; ctx.textBaseline = 'middle';

                    for (let i = Math.ceil(yMin * 2) / 2; i <= Math.floor(yMax * 2) / 2; i+=0.5) {

                        if (i !== 0) ctx.fillText(i, xZero - 5, toCanvasY(i));

                    }

                }

            };


            const drawFunction = (ctx, width, height, xRange, yRange, func, color, lineWidth = 2) => {

                const [xMin, xMax] = xRange;

                const [yMin, yMax] = yRange;

                const toCanvasX = (x) => (x - xMin) / (xMax - xMin) * width;

                const toCanvasY = (y) => height - (y - yMin) / (yMax - yMin) * height;

                

                ctx.strokeStyle = color;

                ctx.lineWidth = lineWidth;

                ctx.beginPath();

                let firstPoint = true;

                for (let px = 0; px <= width; px++) {

                    const x = xMin + (px / width) * (xMax - xMin);

                    const y = func(x);

                    const cx = toCanvasX(x);

                    const cy = toCanvasY(y);

                    if (isFinite(cy) && cy > -1e4 && cy < 1e4) {

                        if (firstPoint) { ctx.moveTo(cx, cy); firstPoint = false; } 

                        else { ctx.lineTo(cx, cy); }

                    } else {

                        firstPoint = true;

                    }

                }

                ctx.stroke();

            };

            

            const renderFormula = (displayEl, formula) => {

                displayEl.innerHTML = `\\( ${formula} \\)`;

                if (window.renderMathInElement) {

                    try { renderMathInElement(displayEl); } catch(e) { console.error("KaTeX failed", e); }

                }

            };


            // --- e^x Setup ---

            const ex = {

                slider: document.getElementById('ex-degree-slider'), degreeValue: document.getElementById('ex-degree-value'),

                focusSlider: document.getElementById('ex-focus-slider'), focusValue: document.getElementById('ex-focus-value'),

                formulaDisplay: document.getElementById('ex-formula-display'), trueValueDisplay: document.getElementById('ex-true-value'),

                approxValueDisplay: document.getElementById('ex-approx-value'), approxDegreeDisplay: document.getElementById('ex-approx-degree'),

                errorValueDisplay: document.getElementById('ex-error-value'), globalCanvas: document.getElementById('ex-global-canvas'),

                zoomedCanvas: document.getElementById('ex-zoomed-canvas'), zoomTitle: document.getElementById('ex-zoom-title'),

                trueValueLabel: document.getElementById('ex-true-value-label'),

                FOCUS_X: 2.0, COLORS: { axis: '#94a3b8', grid: '#e2e8f0', original: '#3b82f6', approx: '#ef4444', text: '#1e293b' },

                originalFunc: (x) => Math.exp(x),

                maclaurinApprox: (x, n) => { let sum = 0; for (let i = 0; i <= n; i++) sum += Math.pow(x, i) / factorial(i); return sum; }

            };

            ex.gCtx = ex.globalCanvas.getContext('2d'); ex.zCtx = ex.zoomedCanvas.getContext('2d');


            function updateEx() {

                const degree = parseInt(ex.slider.value);

                ex.FOCUS_X = parseFloat(ex.focusSlider.value);

                const TRUE_VALUE = ex.originalFunc(ex.FOCUS_X);

                

                ex.degreeValue.textContent = degree; ex.approxDegreeDisplay.textContent = degree;

                ex.focusValue.textContent = ex.FOCUS_X.toFixed(1);

                ex.zoomTitle.textContent = `x = ${ex.FOCUS_X.toFixed(1)} での拡大図`;

                ex.trueValueLabel.innerHTML = `実際の値 (e<sup>${ex.FOCUS_X.toFixed(1)}</sup>)`;

                

                let formula = 'P_n(x) = 1';

                if (degree > 0) formula += ' + x';

                for (let i = 2; i <= Math.min(degree, 5); i++) formula += ` + \\frac{x^${i}}{${i}!}`;

                if (degree > 5) formula += ` + \\dots + \\frac{x^${degree}}{${degree}!}`;

                renderFormula(ex.formulaDisplay, formula);

                

                const approxValue = ex.maclaurinApprox(ex.FOCUS_X, degree);

                ex.trueValueDisplay.textContent = TRUE_VALUE.toFixed(8);

                ex.approxValueDisplay.textContent = approxValue.toFixed(8);

                ex.errorValueDisplay.textContent = Math.abs(TRUE_VALUE - approxValue).toExponential(4);

                drawEx();

            }


            function drawEx() {

                const degree = parseInt(ex.slider.value);

                const TRUE_VALUE = ex.originalFunc(ex.FOCUS_X);

                const gw = ex.globalCanvas.clientWidth, gh = ex.globalCanvas.clientHeight; const gxRange = [-6, 6], gyRange = [-50, 450];

                drawGridAndAxes(ex.gCtx, gw, gh, gxRange, gyRange, false, ex.COLORS.axis, ex.COLORS.grid, ex.COLORS.text);

                drawFunction(ex.gCtx, gw, gh, gxRange, gyRange, ex.originalFunc, ex.COLORS.original, 3);

                drawFunction(ex.gCtx, gw, gh, gxRange, gyRange, (x) => ex.maclaurinApprox(x, degree), ex.COLORS.approx, 3);

                const focusCanvasX = (ex.FOCUS_X - gxRange[0]) / (gxRange[1] - gxRange[0]) * gw; const focusCanvasY = gh - (TRUE_VALUE - gyRange[0]) / (gyRange[1] - gyRange[0]) * gh;

                ex.gCtx.fillStyle = ex.COLORS.approx; ex.gCtx.beginPath(); ex.gCtx.arc(focusCanvasX, focusCanvasY, 5, 0, 2 * Math.PI); ex.gCtx.fill();

                

                const zw = ex.zoomedCanvas.clientWidth, zh = ex.zoomedCanvas.clientHeight; const zoomFactor = Math.max(0.1, 2 - degree * 0.1); const zxRange = [ex.FOCUS_X - zoomFactor, ex.FOCUS_X + zoomFactor]; const zyRange = [TRUE_VALUE - zoomFactor, TRUE_VALUE + zoomFactor];

                drawGridAndAxes(ex.zCtx, zw, zh, zxRange, zyRange, true, ex.COLORS.axis, ex.COLORS.grid, ex.COLORS.text);

            const ln = {

                slider: document.getElementById('ln-degree-slider'), degreeValue: document.getElementById('ln-degree-value'),

                approxValueDisplay: document.getElementById('ln-approx-value'), approxDegreeDisplay: document.getElementById('ln-approx-degree'),

                errorValueDisplay: document.getElementById('ln-error-value'), globalCanvas: document.getElementById('ln-global-canvas'),

                zoomedCanvas: document.getElementById('ln-zoomed-canvas'), zoomTitle: document.getElementById('ln-zoom-title'),

                trueValueLabel: document.getElementById('ln-true-value-label'),

                originalFunc: (x) => (x > -1) ? Math.log(1 + x) : NaN,

                maclaurinApprox: (x, n) => { let sum = 0; for (let i = 1; i <= n; i++) sum += Math.pow(-1, i - 1) * Math.pow(x, i) / i; return sum; }

            };

            ln.gCtx = ln.globalCanvas.getContext('2d'); ln.zCtx = ln.zoomedCanvas.getContext('2d');


            function updateLn() {

                const degree = parseInt(ln.slider.value);

                ln.FOCUS_X = parseFloat(ln.focusSlider.value);

                const TRUE_VALUE = ln.originalFunc(ln.FOCUS_X);


                ln.degreeValue.textContent = degree; ln.approxDegreeDisplay.textContent = degree;

                ln.focusValue.textContent = ln.FOCUS_X.toFixed(1);

                ln.zoomTitle.textContent = `x = ${ln.FOCUS_X.toFixed(1)} での拡大図`;

                ln.trueValueLabel.textContent = `実際の値 (ln(${(1 + ln.FOCUS_X).toFixed(1)}))`;

                

                let formula = 'P_n(x) = x';

                if (degree >= 2) formula += ` - \\frac{x^2}{2}`;

                if (degree >= 3) formula += ` + \\frac{x^3}{3}`;

                if (degree > 3) { const sign = (degree % 2 === 1) ? '+' : '-'; formula += ` - \\dots ${sign} \\frac{x^${degree}}{${degree}}`; }

                renderFormula(ln.formulaDisplay, formula);

                

                const approxValue = ln.maclaurinApprox(ln.FOCUS_X, degree);

                ln.trueValueDisplay.textContent = isNaN(TRUE_VALUE) ? "定義域外" : TRUE_VALUE.toFixed(8);

                ln.approxValueDisplay.textContent = approxValue.toFixed(8);

                ln.errorValueDisplay.textContent = isNaN(TRUE_VALUE) ? "---" : Math.abs(TRUE_VALUE - approxValue).toExponential(4);

                drawLn();

            }


            function drawLn() {

                const degree = parseInt(ln.slider.value);

                const TRUE_VALUE = ln.originalFunc(ln.FOCUS_X);

                const gw = ln.globalCanvas.clientWidth, gh = ln.globalCanvas.clientHeight; const gxRange = [-1.5, 3], gyRange = [-4, 3];

                drawGridAndAxes(ln.gCtx, gw, gh, gxRange, gyRange, false, ln.COLORS.axis, ln.COLORS.grid, ln.COLORS.text);

                ln.gCtx.strokeStyle = 'rgba(150, 150, 150, 0.5)'; ln.gCtx.lineWidth = 2; ln.gCtx.setLineDash([5, 5]); ln.gCtx.beginPath();

                const asymptoteX = (-1 - gxRange[0]) / (gxRange[1] - gxRange[0]) * gw;

                ln.gCtx.moveTo(asymptoteX, 0); ln.gCtx.lineTo(asymptoteX, gh); ln.gCtx.stroke(); ln.gCtx.setLineDash([]);

                drawFunction(ln.gCtx, gw, gh, gxRange, gyRange, ln.originalFunc, ln.COLORS.original, 3);

                drawFunction(ln.gCtx, gw, gh, gxRange, gyRange, (x) => ln.maclaurinApprox(x, degree), ln.COLORS.approx, 3);

                const focusCanvasX = (ln.FOCUS_X - gxRange[0]) / (gxRange[1] - gxRange[0]) * gw; const focusCanvasY = gh - (TRUE_VALUE - gyRange[0]) / (gyRange[1] - gyRange[0]) * gh;

                ln.gCtx.fillStyle = ln.COLORS.approx; ln.gCtx.beginPath(); ln.gCtx.arc(focusCanvasX, focusCanvasY, 5, 0, 2 * Math.PI); ln.gCtx.fill();

                

                const zw = ln.zoomedCanvas.clientWidth, zh = ln.zoomedCanvas.clientHeight; const zoomFactor = Math.max(0.1, 0.5 - degree * 0.02); const zxRange = [ln.FOCUS_X - zoomFactor, ln.FOCUS_X + zoomFactor]; const zyRange = [TRUE_VALUE - zoomFactor, TRUE_VALUE + zoomFactor];

                drawGridAndAxes(ln.zCtx, zw, zh, zxRange, zyRange, true, ln.COLORS.axis, ln.COLORS.grid, ln.COLORS.text);

                drawFunction(ln.zCtx, zw, zh, zxRange, zyRange, ln.originalFunc, ln.COLORS.original, 3);

                drawFunction(ln.zCtx, zw, zh, zxRange, zyRange, (x) => ln.maclaurinApprox(x, degree), ln.COLORS.approx, 3);

            }
