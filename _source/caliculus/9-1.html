        function createAxisLabel(text, position, color) {

        // --- Object Creation ---

        function createParaboloid() {

            if (paraboloidMesh) {

                scene.remove(paraboloidMesh);

                paraboloidMesh.geometry.dispose();

                if(paraboloidMesh.material.dispose) paraboloidMesh.material.dispose();

            }


            const geometry = new THREE.PlaneGeometry(params.width, params.height, params.segments, params.segments);

            geometry.rotateX(-Math.PI / 2);


            const vertices = geometry.attributes.position.array;

            for (let i = 0; i < vertices.length; i += 3) {

                const x = vertices[i];

                const z = vertices[i + 2];

                vertices[i + 1] = surfaceFunc(x, z);

            }

            

            geometry.attributes.position.needsUpdate = true;

            geometry.computeVertexNormals();

            

            let material;

            if (isWireframe) {

        function updateOnPointerMove(event, forceUpdate = false) {

             if (isLocked && !forceUpdate) {

                 return;

             }


             let intersection;


             if (forceUpdate && lastKnownIntersection) {

                 intersection = lastKnownIntersection;

             }

             else if (event) {

                 const rect = renderer.domElement.getBoundingClientRect();

                 pointer.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;

                 pointer.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

                 

                 raycaster.setFromCamera(pointer, camera);

                 const intersects = raycaster.intersectObject(basePlane);

                 

                 if (intersects.length > 0) {

                     intersection = intersects[0].point;

                     if (!isLocked) {

                         lastKnownIntersection = intersection;

                     }

                 }

             }


            if (intersection) {

                const x = intersection.x;

                const y = intersection.z;

                const z_val = surfaceFunc(x, y);


                pointOnSurface.position.set(x, z_val, y);

                

                const { gradX, gradY } = functions[currentFunctionKey].gradient(x, y, params);

                const magnitude = Math.sqrt(gradX * gradX + gradY * gradY);

                

                // --- gradientArrow (オレンジ) の更新 ---

                const gradientDir = new THREE.Vector3(gradX, 0, gradY).normalize();

                const gradientLength = magnitude > 0.001 ? magnitude : 0;

                

                gradientArrow.position.copy(pointOnSurface.position);

                gradientArrow.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), gradientDir);

                gradientArrow.userData.line.scale.y = gradientLength;

                gradientArrow.userData.line.position.y = gradientLength / 2;

                gradientArrow.userData.cone.position.y = gradientLength;

                

                // --- steepestArrow (濃い灰色) の更新 ---

                const steepestDir = new THREE.Vector3(gradX, gradX * gradX + gradY * gradY, gradY).normalize();

                const steepestLength = magnitude > 0.001 ? 1.2 : 0; // 長さは固定

                

                steepestArrow.position.copy(pointOnSurface.position);

                steepestArrow.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), steepestDir);

                steepestArrow.userData.line.scale.y = steepestLength;

                steepestArrow.userData.line.position.y = steepestLength / 2;

                steepestArrow.userData.cone.position.y = steepestLength;


                // --- 接平面の更新 ---

                const normal = new THREE.Vector3(-gradX, 1, -gradY).normalize();

                tangentPlaneMesh.position.copy(pointOnSurface.position);

                

                const defaultPlaneNormal = new THREE.Vector3(0, 0, 1);

                tangentPlaneMesh.quaternion.setFromUnitVectors(defaultPlaneNormal, normal);


                pointOnSurface.visible = true;

                tangentPlaneMesh.visible = true;

                gradientArrow.visible = areVectorsVisible && gradientLength > 0.001;

                steepestArrow.visible = areVectorsVisible && steepestLength > 0.001;

                

                updateInfoPanel(x, y, gradX, gradY, magnitude);

            }

        }


        function onPointerDown(event) {

            pointerDownTime = Date.now();

            pointerDownPosition.set(event.clientX, event.clientY);

        }


        function onPointerUp(event) {

            const timeElapsed = Date.now() - pointerDownTime;

            const distanceMoved = pointerDownPosition.distanceTo(new THREE.Vector2(event.clientX, event.clientY));


            if (timeElapsed < clickTimeThreshold && distanceMoved < dragThreshold) {

                toggleLockState();

            }

        }


        function toggleLockState() {

            isLocked = !isLocked;

            updateLockStatusUI();

        }


        function updateLockStatusUI() {

            const statusContainer = lockStatus.parentElement;

            if (isLocked) {

                lockStatus.textContent = '状態: 位置を固定中 (クリックで解除)';

                statusContainer.classList.remove('bg-blue-100', 'text-blue-800');

                statusContainer.classList.add('bg-red-100', 'text-red-800');

                container.style.cursor = 'default';

            } else {

                lockStatus.textContent = '状態: マウスに追従中 (クリックで固定)';

                statusContainer.classList.remove('bg-red-100', 'text-red-800');

                statusContainer.classList.add('bg-blue-100', 'text-blue-800');

                container.style.cursor = 'crosshair';

            }

        }
