                const z0 = Math.sqrt(-2.0 * Math.log(u1)) * Math.cos(2.0 * Math.PI * u2);

                const noiseVal = 128 + z0 * 128; // Noise centered at gray

                

                data[i] = noiseVal;

                data[i + 1] = noiseVal;

                data[i + 2] = noiseVal;

                data[i + 3] = 255;

            }

            return noiseData;

        }

            function applyNoise(noiseFactor) {

                if (!originalImageData) return;

                const newImageData = f_ctx.createImageData(CANVAS_SIZE, CANVAS_SIZE);

                const originalData = originalImageData.data;

                const newData = newImageData.data;

                for (let i = 0; i < originalData.length; i += 4) {

                    const u1 = Math.random();

                    const u2 = Math.random();

                    const z0 = Math.sqrt(-2.0 * Math.log(u1)) * Math.cos(2.0 * Math.PI * u2);

                    

                    const noiseOffset = z0 * 128;


                    const r = originalData[i], g = originalData[i + 1], b = originalData[i + 2], a = originalData[i + 3];

                    newData[i] = r * (1 - noiseFactor) + (128 + noiseOffset) * noiseFactor;

                    newData[i + 1] = g * (1 - noiseFactor) + (128 + noiseOffset) * noiseFactor;

                    newData[i + 2] = b * (1 - noiseFactor) + (128 + noiseOffset) * noiseFactor;

                    newData[i + 3] = a;

                }

                f_ctx.putImageData(newImageData, 0, 0);

            }
