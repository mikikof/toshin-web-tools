        function resizeCanvases() {

        function drawSpeedometer() {

            const ctx = spCtx;

            const canvas = speedometerCanvas;

            ctx.clearRect(0, 0, canvas.width, canvas.height);


            const x = canvas.width / 2;

            const y = canvas.height / 2 + 10;

            const radius = canvas.width / 2 - 40;

            const maxSpeed = MAX_SPEED;


            const startAngleDeg = 135, endAngleDeg = 405;

            const totalAngleDeg = endAngleDeg - startAngleDeg;

            const startAngleRad = startAngleDeg * Math.PI / 180;

            const endAngleRad = endAngleDeg * Math.PI / 180;

            const totalAngleRad = totalAngleDeg * Math.PI / 180;


            ctx.beginPath();

            ctx.arc(x, y, radius, startAngleRad, endAngleRad);

            ctx.strokeStyle = '#e2e8f0';

            ctx.lineWidth = 15;

            ctx.lineCap = 'round';

            ctx.stroke();


            const speedRatio = Math.max(0, Math.min(speed, maxSpeed)) / maxSpeed;

            const currentAngleRad = startAngleRad + (totalAngleRad * speedRatio);


            if (speedRatio > 0) {

                ctx.beginPath();

                ctx.arc(x, y, radius, startAngleRad, currentAngleRad);

                ctx.strokeStyle = '#4299e1';

                ctx.lineWidth = 15;

                ctx.lineCap = 'round';

                ctx.stroke();

            }


            ctx.save();

            ctx.translate(x, y);

            ctx.rotate(currentAngleRad);

            ctx.beginPath();

            ctx.moveTo(0, 0);

            ctx.lineTo(radius - 10, 0);

            ctx.strokeStyle = '#1a202c';

            ctx.lineWidth = 4;

            ctx.stroke();

            ctx.beginPath();

            ctx.arc(0, 0, 5, 0, 2 * Math.PI);

            ctx.fillStyle = '#1a202c';

            ctx.fill();

            ctx.restore();


            ctx.font = 'bold 12px Inter';

            ctx.fillStyle = '#4a5568';

            const drawLabel = (value) => {

                const labelRatio = value / maxSpeed;

                const labelAngleRad = startAngleRad + (totalAngleRad * labelRatio);

                const labelRadius = radius + 20;

                const labelX = x + labelRadius * Math.cos(labelAngleRad);

                const labelY = y + labelRadius * Math.sin(labelAngleRad);

                ctx.fillText(value, labelX, labelY);

            };

            drawLabel(0);

            drawLabel(60);

            drawLabel(120);

        }


        function drawTimeDistChart() {

            tdCtx.clearRect(0, 0, timeDistCanvas.width, timeDistCanvas.height);

            const padding = 40;

            const chartWidth = timeDistCanvas.width - padding * 2;

            const chartHeight = timeDistCanvas.height - padding * 2;

            const xMax = Math.max(INITIAL_MAX_TIME, time);

            const yMax = Math.max(100, data.length > 0 ? Math.max(...data.map(p => p.distance)) : 100);


            drawAxes(tdCtx, timeDistCanvas, padding, '時間 (s)', '距離 (m)');

            drawGridAndTicks(tdCtx, padding, chartWidth, chartHeight, xMax, yMax, Math.ceil(xMax / 6 / 5) * 5, Math.ceil(yMax / 5 / 50) * 50);

            drawPath(tdCtx, padding, chartWidth, chartHeight, data, p => p.time, p => p.distance, xMax, yMax, '#805ad5');


            if (hoverTime && data.length > 1) {

                const hoverIndex = findClosestIndex(data, hoverTime);

                const point = data[hoverIndex];

                

                const x = padding + (point.time / xMax) * chartWidth;

                const y = timeDistCanvas.height - padding - (point.distance / yMax) * chartHeight;


                const speed_ms = point.speed * 1000 / 3600;

                const canvasSlope = -speed_ms * (chartHeight / yMax) / (chartWidth / xMax);


                const lineLength = 40;

                const dx = lineLength / Math.sqrt(1 + canvasSlope * canvasSlope);

                const dy = canvasSlope * dx;


                tdCtx.beginPath();

                tdCtx.moveTo(x - dx, y - dy);

                tdCtx.lineTo(x + dx, y + dy);

                tdCtx.strokeStyle = '#3182ce';

                tdCtx.lineWidth = 2;

                tdCtx.stroke();

                

                drawMarker(tdCtx, x, y, '#805ad5');

                

                const infoText = [

                    `傾き (速度): ${point.speed.toFixed(1)} km/h`,

                    `速度グラフの値: ${point.speed.toFixed(1)} km/h`

                ];

                drawInfoBox(tdCtx, x, y, infoText);

            }

        }

        

        function drawTimeSpeedChart() {

            const chartHeight = timeSpeedCanvas.height - padding * 2;

            const xMax = Math.max(INITIAL_MAX_TIME, time);

            drawAxes(tsCtx, timeSpeedCanvas, padding, '時間 (s)', '速度 (km/h)');

            drawPath(tsCtx, padding, chartWidth, chartHeight, data, p => p.time, p => p.speed, xMax, yMax, '#3182ce');


            if (hoverTime && data.length > 1) {

                tsCtx.fillStyle = 'rgba(129, 230, 129, 0.5)';

                tsCtx.beginPath();

                tsCtx.moveTo(padding, timeSpeedCanvas.height - padding);

                data.forEach(p => {

                    if (p.time > hoverTime) return;

                    const x = padding + (p.time / xMax) * chartWidth;

                    const y = timeSpeedCanvas.height - padding - (p.speed / yMax) * chartHeight;

                    tsCtx.lineTo(x, y);

                });

                const lastHoverPoint = data[findClosestIndex(data, hoverTime)];

                tsCtx.lineTo(padding + (lastHoverPoint.time / xMax) * chartWidth, timeSpeedCanvas.height - padding);

                tsCtx.closePath();

                tsCtx.fill();

                

                const hoverIndex = findClosestIndex(data, hoverTime);

                const point = data[hoverIndex];

                const x = padding + (point.time / xMax) * chartWidth;

                const y = timeSpeedCanvas.height - padding - (point.speed / yMax) * chartHeight;

                

                drawMarker(tsCtx, x, y, '#3182ce');


                const infoText = [

                    `面積 (距離): ${point.distance.toFixed(1)} m`,

                    `距離グラフの値: ${point.distance.toFixed(1)} m`

                ];

                drawInfoBox(tsCtx, x, y, infoText);

            }

        }

            }

        function drawInfoBox(ctx, x, y, lines) {

            ctx.font = 'bold 12px Inter';

            const textMetrics = lines.map(line => ctx.measureText(line));

            const maxWidth = Math.max(...textMetrics.map(m => m.width));

            

            const padding = 8;

            const lineHeight = 18;

            const boxWidth = maxWidth + padding * 2;

            const boxHeight = lines.length * lineHeight + padding;


            let boxX = x + 20;

            let boxY = y - boxHeight / 2;


            if (boxX + boxWidth > ctx.canvas.width - 10) {

                boxX = x - boxWidth - 20;

            }

            if (boxY < 10) {

                boxY = 10;

            }

            if (boxY + boxHeight > ctx.canvas.height - 10) {

                boxY = ctx.canvas.height - 10 - boxHeight;

            }


            ctx.fillStyle = 'rgba(26, 32, 44, 0.8)';

            ctx.strokeStyle = 'rgba(255, 255, 255, 0.5)';

            ctx.lineWidth = 1;

            ctx.beginPath();

            ctx.rect(boxX, boxY, boxWidth, boxHeight);

            ctx.fill();

            ctx.stroke();


            ctx.fillStyle = 'white';

            ctx.textAlign = 'left';

            ctx.textBaseline = 'top';

            lines.forEach((line, index) => {

                ctx.fillText(line, boxX + padding, boxY + padding / 2 + index * lineHeight);

            });

        }


        function drawAxes(ctx, canvas, padding, xLabel, yLabel) {

            ctx.beginPath();

            ctx.moveTo(padding, padding);

            ctx.lineTo(padding, canvas.height - padding);

            ctx.lineTo(canvas.width - padding, canvas.height - padding);

            ctx.strokeStyle = '#a0aec0';

            ctx.stroke();

            ctx.fillStyle = '#4a5568';

            ctx.textAlign = 'center';

            ctx.textBaseline = 'top';

            ctx.font = '12px Noto Sans JP';

            ctx.fillText(xLabel, canvas.width / 2, canvas.height - padding + 15);

            ctx.save();

            ctx.translate(padding - 30, canvas.height / 2);

            ctx.rotate(-Math.PI / 2);

            ctx.textAlign = 'center';

            ctx.textBaseline = 'middle';

            ctx.fillText(yLabel, 0, 0);

            ctx.restore();

        }


        function drawGridAndTicks(ctx, padding, width, height, xMax, yMax, xInterval, yInterval) {

            ctx.textAlign = 'center';

            ctx.textBaseline = 'top';

            for (let i = 0; i <= xMax; i += xInterval) {

                const xPos = padding + (i / xMax) * width;

                if(xPos <= ctx.canvas.width - padding) drawTick(ctx, xPos, ctx.canvas.height - padding, Math.round(i), 'bottom');

            }

            ctx.textAlign = 'right';

            ctx.textBaseline = 'middle';

            for (let i = 0; i <= yMax; i += yInterval) {

                const yPos = ctx.canvas.height - padding - (i / yMax) * height;

                drawTick(ctx, padding, yPos, Math.round(i), 'left');

            }

        }

        

        function drawTick(ctx, x, y, text, position) {

            ctx.fillStyle = '#718096';

            ctx.font = '10px Inter';

            if (position === 'bottom') {

                ctx.fillText(text, x, y + 5);

            } else { 

                ctx.fillText(text, x - 5, y);

            }

        }


        function drawPath(ctx, padding, width, height, data, getX, getY, xMax, yMax, color, yMin = 0) {

            const yRange = yMax - yMin;

            ctx.beginPath();

            if (data.length > 0) {

                 const startX = padding + (getX(data[0]) / xMax) * width;

                 const startY = ctx.canvas.height - padding - (getY(data[0]) - yMin) / yRange * height;

                 ctx.moveTo(startX, startY);

            }

            data.forEach(p => {

                const x = padding + (getX(p) / xMax) * width;

                const y = ctx.canvas.height - padding - (getY(p) - yMin) / yRange * height;

                ctx.lineTo(x, y);

            });

            ctx.strokeStyle = color;

            ctx.lineWidth = 3;

            ctx.stroke();

            ctx.lineWidth = 1;

        }


        function drawMarker(ctx, x, y, color) {

            ctx.beginPath();

            ctx.arc(x, y, 5, 0, 2 * Math.PI);

            ctx.fillStyle = color;

            ctx.fill();

            ctx.strokeStyle = 'white';

            ctx.lineWidth = 2;

            ctx.stroke();

        }


        function findClosestIndex(arr, targetTime) {

            let closestIndex = 0;

            let minDiff = Infinity;

            arr.forEach((point, index) => {

                const diff = Math.abs(point.time - targetTime);

                if (diff < minDiff) {

                    minDiff = diff;

                    closestIndex = index;

                }

            });

            return closestIndex;

        }

        

        function gameLoop(timestamp) {

            if (!animationFrameId) return;

            if (!lastTimestamp) lastTimestamp = timestamp;

            const deltaTime = (timestamp - lastTimestamp) / 1000;

            lastTimestamp = timestamp;


            time += deltaTime;

            const totalAcceleration = acceleration - deceleration;

            speed += totalAcceleration * deltaTime;

            speed = Math.max(0, Math.min(speed, MAX_SPEED));

            const speed_ms = speed * 1000 / 3600;

            distance += speed_ms * deltaTime;

            roadOffset -= speed_ms * 3;

            roadLines.style.backgroundPositionX = roadOffset + 'px';


            data.push({ time, speed, distance, accel: totalAcceleration });

            

            car.classList.toggle('moving', speed > 1 && animationFrameId);

            car.classList.toggle('braking', deceleration > 0 && speed > 0);


            drawAllCharts();

            updateDisplay();

            animationFrameId = requestAnimationFrame(gameLoop);

        }


        function updateDisplay() {

            timeDisplay.textContent = time.toFixed(1);

            speedDisplay.textContent = Math.round(speed);

            distanceDisplay.textContent = distance.toFixed(1);

        }

        

        startBtn.addEventListener('click', () => { if (!animationFrameId) { lastTimestamp = null; animationFrameId = requestAnimationFrame(gameLoop); } });

        stopBtn.addEventListener('click', () => { if (animationFrameId) { cancelAnimationFrame(animationFrameId); animationFrameId = null; car.classList.remove('moving'); } });


        resetBtn.addEventListener('click', () => {

            if (animationFrameId) { cancelAnimationFrame(animationFrameId); animationFrameId = null; }

            time = 0; speed = 0; distance = 0; acceleration = 0; deceleration = 0;

            accelSlider.value = 0; brakeSlider.value = 0;

            data = []; roadOffset = 0; hoverTime = null;

            roadLines.style.backgroundPositionX = '0px';

            car.classList.remove('moving', 'braking');

            drawAllCharts();

            updateDisplay();

        });


        accelSlider.addEventListener('input', (e) => { acceleration = e.target.value * ACCEL_FACTOR; });

        brakeSlider.addEventListener('input', (e) => { deceleration = e.target.value * BRAKE_FACTOR; });

        

        [timeDistCanvas, timeSpeedCanvas].forEach(canvas => {

            canvas.addEventListener('mousemove', (e) => {

                if (data.length === 0 || animationFrameId) return;

                const rect = canvas.getBoundingClientRect();

                const x = e.clientX - rect.left;

                const padding = 40;

                const chartWidth = canvas.width - padding * 2;

                const xMax = Math.max(INITIAL_MAX_TIME, time);

                const mouseTime = (x - padding) / chartWidth * xMax;

                if (mouseTime >= 0 && mouseTime <= time) {

                    hoverTime = mouseTime;

                    drawAllCharts();

                }

            });

            canvas.addEventListener('mouseleave', () => {

                if(animationFrameId) return;

                hoverTime = null;

                drawAllCharts();

            });

        });

        

        window.addEventListener('resize', resizeCanvases);


        // 初期化

        resizeCanvases();

        updateDisplay();
