    // ★★★ 変更点: Plotlyが期待するデータ構造(z[y][x])に合うようにループ順を修正 ★★★

    function generateData(func, range = 2, steps = 60, funcKey) {

        const x = [], y = [], z = [];

        const step = (2 * range) / (steps - 1);

        let zMin = Infinity, zMax = -Infinity;


        // xとyの座標ベクトルを先に作成

        for (let i = 0; i < steps; i++) {

            const val = -range + i * step;

            x.push(val);

            y.push(val);

        }


        // yを外側、xを内側のループにすることで z[y][x] の構造を作る

        for (let i = 0; i < steps; i++) {

            const yi = y[i];

            const z_row = [];

            for (let j = 0; j < steps; j++) {

                const xj = x[j];

                

                let val;

                // お椀型の場合、円の外側のデータをnullにして円形に描画

                if (funcKey === 'paraboloid') {

                    const r = Math.sqrt(xj*xj + yi*yi);

                    if (r > range) {

                        val = null;

                    } else {

                        val = func(xj, yi);

                    }

                } else {

                    val = func(xj, yi);

                }


                z_row.push(val);

                if (val !== null) {

                    if (val < zMin) zMin = val;

                    if (val > zMax) zMax = val;

                }

            }

            z.push(z_row);

        }

        return { x, y, z, zMin, zMax };

    }

        zSlider.step = range > 1e-9 ? range / 100 : 0.01;

        zSlider.value = (currentData.zMin + currentData.zMax) / 2;

        

        formulaDisplay.innerHTML = selectedFunc.mathML;


        updatePlots();

    }

    

    function toggleTheme() {

        document.body.classList.toggle('light-mode');

        const isLight = document.body.classList.contains('light-mode');

        document.getElementById('theme-icon-sun').classList.toggle('hidden', isLight);

        document.getElementById('theme-icon-moon').classList.toggle('hidden', !isLight);

    }
