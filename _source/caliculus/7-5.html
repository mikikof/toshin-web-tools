  <script type="module">

    import * as THREE from 'three';

    import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

    import { ParametricGeometry } from 'three/addons/geometries/ParametricGeometry.js';


    // --- DOM Elements ---

    const container = document.getElementById('graph-container');

    const coordsDisplay = document.getElementById('coords');

    const zValueDisplay = document.getElementById('z-value');

    const limitValueDisplay = document.getElementById('limit-value');

    const explanationDisplay = document.getElementById('explanation');

    const cSlider = document.getElementById('c-slider');

    const cVal = document.getElementById('c-val');

    const themeToggle = document.getElementById('theme-toggle');

    const sunIcon = document.getElementById('theme-icon-sun');

    const moonIcon = document.getElementById('theme-icon-moon');

    const uiToggle = document.getElementById('ui-toggle');

    const controlPanel = document.getElementById('control-panel');


    // --- Parameters ---

    const EPS = 1e-10;      // xの下限（描画用）。さらに厳密に0に近づけます。

    const XMAX = 1.5;

    const YMIN = -2.5;     // Expanded Y range

    const YMAX = 2.5;      // Expanded Y range

    const ZCAP = 5.0;      // Upper bound for z in rendering

    let paramC = parseFloat(cSlider.value);


    // --- Math helpers ---

    function xPowY(x, y) {

      if (x === 0) return (y > 0 ? 0 : NaN); // Simple handling for 0^y

      if (x < 0) return NaN;                  // Don't handle non-real values

      const z = Math.exp(y * Math.log(x));

      // We no longer clamp here, the shader will handle the visual fade-out

      return z;

    }


    function pathX(t) {

      // t from 0 to 1 moves from 1 to EPS

      return EPS + (1.0 - EPS) * (1.0 - t);

    }


    // --- Path definitions ---

    const pathData = {

      'y_eq_0': {

        func: (t) => {

          const x = pathX(t), y = 0;

          const z = xPowY(x, y);

          return new THREE.Vector3(x, y, z);

        },

        limit: () => 1,

        explanation: '経路 <code>y=0</code> では <span class="font-mono">f(x,0)=1</span>。よって <span class="text-green-500 dark:text-green-400 font-bold">z=1</span> に収束。'

      },

      'x_eq_0': {

        func: (t) => {

          const x = 0;

          const y = 1.0 - t; // Approach 0 while keeping y>0

          const z = (y >= 0 ? 0 : NaN); // Use >= to prevent NaN for TubeGeometry

          return new THREE.Vector3(x, y, z);

        },

        limit: () => 0,

        explanation: '経路 <code>x=0^{+}</code>, <code>y>0</code> では <span class="font-mono">f(0,y)=0</span>。よって <span class="text-red-500 dark:text-red-400 font-bold">z=0</span> に収束。'

      },

      'y_eq_x': {

        func: (t) => {

          const x = pathX(t), y = x;

          const z = xPowY(x, y);

          return new THREE.Vector3(x, y, z);

        },

        limit: () => 1,

        explanation: '経路 <code>y=x</code> では <span class="font-mono">x^x=exp(x ln x)</span>。<span class="font-mono">x ln x → 0</span> より <span class="text-green-500 dark:text-green-400 font-bold">z=1</span>。'

      },

      'y_eq_x2': {

        func: (t) => {

          const x = pathX(t), y = x * x;

          const z = xPowY(x, y);

          return new THREE.Vector3(x, y, z);

        },

        limit: () => 1,

        explanation: '経路 <code>y=x^2</code> でも <span class="font-mono">x^(x^2)=exp(x^2 ln x)→1</span>。'

      },

      'y_eq_c': {

        func: (t) => {

          const x = pathX(t), y = paramC; // c>0

          const z = xPowY(x, y);

          return new THREE.Vector3(x, y, z);

        },

        limit: () => 0, // 0 since c>0

        explanation: () => `経路 <code>y=c(=${paramC.toFixed(1)})</code> では <span class="font-mono">x^c→0</span>。<span class="text-red-500 dark:text-red-400 font-bold">z=0</span> に収束。`

      },

      'y_eq_c_over_log': {

        func: (t) => {

          const x = pathX(t);

          const denom = Math.log(1 / x);

          const y = (denom > 1e-6) ? (paramC / denom) : Math.min(YMAX, paramC / 1e-6);

          const z = xPowY(x, y); // = exp(y ln x) ≈ exp(-c)

          return new THREE.Vector3(x, y, z);

        },

        limit: () => Math.exp(-paramC),

        explanation: () => `経路 <code>y=c/ln(1/x)</code> では <span class="font-mono">y ln x=-c</span>。<span class="text-sky-500 dark:text-sky-400 font-bold">z=e^(-c)≈${Math.exp(-paramC).toFixed(3)}</span> に収束。`

      }

    };


    // --- Scene setup ---

    let scene, camera, renderer, controls, ambientLight, dirLight, hemisphereLight;

    let surface, particle, pathLine, limitPlaneZ1, limitPlaneZ0, limitPlaneZexp;


    function init() {

      scene = new THREE.Scene();

      scene.background = new THREE.Color(0x121212); // Default background

      scene.fog = new THREE.FogExp2(0x121212, 0.08);


      camera = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);

      camera.up.set(0, 0, 1); // Set Z as up

      camera.position.set(1.0, -1.0, 0.9); // Zoomed in camera position


      renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

      renderer.setSize(container.clientWidth, container.clientHeight);

      renderer.setPixelRatio(window.devicePixelRatio);

      container.appendChild(renderer.domElement);


      controls = new OrbitControls(camera, renderer.domElement);

      controls.enableDamping = true; controls.dampingFactor = 0.05;

      controls.minDistance = 0.5; // Allow closer zoom

      controls.maxDistance = 20;

      controls.target.set(0.2, 0.0, 0.2); // Zoomed in camera target


      // Lights

      ambientLight = new THREE.AmbientLight(0xffffff, 0.5);

      scene.add(ambientLight);

      dirLight = new THREE.DirectionalLight(0xffffff, 1);

      dirLight.position.set(2, 5, 3);

      scene.add(dirLight);

      hemisphereLight = new THREE.HemisphereLight(0xcce0ff, 0xffffff, 0); // Sky, Ground, Intensity (off by default)

      scene.add(hemisphereLight);

      

      const createAxis = (start, end, color) => {

          const path = new THREE.LineCurve3(start, end);

          const geometry = new THREE.TubeGeometry(path, 1, 0.01, 8, false);

          const material = new THREE.MeshStandardMaterial({ color: color });

          return new THREE.Mesh(geometry, material);

      };


      const axisLength = 3.0; // Increased axis length

      scene.add(createAxis(new THREE.Vector3(0, 0, 0), new THREE.Vector3(axisLength, 0, 0), 0x0000ff)); // X: Blue

      scene.add(createAxis(new THREE.Vector3(0, 0, 0), new THREE.Vector3(0, axisLength, 0), 0xff0000)); // Y: Red

      scene.add(createAxis(new THREE.Vector3(0, 0, 0), new THREE.Vector3(0, -axisLength, 0), 0xff0000)); // Negative Y: Red

      scene.add(createAxis(new THREE.Vector3(0, 0, 0), new THREE.Vector3(0, 0, axisLength), 0x00ff00)); // Z: Green


      const surfaceFunc = (u, v, target) => {

        const x = EPS + (XMAX - EPS) * u;

        const y = YMIN + (YMAX - YMIN) * v;

        const z = xPowY(x, y);

        // Clamp z for positioning, but the shader will handle the visual appearance

        const zClamped = Math.min(Math.max(z, -ZCAP), ZCAP);

        target.set(x, y, zClamped);

      };

      

      const geometry = new ParametricGeometry(surfaceFunc, 250, 250);

      

      const material = new THREE.MeshStandardMaterial({

        side: THREE.DoubleSide,

        transparent: true // Crucial for the fade effect

      });


      // Shader modification to fade out the graph at the top

      material.onBeforeCompile = shader => {

          shader.uniforms.zCap = { value: ZCAP };

          shader.uniforms.fadeRange = { value: 1.5 }; // Fade over 1.5 units


          shader.vertexShader = 'varying float vWorldZ;\n' + shader.vertexShader;

          shader.vertexShader = shader.vertexShader.replace(

              '#include <worldpos_vertex>',

              `

              #include <worldpos_vertex>

              // This is the robust way to get world position for the varying.

              vec4 worldPositionTemp = modelMatrix * vec4( transformed, 1.0 );

              vWorldZ = worldPositionTemp.z;

              `

          );


          shader.fragmentShader = 'varying float vWorldZ;\n' + 'uniform float zCap;\n' + 'uniform float fadeRange;\n' + shader.fragmentShader;

          shader.fragmentShader = shader.fragmentShader.replace(

              '#include <color_fragment>',

              `

              #include <color_fragment>

              // smoothstep(edge0, edge1, x) -> 0 if x<=edge0, 1 if x>=edge1

              float alpha = 1.0 - smoothstep(zCap - fadeRange, zCap, vWorldZ);

              diffuseColor.a *= alpha;

              `

          );

      };


      surface = new THREE.Mesh(geometry, material);

      scene.add(surface);


      const particleGeo = new THREE.SphereGeometry(0.04, 32, 32);

      const particleMat = new THREE.MeshStandardMaterial({

        color: 0x00f2ff, emissive: 0x00f2ff, metalness: 0.5, roughness: 0.2

      });

      particle = new THREE.Mesh(particleGeo, particleMat);

      scene.add(particle);


      // Path Tube

      const tubeMaterial = new THREE.MeshStandardMaterial({ color: 0xfff700, emissive: 0xcc8400 });

      const tubeGeometry = new THREE.BufferGeometry(); // Start with empty geo

      pathLine = new THREE.Mesh(tubeGeometry, tubeMaterial);

      scene.add(pathLine);


      const planeGeo = new THREE.PlaneGeometry(XMAX, YMAX - YMIN); // Plane covers the new area

      

      const planeMaterialDefaults = {

          transparent: true,

          side: THREE.DoubleSide,

          depthWrite: false, // This is the key fix for transparency

          opacity: 0.35

      };


      limitPlaneZ1 = new THREE.Mesh(

        planeGeo,

        new THREE.MeshStandardMaterial({ ...planeMaterialDefaults, color: 0x4dcb7a })

      );

      limitPlaneZ1.position.set(XMAX / 2, (YMIN + YMAX) / 2, 1.0); // Center the plane

      limitPlaneZ1.visible = false;

      scene.add(limitPlaneZ1);

      

      limitPlaneZ0 = new THREE.Mesh(

        planeGeo,

        new THREE.MeshStandardMaterial({ ...planeMaterialDefaults, color: 0xff6b6b })

      );

      limitPlaneZ0.position.set(XMAX / 2, (YMIN + YMAX) / 2, 0.0); // Center the plane

      limitPlaneZ0.visible = false;

      scene.add(limitPlaneZ0);

      

      limitPlaneZexp = new THREE.Mesh(

        planeGeo,

        new THREE.MeshStandardMaterial({ ...planeMaterialDefaults, color: 0x4da6ff })

      );

      limitPlaneZexp.position.set(XMAX / 2, (YMIN + YMAX) / 2, Math.exp(-paramC)); // Center the plane

      limitPlaneZexp.visible = false;

      scene.add(limitPlaneZexp);


      updatePath(currentPath);

      updateTheme(); // Set initial theme for 3D scene

      window.addEventListener('resize', onWindowResize, false);

    }


    function onWindowResize() {

      camera.aspect = container.clientWidth / container.clientHeight;

      camera.updateProjectionMatrix();

      renderer.setSize(container.clientWidth, container.clientHeight);

    }


    let currentPath = 'y_eq_0';

    let animationTime = 0;

    let animationDirection = 1;


    function animate() {

      requestAnimationFrame(animate);

      controls.update();


      animationTime += 0.005 * animationDirection;

      if (animationTime > 1 || animationTime < 0) {

        animationDirection *= -1;

        animationTime = Math.max(0, Math.min(1, animationTime));

      }


      const pos = pathData[currentPath].func(animationTime);

      particle.position.copy(pos);

      updateInfoPanel(pos);

      renderer.render(scene, camera);

    }


    function updateInfoPanel(pos) {

      coordsDisplay.textContent = `(${pos.x.toFixed(3)}, ${pos.y.toFixed(3)})`;

      zValueDisplay.textContent = (Number.isFinite(pos.z) ? pos.z.toFixed(4) : '—');

    }


    function updatePath(pathId) {

      currentPath = pathId;


      // Update path tube geometry

      const pts = [];

      for (let i = 0; i <= 150; i++) pts.push(pathData[pathId].func(i / 150));

      const curve = new THREE.CatmullRomCurve3(pts);

      

      if (pathLine.geometry) {

          pathLine.geometry.dispose();

      }

      pathLine.geometry = new THREE.TubeGeometry(curve, 150, 0.02, 8, false);



      const lim = pathData[pathId].limit();

      const limStr = (lim === 0) ? '0' : (lim === 1) ? '1' : (Number.isFinite(lim) ? lim.toFixed(4) : '∞');


      limitValueDisplay.textContent = limStr;

      explanationDisplay.innerHTML = (typeof pathData[pathId].explanation === 'function')

          ? pathData[pathId].explanation()

          : pathData[pathId].explanation;


      limitPlaneZ1.visible = (lim === 1);

      limitPlaneZ0.visible = (lim === 0);

      if (Number.isFinite(lim) && lim !== 0 && lim !== 1) {

        limitPlaneZexp.visible = true;

        limitPlaneZexp.position.z = lim;

      } else {

        limitPlaneZexp.visible = false;

      }


      const isLight = !document.documentElement.classList.contains('dark');

      limitValueDisplay.className = 'font-mono text-xl font-bold ' + 

        ((lim === 0) ? 'text-red-500 dark:text-red-400'

        : (lim === 1) ? 'text-green-500 dark:text-green-400'

        : 'text-sky-500 dark:text-sky-400');

    }


    // --- Theme Handling ---

        const isLight = !document.documentElement.classList.contains('dark');

        

        sunIcon.classList.toggle('hidden', isLight);

        moonIcon.classList.toggle('hidden', !isLight);


        if (scene) { // Ensure scene is initialized

            

            scene.background.set(isLight ? 0xf9fafb : 0x121212);

            scene.fog.color.set(isLight ? 0xf9fafb : 0x121212);

            scene.fog.density = isLight ? 0.06 : 0.08;


            // Configure surface material based on theme

            surface.material.wireframe = true; 

            surface.material.color.set(isLight ? 0x00529B : 0x6a0dad); 

            surface.material.opacity = isLight ? 1.0 : 0.4;



            // Configure lighting based on theme

            ambientLight.intensity = isLight ? 0.3 : 0.5;

            dirLight.intensity = isLight ? 1.2 : 1.0;

            hemisphereLight.intensity = isLight ? 0.9 : 0; // Only on for light mode

            

            // Other elements

            particle.material.color.set(isLight ? 0x2563eb : 0x00f2ff);

            particle.material.emissive.set(isLight ? 0x2563eb : 0x00f2ff);

            pathLine.material.color.set(isLight ? 0xf97316 : 0xfff700);

            pathLine.material.emissive.set(isLight ? 0xd97706 : 0xcc8400);

        }

    }
