        function createCustomAxes() {

            const axisRadius = 0.025;

            const axisLength = 4.5;

            const arrowHeight = 0.3;


            const createAxisGroup = (color) => {

                cylinder.position.y = axisLength / 2;

            const yAxis = createAxisGroup(0x00ff00);

        }

        // --- Function and Path Updates ---

        function updateFunction() {

            clearAnimations();

            const selectedFuncKey = functionSelect.value;

            const selectedFunc = functions[selectedFuncKey];


            functionFormulaEl.innerHTML = selectedFunc.formula;

            

            pathSelect.innerHTML = '';

            Object.keys(selectedFunc.paths).forEach(pathName => {

                const option = document.createElement('option');

                option.value = pathName;

                option.textContent = pathName;

                pathSelect.appendChild(option);

            });

            

            limitValueDisplayEl.innerHTML = '';

            createSurface(selectedFunc.func);

        }

        

        // --- 3D Surface Creation ---

        function createSurface(func) {

            if (mesh) scene.remove(mesh);

            const geometry = new THREE.PlaneGeometry(8, 8, 250, 250);

            const position = geometry.attributes.position;

            const safeFunc = (x, y) => {

                if (Math.sqrt(x*x + y*y) < 1e-6) {

                    return 0; // Avoid division by zero at the origin

                }

                return func(x, y);

            };


            for (let i = 0; i < position.count; i++) {

                const x = position.getX(i);

                const y = position.getY(i);

                const z = safeFunc(x, y);

                position.setZ(i, !isNaN(z) && isFinite(z) ? z : -4);

            }

            position.needsUpdate = true;

            geometry.computeVertexNormals();

            const material = new THREE.MeshStandardMaterial({

                color: isLightMode ? colors.light.surface : colors.dark.surface, 

                wireframe: false, side: THREE.DoubleSide,

                transparent: true, opacity: 0.85

            });

            mesh = new THREE.Mesh(geometry, material);

            mesh.rotation.x = -Math.PI / 2;

            scene.add(mesh);

        }


        // --- Animation Control ---

        function clearAnimations() {

            if (animationFrameId) cancelAnimationFrame(animationFrameId);

            activeAnimationObjects.forEach(obj => {

                scene.remove(obj.sphere);

                if(obj.pathTube) scene.remove(obj.pathTube);

            });

            activeAnimationObjects = [];

            limitValueDisplayEl.innerHTML = '';

        }


        function startSingleAnimation() {

            const selectedFuncKey = functionSelect.value;

            const funcData = functions[selectedFuncKey];

            const pathName = pathSelect.value;

            const pathFunc = funcData.paths[pathName];

            

            const animationData = [{ pathName, pathFunc, color: pathColors[0] }];

            runAnimation(funcData, animationData);

        }


        function startMultiAnimation() {

            const selectedFuncKey = functionSelect.value;

            const funcData = functions[selectedFuncKey];

            const pathNames = funcData.multiPaths;


            const animationData = pathNames.map((name, index) => ({

                pathName: name,

                pathFunc: funcData.paths[name],

                color: pathColors[index]

            }));

            runAnimation(funcData, animationData);

        }


        function runAnimation(funcData, animationData) {

            clearAnimations();


            animationData.forEach(data => {

                const sphereGeo = new THREE.SphereGeometry(0.1, 32, 32);

                const sphereMat = new THREE.MeshPhongMaterial({ color: data.color });

                const sphere = new THREE.Mesh(sphereGeo, sphereMat);

                scene.add(sphere);


                activeAnimationObjects.push({

                    sphere, pathTube: null, points: [], ...data

                });

            });


            let t = 0;

            const duration = 3;

            const steps = 180;

            const dt = duration / steps;

            const endProgress = 0.995; // Stop just before the origin


            function animatePath() {

                t += dt;

                const progress = Math.min(t / duration, 1);


                activeAnimationObjects.forEach(obj => {

                    const currentProgress = Math.min(progress, endProgress);

                    const { x, y } = obj.pathFunc(currentProgress * 3);

                    const z = funcData.func(x, y);

                    

                    if (!isNaN(z) && isFinite(z)) {

                        obj.sphere.position.set(x, z, y);

                        obj.points.push(new THREE.Vector3(x, z, y));


                        if (obj.pathTube) scene.remove(obj.pathTube);

                        if (obj.points.length > 1) {

                            const curve = new THREE.CatmullRomCurve3(obj.points);

                            const tubeGeo = new THREE.TubeGeometry(curve, Math.max(2, obj.points.length), 0.03, 8, false);

                            const tubeMat = new THREE.MeshBasicMaterial({ color: obj.color });

                            obj.pathTube = new THREE.Mesh(tubeGeo, tubeMat);

                            scene.add(obj.pathTube);

                        }

                    }

                });


                if (progress < 1) {

                    animationFrameId = requestAnimationFrame(animatePath);

                } else {

                    let resultHTML = '';

                    if (funcData.limitExists) {

                         resultHTML = `<div class="font-bold text-lg">収束値: ${funcData.limitValue.toFixed(2)}</div>`;

                    } else {

                        activeAnimationObjects.forEach(obj => {

                            const finalZ = obj.sphere.position.y;

                            const textColor = '#' + new THREE.Color(obj.color).getHexString();

                            const finalValueColor = isLightMode ? '#c2410c' : '#fb923c';

                            resultHTML += `<div class="flex items-center justify-center gap-2">

                                <span style="color:${textColor};" class="font-bold text-sm">${obj.pathName}:</span>

                                <span style="color:${finalValueColor};" class="font-bold text-lg">${finalZ.toFixed(2)}</span>

                            </div>`;

                        });

                        resultHTML += `<div class="font-bold text-red-500 mt-1 text-xs">近づき方によって値が違うため、極限は存在しません。</div>`;

                    }

                    limitValueDisplayEl.innerHTML = resultHTML;

                }

            }

            animatePath();

        }


        // --- Render Loop & Resize ---

        function animate() {

            requestAnimationFrame(animate);

            controls.update();

            renderer.render(scene, camera);

        }

            camera.updateProjectionMatrix();

            renderer.setSize(window.innerWidth, window.innerHeight);

            renderer.setPixelRatio(window.devicePixelRatio);

        }
