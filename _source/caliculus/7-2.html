        // --- 軸ラベルを作成する関数 ---

        function createAxisLabel(text, position, color) {

            const canvas = document.createElement('canvas');

            const context = canvas.getContext('2d');

            const fontSize = 80;

            canvas.width = 128; canvas.height = 128;

            context.font = `Bold ${fontSize}px Arial`;

            context.fillStyle = color;

            context.textAlign = 'center'; context.textBaseline = 'middle';

            context.fillText(text, canvas.width / 2, canvas.height / 2);

            const texture = new THREE.CanvasTexture(canvas);

            const material = new THREE.SpriteMaterial({ map: texture, transparent: true });

            const sprite = new THREE.Sprite(material);

            sprite.scale.set(1, 1, 1);

            sprite.position.copy(position);

            return sprite;

        }

        // --- サーフェス（曲面）の更新 ---

        function updateSurface() {

            if (surfaceMesh) {

                scene.remove(surfaceMesh);

                surfaceMesh.geometry.dispose();

                surfaceMesh.material.dispose();

            }

            const a = parseFloat(paramA_slider.value);

            const b = parseFloat(paramB_slider.value);

            const graphFunction = (u, v, target) => {

                const x = (u - 0.5) * 10;

                const y = (v - 0.5) * 10;

                const z = functions[currentFunctionKey].func(x, y, a, b);

                target.set(x, z, y);

            };

            const geometry = new THREE.ParametricGeometry(graphFunction, 50, 50);

            const material = new THREE.MeshStandardMaterial({ color: 0x1e90ff, side: THREE.DoubleSide, transparent: true, opacity: 0.8 });

            surfaceMesh = new THREE.Mesh(geometry, material);

            scene.add(surfaceMesh);

            updatePoint();

        }

            const b = parseFloat(paramB_slider.value);

            const x = parseFloat(pointX_slider.value);

            const y = parseFloat(pointY_slider.value);

            const z = functions[currentFunctionKey].func(x, y, a, b);


            const surfacePointPos = new THREE.Vector3(x, z, y);

            const xyPlanePointPos = new THREE.Vector3(x, 0, y);


            pointSphere.position.copy(surfacePointPos);

            xyPlanePoint.position.copy(xyPlanePointPos);

            

            const positions = zLine.geometry.attributes.position.array;

            positions[0] = xyPlanePointPos.x; positions[1] = xyPlanePointPos.y; positions[2] = xyPlanePointPos.z;

            positions[3] = surfacePointPos.x; positions[4] = surfacePointPos.y; positions[5] = surfacePointPos.z;

            zLine.geometry.attributes.position.needsUpdate = true;

            zLine.computeLineDistances();

            zValueSpan.textContent = z.toFixed(3);

        }


        // --- UIの表示を更新 ---

        function updateUI() {

            const a = parseFloat(paramA_slider.value);

            const b = parseFloat(paramB_slider.value);

            const x = parseFloat(pointX_slider.value);

            const y = parseFloat(pointY_slider.value);


            aValueSpan.textContent = a.toFixed(1);

            bValueSpan.textContent = b.toFixed(1);

            xValueSpan.textContent = x.toFixed(1);

            yValueSpan.textContent = y.toFixed(1);


            const mathMLString = functions[currentFunctionKey].mathML(a, b);

            formulaDisplay.innerHTML = mathMLString;

        }


        // --- アニメーションループ ---

        function animate() {

            requestAnimationFrame(animate);

            controls.update();

            renderer.render(scene, camera);

        }

        

        // --- ウィンドウリサイズへの対応 ---

        function onWindowResize() {

            camera.aspect = canvasContainer.clientWidth / canvasContainer.clientHeight;

            camera.updateProjectionMatrix();

            renderer.setSize(canvasContainer.clientWidth, canvasContainer.clientHeight);

        }

        

        // --- イベントリスナーの設定 ---

        function setupEventListeners() {

            window.addEventListener('resize', onWindowResize);


            const visibilityToggle = document.getElementById('visibility-toggle');

            visibilityToggle.addEventListener('change', (event) => {

                const isVisible = event.target.checked;

                axesGroup.visible = isVisible;

                gridHelper.visible = isVisible;

            });


            // Populate custom select

            Object.keys(functions).forEach(key => {

                const option = document.createElement('div');

                option.classList.add('custom-option', 'relative', 'cursor-pointer', 'select-none', 'p-3', 'hover:bg-indigo-100');

                option.dataset.value = key;

                option.innerHTML = functions[key].displayMathML;

                

                option.addEventListener('click', () => {

                    currentFunctionKey = key;

                    customSelectDisplay.innerHTML = functions[key].displayMathML;

                    customSelectOptions.classList.add('hidden');

                    if (key === 'hills') {

                        paramA_slider.value = 2.0;

                    } else if (key === 'sincos') {

                        paramA_slider.value = 1.0;

                    }

                    updateSurface();

                    updateUI();

                });

                customSelectOptions.appendChild(option);

            });

            

            // Set initial display for custom select

            customSelectDisplay.innerHTML = functions[currentFunctionKey].displayMathML;


            // Custom select toggle

            customSelectButton.addEventListener('click', () => {

                customSelectOptions.classList.toggle('hidden');

            });


            // Close custom select when clicking outside

            document.addEventListener('click', (event) => {

                const container = document.getElementById('custom-select-container');

                if (!container.contains(event.target)) {

                    customSelectOptions.classList.add('hidden');

                }

            });



            paramA_slider.addEventListener('input', () => { updateSurface(); updateUI(); });

            paramB_slider.addEventListener('input', () => { updateSurface(); updateUI(); });

            pointX_slider.addEventListener('input', () => { updatePoint(); updateUI(); });

            pointY_slider.addEventListener('input', () => { updatePoint(); updateUI(); });

        }


        // --- 実行 ---

        init();
