        function createSurface() {

            const range = 2;

            const segments = 80;

            const geometry = new THREE.PlaneGeometry(range * 2, range * 2, segments, segments);

            geometry.rotateX(-Math.PI / 2); // Rotate to lie on the XZ plane


            const vertices = geometry.attributes.position.array;

            for (let i = 0; i < vertices.length; i += 3) {

                const x = vertices[i];     // Corresponds to math x

                const y = vertices[i + 2]; // Corresponds to math y

                const z = func(x, y);      // Corresponds to math z (height)

                vertices[i + 1] = z;       // Set height on the Scene's Y axis

            }

            geometry.attributes.position.needsUpdate = true;

            geometry.computeVertexNormals();

            

            const material = new THREE.MeshStandardMaterial({

                color: 0x818cf8, // Indigo color from reference

                side: THREE.DoubleSide,

                roughness: 0.7,

                metalness: 0.1,

                transparent: true,

                opacity: 0.9,

                wireframe: false

            });

            surfaceMesh = new THREE.Mesh(geometry, material);

            scene.add(surfaceMesh);

        }

        

        function createAxisLabel(text, position, color, background) {

            const canvas = document.createElement('canvas');

            const context = canvas.getContext('2d');

            const size = 256;

            canvas.width = size; canvas.height = size;


            context.font = `Bold 90px Arial`;

            context.textAlign = 'center';

            context.textBaseline = 'middle';

            

            context.fillStyle = background;

            context.arc(size / 2, size / 2, size / 2.2, 0, 2 * Math.PI);

            context.fill();


            context.fillStyle = color;

            context.fillText(text, size / 2, size / 2);


            const texture = new THREE.CanvasTexture(canvas);

            const material = new THREE.SpriteMaterial({ map: texture, depthTest: false });

            const sprite = new THREE.Sprite(material);

            sprite.scale.set(0.4, 0.4, 0.4);

            sprite.position.copy(position);

            scene.add(sprite);

        }


        function createThickAxes() {

            const axisLength = 2.5, axisRadius = 0.015, arrowRadius = 0.05, arrowHeight = 0.2;

            

            const createAxis = (color, direction) => {

                const material = new THREE.MeshStandardMaterial({ color, metalness: 0.3, roughness: 0.5 });

                const group = new THREE.Group();

                const cylinder = new THREE.Mesh(new THREE.CylinderGeometry(axisRadius, axisRadius, axisLength, 12), material);

                const cone = new THREE.Mesh(new THREE.ConeGeometry(arrowRadius, arrowHeight, 12), material);

                cone.position.y = axisLength / 2;

                group.add(cylinder, cone);

                if (!direction.equals(new THREE.Vector3(0, 1, 0))) {

                    group.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), direction);

                }

                scene.add(group);

            };

            

            const colorX = 0x60a5fa, colorY = 0xf87171, colorZ = 0x4ade80; // Blue, Red, Green

            createAxis(colorX, new THREE.Vector3(1, 0, 0)); // X-axis (math x)

            createAxis(colorY, new THREE.Vector3(0, 0, 1)); // Z-axis (math y)

            createAxis(colorZ, new THREE.Vector3(0, 1, 0)); // Y-axis (math z, height)


            const labelPos = axisLength / 2 + 0.3;

            createAxisLabel('x', new THREE.Vector3(labelPos, 0, 0), '#fff', '#60a5fa');

            createAxisLabel('y', new THREE.Vector3(0, 0, labelPos), '#fff', '#f87171');

            createAxisLabel('z', new THREE.Vector3(0, labelPos, 0), '#fff', '#4ade80');

        }


        function clearSliceVisuals() {

            if (axisLine) {

                scene.remove(axisLine);

                axisLine.geometry.dispose(); axisLine.material.dispose(); axisLine = null;

            }

            if (slicePlaneMesh) {

                scene.remove(slicePlaneMesh);

                slicePlaneMesh.geometry.dispose(); slicePlaneMesh.material.dispose(); slicePlaneMesh = null;

            }

        function updatePathLine(m) {

            if (pathLine) {

                scene.remove(pathLine);

                pathLine.geometry.dispose(); pathLine.material.dispose();

            }

            clearSliceVisuals();


            const points = [];

            const range = 2.1;

            for (let x = -range; x <= range; x += 0.04) {

                const y = m * x; // math y

                if(Math.sqrt(x*x + y*y) > range) continue;

                const z = func(x, y); // math z (height)

                points.push(new THREE.Vector3(x, z, y)); // map to scene (math x, math z, math y)

            }

            

            if (points.length < 2) return; // Not enough points to form a curve


            // Create a thick tube instead of a thin line for better visibility

            const curve = new THREE.CatmullRomCurve3(points);

            const tubeGeometry = new THREE.TubeGeometry(curve, 64, 0.025, 8, false); // path, segments, radius, radialSegments, closed

            const tubeMaterial = new THREE.MeshBasicMaterial({ color: 0xfacc15 }); // yellow-400

            pathLine = new THREE.Mesh(tubeGeometry, tubeMaterial);

            pathLine.visible = slopeToggle.checked;

            scene.add(pathLine);

        }

        

        function showAxisSlice(axis) {

            if (pathLine) {

                scene.remove(pathLine);

                pathLine.geometry.dispose(); pathLine.material.dispose(); pathLine = null;

            }

            clearSliceVisuals();


            const points = [];

            const range = 2.1;

            for (let p = -range; p <= range; p += 0.04) {

                 const x = (axis === 'x') ? p : 0;

                 const y = (axis === 'y') ? p : 0;

                 const z = func(x, y);

                 points.push(new THREE.Vector3(x, z, y));

            }

            

            const color = (axis === 'x') ? 0x60a5fa : 0xf87171; // Blue for x-slice, Red for y-slice

            const material = new THREE.LineBasicMaterial({ color: color, linewidth: 5 });

            const geometry = new THREE.BufferGeometry().setFromPoints(points);

            axisLine = new THREE.Line(geometry, material);

            scene.add(axisLine);


            const planeGeom = new THREE.PlaneGeometry(4.2, 3);

            const planeMat = new THREE.MeshBasicMaterial({ color: color, transparent: true, opacity: 0.2, side: THREE.DoubleSide });

            slicePlaneMesh = new THREE.Mesh(planeGeom, planeMat);

            

            if (axis === 'y') { // y-slice (math x=0) should be on the scene's YZ plane

                slicePlaneMesh.rotation.y = Math.PI / 2;

            }

            // For x-slice (math y=0), the plane should be on the scene's XY plane, which is the default for PlaneGeometry. No rotation needed.

            

            scene.add(slicePlaneMesh);


            const container2d = document.getElementById('slice-2d-container');

            const title2d = document.getElementById('slice-2d-title');

            container2d.classList.remove('hidden');

            title2d.textContent = axis === 'x' ? 'x軸による断面 (y=0)' : 'y軸による断面 (x=0)';

            draw2DSlice(axis);

        }


        function draw2DSlice(axis) {

            const canvas = document.getElementById('2d-slice-canvas');

            if (!canvas) return;

            const ctx = canvas.getContext('2d');


            // High-DPI scaling for crisp rendering

            const dpr = window.devicePixelRatio || 1;

            const rect = canvas.getBoundingClientRect();

            canvas.width = rect.width * dpr;

            canvas.height = rect.height * dpr;

            ctx.scale(dpr, dpr);

            const w = canvas.clientWidth;

            const h = canvas.clientHeight;

            

            const isDark = document.documentElement.classList.contains('dark');

            ctx.clearRect(0, 0, w, h);


            const range = 2; // Match 3D range

            const pixelsPerUnitX = w / (range * 2.2); // Give some padding

            const pixelsPerUnitY = h / 3; // z range is [-1, 1], give some padding


            const originX = w / 2;

            const originY = h / 2;

            

            ctx.save();

            ctx.translate(0.5, 0.5); // For sharper lines


            // Draw axes

            ctx.strokeStyle = isDark ? '#4a5568' : '#cbd5e0';

            ctx.lineWidth = 1;

            ctx.beginPath();

            ctx.moveTo(0, originY); ctx.lineTo(w, originY); // Horizontal axis

            ctx.moveTo(originX, 0); ctx.lineTo(originX, h); // Vertical axis

            ctx.stroke();

            

            // Draw ticks and labels

            ctx.fillStyle = isDark ? '#9ca3af' : '#6b7280';

            ctx.font = '10px Inter';

            ctx.textAlign = 'center';

            ctx.textBaseline = 'top';

            for (let i = -range; i <= range; i++) {

                if (i === 0) continue;

                const posX = originX + i * pixelsPerUnitX;

                ctx.beginPath();

                ctx.moveTo(posX, originY - 4); ctx.lineTo(posX, originY + 4);

                ctx.stroke();

                ctx.fillText(i, posX, originY + 8);

            }


            ctx.textAlign = 'left';

            ctx.textBaseline = 'middle';

            for (let i = -1; i <= 1; i++) {

                 if (i === 0) continue;

                const posY = originY - i * pixelsPerUnitY;

                ctx.beginPath();

                ctx.moveTo(originX - 4, posY); ctx.lineTo(originX + 4, posY);

                ctx.stroke();

                ctx.fillText(i, originX + 8, posY);

            }


            // Axis labels

            ctx.fillStyle = isDark ? '#cbd5e0' : '#4a5568';

            ctx.font = '13px Inter';

            const hLabel = axis === 'x' ? 'x' : 'y';

            const vLabel = 'z';

            ctx.textAlign = 'right';

            ctx.textBaseline = 'bottom';

            ctx.fillText(hLabel, w - 8, originY - 8);

            ctx.textAlign = 'left';

            ctx.textBaseline = 'top';

            ctx.fillText(vLabel, originX + 8, 8);

            

            // Draw function curve: f(p,0)=0

            ctx.strokeStyle = (axis === 'x') ? '#60a5fa' : '#f87171'; // Blue for x, Red for y

            ctx.lineWidth = 3;

            ctx.beginPath();

            ctx.moveTo(0, originY);

            ctx.lineTo(w, originY);

            ctx.stroke();


            // Draw tangent line at origin

            ctx.strokeStyle = '#facc15'; // yellow-400

            ctx.lineWidth = 2;

            ctx.setLineDash([5, 5]);

            ctx.beginPath();

            const tangentLength = 80; // pixels

            ctx.moveTo(originX - tangentLength / 2, originY);

            ctx.lineTo(originX + tangentLength / 2, originY);

            ctx.stroke();

            ctx.setLineDash([]);

            

            ctx.fillStyle = '#facc15';

            ctx.textAlign = 'center';

            ctx.textBaseline = 'bottom';

            ctx.fillText('接線 (傾き = 0)', originX, originY - 8);

            

            ctx.restore();

        }
