            <h2 class="text-2xl font-bold mb-4">微分方程式が描く「磁力線」も積分結果！</h2>

            <p class="text-base md:text-lg leading-relaxed">

                微分方程式の見方を変えてみましょう。 $$ \frac{dy}{dx} = f(x, y) $$ という式は、平面上の各点 $(x,y)$ における「進むべき方向（傾き）」を指定しています。<br><br>

                <strong>傾きの場（Vector Field）:</strong><br>

                この式に従って、平面全体に小さな矢印（砂鉄）を無数に描くと、それはまるで「流れ」や「場」のように見えます。<br><br>

                <strong>解曲線 ＝ 磁力線:</strong><br>

                この流れの中に身を任せたとき、描かれる軌跡が微分方程式の解（積分曲線）です。これは、砂鉄が作る磁力線に非常によく似ています。一本一本の砂鉄は傾きを表し、その連なりが全体の構造（磁力線）を形作ります。

            </p>

             <div class="mt-6 p-4 bg-indigo-50 border-l-4 border-indigo-500 rounded-r-lg">

                <p class="font-semibold text-indigo-800">🌌 核心的イメージ：積分するとは、局所的なルール（傾き）を繋ぎ合わせて、大域的な構造（解曲線）を発見する行為なのです。積分定数 C の違いは、どの磁力線を選ぶか（＝出発点をどこにするか）の違いに対応します。</p>

    <script>

        document.addEventListener('DOMContentLoaded', () => {

            if (typeof renderMathInElement === 'function') {

                renderMathInElement(document.body);

            }


            const canvas = document.getElementById('vector-field-canvas');

            const ctx = canvas.getContext('2d');

            const clearButton = document.getElementById('clear-button');

            const modeSelector = document.getElementById('mode-selector');


            let width, height, dpr;

            let curves = [];

            let currentMode = 'single';

            let poles = [];


            const poleConfigs = {

                single: [

                    { x: -4, y: 0, charge: 1 }, { x: 4, y: 0, charge: -1 }

                ],

                plate: [

                    { x: -3, y: 4, charge: 1 }, { x: 3, y: 4, charge: -1 }, // Real magnet

                    { x: -3, y: -4, charge: -1 }, { x: 3, y: -4, charge: 1 } // Image magnet

                ],

                attract_opposite: [

                    { x: -6, y: 0, charge: 1 }, { x: -2, y: 0, charge: -1 },

                    { x: 2, y: 0, charge: 1 }, { x: 6, y: 0, charge: -1 }

                ],

                repel_opposite: [

                    { x: -6, y: 0, charge: -1 }, { x: -2, y: 0, charge: 1 },

                    { x: 2, y: 0, charge: 1 }, { x: 6, y: 0, charge: -1 }

                ],

                attract_parallel: [

                    { x: -4, y: 3, charge: 1 }, { x: 4, y: 3, charge: -1 },

                    { x: -4, y: -3, charge: -1 }, { x: 4, y: -3, charge: 1 }

                ],

                repel_parallel: [

                    { x: -4, y: 3, charge: 1 }, { x: 4, y: 3, charge: -1 },

                    { x: -4, y: -3, charge: 1 }, { x: 4, y: -3, charge: -1 }

                ]

            };


            const getDirectionVector = (x, y) => {

                let total_dx = 0;

                let total_dy = 0;

                

                poles.forEach(pole => {

                    const dx = x - pole.x;

                    const dy = y - pole.y;

                    const dist_sq = dx * dx + dy * dy;

                    if (dist_sq > 0.01) {

                        const strength = pole.charge / dist_sq;

                        total_dx += dx * strength;

                        total_dy += dy * strength;

                    }

                });

                return { dx: total_dx, dy: total_dy };

            };


            function setupCanvas() {

                dpr = window.devicePixelRatio || 1;

                const rect = canvas.parentElement.getBoundingClientRect();

                width = rect.width;

                height = rect.height;

                canvas.width = width * dpr;

                canvas.height = height * dpr;

                canvas.style.width = `${width}px`;

                canvas.style.height = `${height}px`;

                ctx.scale(dpr, dpr);

                setMode(currentMode);

            }


            const toMathX = (cx) => (cx - width / 2) / (width / 25);

            const toMathY = (cy) => -(cy - height / 2) / (height / 25);

            const toCanvasX = (mx) => mx * (width / 25) + width / 2;

            const toCanvasY = (my) => -my * (height / 25) + height / 2;


            function drawVectorField() {

                const gridSize = 20;

                const lineLength = 10;

                ctx.strokeStyle = 'rgba(107, 114, 128, 0.5)';

                ctx.lineWidth = 1;


                for (let i = 0; i < width; i += gridSize) {

                    for (let j = 0; j < height; j += gridSize) {

                        const mx = toMathX(i);

                        const my = toMathY(j);


                        if (currentMode === 'plate' && my < 0) continue;

                        

                        const vec = getDirectionVector(mx, my);

                        const magnitude = Math.sqrt(vec.dx * vec.dx + vec.dy * vec.dy);

                        if (magnitude === 0) continue;


                        const dx_norm = vec.dx / magnitude;

                        const dy_norm = vec.dy / magnitude;


                        ctx.beginPath();

                        ctx.moveTo(i - dx_norm * lineLength / 2, j - dy_norm * lineLength / 2);

                        ctx.lineTo(i + dx_norm * lineLength / 2, j + dy_norm * lineLength / 2);

                        ctx.stroke();

                    }

                }

            }

            

            function drawPoles() {

                poles.forEach(pole => {

                    if (currentMode === 'plate' && pole.y < 0) return;


                    const cx = toCanvasX(pole.x);

                    const cy = toCanvasY(pole.y);

                    const color = pole.charge > 0 ? 'rgba(239, 68, 68, 0.9)' : 'rgba(59, 130, 246, 0.9)';

                    const text = pole.charge > 0 ? 'N' : 'S';


                    ctx.fillStyle = color;

                    ctx.beginPath();

                    ctx.arc(cx, cy, 8, 0, 2 * Math.PI);

                    ctx.fill();

                    ctx.fillStyle = 'white';

                    ctx.font = 'bold 12px Inter';

                    ctx.textAlign = 'center';

                    ctx.textBaseline = 'middle';

                    ctx.fillText(text, cx, cy);

                });

            }

            

            function drawIronPlate() {

                if (currentMode !== 'plate') return;

                const plateSurfaceY = toCanvasY(0);

                ctx.fillStyle = '#4b5563';

                ctx.fillRect(0, plateSurfaceY, width, height - plateSurfaceY);

            }


            function traceCurve(startPoint) {

                const stepSize = 0.1;

                const numSteps = 2000;

                const curvePoints = [];

                const sinks = poles.filter(p => p.charge < 0);

                const sources = poles.filter(p => p.charge > 0);


                // 正の方向にトレース

                let x = startPoint.x;

                let y = startPoint.y;

                for (let i = 0; i < numSteps; i++) {

                    curvePoints.push({ x, y });

                    const vec = getDirectionVector(x, y);

                    const magnitude = Math.sqrt(vec.dx * vec.dx + vec.dy * vec.dy);

                    if (magnitude < 0.01) break;

                    

                    x += (vec.dx / magnitude) * stepSize;

                    y += (vec.dy / magnitude) * stepSize;

                    

                    let stopped = sinks.some(sink => Math.hypot(x - sink.x, y - sink.y) < 0.5);

                    if (currentMode === 'plate' && y < 0) stopped = true;

                    if (stopped || Math.abs(x) > 15 || Math.abs(y) > 15) break;

                }


                // 負の方向にトレース

                x = startPoint.x;

                y = startPoint.y;

                for (let i = 0; i < numSteps; i++) {

                    curvePoints.unshift({ x, y });

                    const vec = getDirectionVector(x, y);

                    const magnitude = Math.sqrt(vec.dx * vec.dx + vec.dy * vec.dy);

                    if (magnitude < 0.01) break;

                    

                    x -= (vec.dx / magnitude) * stepSize;

                    y -= (vec.dy / magnitude) * stepSize;


                    let stopped = sources.some(source => Math.hypot(x - source.x, y - source.y) < 0.5);

                    if (currentMode === 'plate' && y < 0) stopped = true;

                    if (stopped || Math.abs(x) > 15 || Math.abs(y) > 15) break;

                }

                

                return curvePoints;

            }


            function renderCurve(points, color) {

                if (points.length < 2) return;

                ctx.beginPath();

                ctx.strokeStyle = color;

                ctx.lineWidth = 2.5;

                

                const startCanvasPoint = { x: toCanvasX(points[0].x), y: toCanvasY(points[0].y) };

                ctx.moveTo(startCanvasPoint.x, startCanvasPoint.y);


                for (let i = 1; i < points.length; i++) {

                    const canvasPoint = { x: toCanvasX(points[i].x), y: toCanvasY(points[i].y) };

                    ctx.lineTo(canvasPoint.x, canvasPoint.y);

                }

                ctx.stroke();

            }


            function drawScene() {

                ctx.clearRect(0, 0, width, height);

                drawVectorField();

                drawIronPlate();

                drawPoles();

                curves.forEach(curveData => {

                    renderCurve(curveData.points, curveData.color);

                });

            }


            function handlePointerDown(e) {

                const rect = canvas.getBoundingClientRect();

                const x = e.clientX - rect.left;

                const y = e.clientY - rect.top;


                const startPoint = { x: toMathX(x), y: toMathY(y) };

                if (currentMode === 'plate' && startPoint.y < 0) return;


                const newCurvePoints = traceCurve(startPoint);

                const randomColor = `hsl(${Math.random() * 360}, 80%, 60%)`;

                curves.push({ points: newCurvePoints, color: randomColor });

                

                drawScene();

            }

            

            function setMode(mode) {

                currentMode = mode;

                poles = poleConfigs[mode];

                curves = [];

                

                document.querySelectorAll('.btn-mode').forEach(btn => {

                    btn.classList.toggle('active', btn.dataset.mode === mode);

                });

                

                drawScene();

            }


            modeSelector.addEventListener('click', (e) => {

                if (e.target.classList.contains('btn-mode')) {

                    setMode(e.target.dataset.mode);

                }

            });


            canvas.addEventListener('pointerdown', handlePointerDown);

            clearButton.addEventListener('click', () => {

                curves = [];

                drawScene();

            });

            window.addEventListener('resize', setupCanvas);


            setupCanvas();

        });
