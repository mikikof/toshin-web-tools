        // --- ユーティリティ関数 ---

        const parabolaRadius = 0.05;

        

        function createParabolaPoints2D(a_val) {

            const points = [];

            if (Math.abs(a_val) < 0.01) a_val = 0.01 * (a_val >= 0 ? 1 : -1);

            const xMax = Math.sqrt(Math.abs(5 / a_val));

            for (let i = -100; i <= 100; i++) {

                const x = (i / 100) * xMax;

                const y = a_val * x * x;

                if (Math.abs(y) <= 5) points.push(new THREE.Vector3(x, y, 0));

            }

            return points;

        }

        

        function createParabolaPoints3D(a_val) {

            const points = [];

            if (Math.abs(a_val) < 0.01) a_val = 0.01 * (a_val >= 0 ? 1 : -1);

            const scale = 1 / 4;

            const xMax = Math.sqrt(Math.abs(5 / (a_val * scale)));

            for (let i = -100; i <= 100; i++) {

                const x = (i / 100) * xMax;

                const y_output = a_val * x * x * scale;

                if (Math.abs(y_output) <= 5) points.push(new THREE.Vector3(x, y_output, 0));

            }

            return points;

        }


        function createParabolaTube(points, material) {

            if (points.length < 2) return new THREE.Mesh();

            const curve = new THREE.CatmullRomCurve3(points);

            const geometry = new THREE.TubeGeometry(curve, 64, parabolaRadius, 8, false);

            return new THREE.Mesh(geometry, material);

        }

        

        function updateParabolaTube(tubeMesh, points) {

             if (points.length < 2) { tubeMesh.visible = false; return; }

             tubeMesh.visible = true;

             const curve = new THREE.CatmullRomCurve3(points);

             tubeMesh.geometry.dispose();

             tubeMesh.geometry = new THREE.TubeGeometry(curve, 64, parabolaRadius, 8, false);

        }


        function createAxes(size, thickness) {

            const axes = new THREE.Group();

            const materials = [ new THREE.MeshBasicMaterial({ color: 0x38bdf8 }), new THREE.MeshBasicMaterial({ color: 0x4ade80 }) ];

            const geometries = [ new THREE.BoxGeometry(size, thickness, thickness), new THREE.BoxGeometry(thickness, size, thickness) ];

            const xAxis = new THREE.Mesh(geometries[0], materials[0]);

            xAxis.position.x = size / 2;

            const yAxis = new THREE.Mesh(geometries[1], materials[1]);

            yAxis.position.y = size / 2;

            axes.add(xAxis, yAxis);

            return axes;

        }


        function createAxes3D(size, thickness) {

            const axes = new THREE.Group();

            const materials = [ new THREE.MeshBasicMaterial({ color: 0x38bdf8 }), new THREE.MeshBasicMaterial({ color: 0x4ade80 }), new THREE.MeshBasicMaterial({ color: 0xfbbf24 }) ]; // x, y, a

            const geometries = [ new THREE.BoxGeometry(size, thickness, thickness), new THREE.BoxGeometry(thickness, size, thickness), new THREE.BoxGeometry(thickness, thickness, size) ];

            const xAxis = new THREE.Mesh(geometries[0], materials[0]); // x-axis

            xAxis.position.x = size / 2;

            const yAxis = new THREE.Mesh(geometries[1], materials[1]); // y-axis (output)

            yAxis.position.y = size / 2;

            const aAxis = new THREE.Mesh(geometries[2], materials[2]); // a-axis (parameter, depth)

            aAxis.position.z = -size / 2;

            axes.add(xAxis, yAxis, aAxis);

            return axes;

        }


        // --- グラフ作成 ---

        function create2DGraph() {

            axes2D = createAxes(10, 0.02);

            scene.add(axes2D);

            const material = new THREE.MeshBasicMaterial({ color: currentColors.parabola });

            const points = createParabolaPoints2D(params.a);

            parabola = createParabolaTube(points, material);

            scene.add(parabola);

        }


        function create3DGraph() {

            const surfaceFunc = (u, v, target) => {

                const x = (u - 0.5) * 8; // x-axis

                const a_param = (v - 0.5) * 8; // a-axis (parameter)

                const y_output = a_param * x * x / 4; // y-axis (output)

                target.set(x, y_output, -a_param); // map (x, y_out, a_in) to (x, y, z)

            };

            const geometry = new THREE.ParametricGeometry(surfaceFunc, 50, 50);

            const material = new THREE.MeshStandardMaterial({ color: currentColors.surface, side: THREE.DoubleSide, metalness: 0.3, roughness: 0.6, transparent: true, opacity: 0.8 });

            surface = new THREE.Mesh(geometry, material);

            surface.visible = false;

            scene.add(surface);


            axes3D = createAxes3D(5, 0.03);

            axes3D.visible = false;

            scene.add(axes3D);

            

            gridHelper = new THREE.GridHelper(10, 10, currentColors.grid, currentColors.grid);

            gridHelper.rotation.x = Math.PI / 2;

            gridHelper.visible = false;

            scene.add(gridHelper);


            const crossSectionMaterial = new THREE.MeshBasicMaterial({ color: currentColors.crossSection });

            const points = createParabolaPoints3D(params.a_var);

            crossSectionLine = createParabolaTube(points, crossSectionMaterial);

            crossSectionLine.visible = false;

            scene.add(crossSectionLine);


            const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);

            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);

            directionalLight.position.set(5, 10, 7.5);

            scene.add(directionalLight);

        }

        function updateParabola() {

            const points = createParabolaPoints2D(params.a);

        function addGhostParabola() {

            if (ghostParabolas.length > 20) {

                const oldGhost = ghostParabolas.shift();

                scene.remove(oldGhost);

                oldGhost.geometry.dispose();

                oldGhost.material.dispose();

            }

            const ghostMaterial = new THREE.MeshBasicMaterial({ color: currentColors.ghost, transparent: true, opacity: 0.25 });

            const points = createParabolaPoints2D(params.a_var);

            const ghost = createParabolaTube(points, ghostMaterial);

            ghostParabolas.push(ghost);

            scene.add(ghost);

        }

        

        function clearGhostParabolas() {

            ghostParabolas.forEach(ghost => {

                scene.remove(ghost);

                ghost.geometry.dispose();

                ghost.material.dispose();

            });

            ghostParabolas.length = 0;

        }

        // --- スクロールアニメーション ---

        function setupScrollAnimations() {

            gsap.registerPlugin(ScrollTrigger);

            sections.forEach((section, i) => {

                if (i >= sections.length -1) return;

                const panel = section.querySelector('.text-panel, .flex.items-center.justify-center');

                gsap.fromTo(panel, 

                    { opacity: 0, y: 50 },

                    {

                        opacity: 1, y: 0, duration: 1,

                        scrollTrigger: {

                            trigger: section, start: 'top center', end: 'center center', scrub: 1,

                            onEnter: () => handleSectionChange(i),

                            onEnterBack: () => handleSectionChange(i),

                        }

                    }

                );

            });

        }


        function handleSectionChange(index) {

            currentSection = index;

            document.querySelectorAll('.scroll-down-indicator').forEach((indicator, i) => {

                 gsap.to(indicator, { opacity: i === index && i < sections.length - 2 ? 1 : 0, duration: 0.5 });

            });


            if (index < 3) {

                if (params.is3D) transitionTo2D();

                const newParam = index === 2 ? params.a_var : params.a;

                const points = createParabolaPoints2D(newParam);

                updateParabolaTube(parabola, points);

                if (index < 2) clearGhostParabolas();

            } else {

                if (!params.is3D) transitionTo3D();

            }

        }

        

        function transitionTo3D() {

            params.is3D = true;

            controls.enabled = true;

            controls.target.set(0, 0, 0);

            gsap.to(camera.position, { 

                duration: 2, x: params.cameraPos3D.x, y: params.cameraPos3D.y, z: params.cameraPos3D.z, ease: 'power3.inOut',

                onUpdate: () => { camera.lookAt(0, 0, 0); }

            });

            parabola.visible = false;

            ghostParabolas.forEach(g => g.visible = false);

            axes2D.visible = false;

            surface.visible = true;

            axes3D.visible = true;

            gridHelper.visible = true;

            crossSectionLine.visible = true;

            gsap.fromTo(surface.material, {opacity: 0}, {opacity: 0.8, duration: 1.5, delay: 0.5});

        }


        function transitionTo2D() {

            params.is3D = false;

            controls.enabled = false;

            gsap.timeline()

                .to(camera.position, { duration: 2, x: params.cameraPos2D.x, y: params.cameraPos2D.y, z: params.cameraPos2D.z, ease: 'power3.inOut' })

                .to(camera.rotation, { duration: 2, x: 0, y: 0, z: 0, ease: 'power3.inOut' }, "<");

            parabola.visible = true;

            ghostParabolas.forEach(g => g.visible = currentSection === 2);

            axes2D.visible = true;

            surface.visible = false;

            axes3D.visible = false;

            gridHelper.visible = false;

            crossSectionLine.visible = false;

        }


        // --- テーマ切り替え ---

        function setupTheme() {

            const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;

            const savedTheme = localStorage.getItem('theme');

            if (savedTheme === 'dark' || (!savedTheme && prefersDark)) {

                setTheme('dark');

            } else {

                setTheme('light');

            }


            ui.toggleThemeBtn.addEventListener('click', () => {

                const newTheme = document.body.classList.contains('dark') ? 'light' : 'dark';

                setTheme(newTheme);

            });

        }


        function setTheme(theme) {

            localStorage.setItem('theme', theme);

            currentColors = colors[theme];


            if (theme === 'dark') {

                document.body.classList.add('dark');

                ui.iconSun.classList.add('hidden');

                ui.iconMoon.classList.remove('hidden');

            } else {

                document.body.classList.remove('dark');

                ui.iconSun.classList.remove('hidden');

                ui.iconMoon.classList.add('hidden');

            }

            updateAllColors();

        }


        function updateAllColors() {

            if (!parabola) return; // 未初期化の場合は何もしない

            parabola.material.color.set(currentColors.parabola);

            surface.material.color.set(currentColors.surface);

            crossSectionLine.material.color.set(currentColors.crossSection);
