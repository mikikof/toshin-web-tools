    function usePlot(domain, yRange) {

      const width = 760; // コンテンツでの標準幅

      const height = 320;

      const pad = { l: 48, r: 20, t: 20, b: 36 };

      const [xmin, xmax] = domain;

      const [ymin, ymax] = yRange;

      const w = width - pad.l - pad.r;

      const h = height - pad.t - pad.b;

      const X = (x) => pad.l + ((x - xmin) / (xmax - xmin)) * w;

      const Y = (y) => pad.t + h - ((y - ymin) / (ymax - ymin)) * h;

      return { width, height, pad, X, Y, xmin, xmax, ymin, ymax, w, h };

    }


    function samplePath(f, X, Y, xmin, xmax, n = 400) {

      const step = (xmax - xmin) / (n - 1);

      let d = "";

      for (let i = 0; i < n; i++) {

        const x = xmin + i * step;

        const y = f(x);

        const px = X(x);

        const py = Y(y);

        d += i === 0 ? `M ${px} ${py}` : ` L ${px} ${py}`;

      }

      return d;

    }


    // 区間[a,b]内で f'(c) = m を満たす c を（存在すれば）探す：符号変化/最小値探索

    function findC(df, m, a, b) {

      const N = 800;

      let prev = df(a) - m;

      let prevX = a;

      let bestX = a;

      let bestVal = Math.abs(prev);

      for (let i = 1; i <= N; i++) {

        const x = a + ((b - a) * i) / N;

        const val = df(x) - m;

        const abs = Math.abs(val);

        if (abs < bestVal) {

          bestVal = abs;

          bestX = x;

        }

        if (val === 0) return x;

        if (prev * val < 0) { // 線形補間

          const t = prev / (prev - val);

          return prevX + t * (x - prevX);

        }

        prev = val;

        prevX = x;

      }

      return bestX; // 近い点

    }


    // 軸描画

    function Axes({ plot, ticks = 6 }) {

      const { X, Y, xmin, xmax, ymin, ymax, pad, width, height } = plot;

      const g = [];

      // x軸

      if (ymin < 0 && ymax > 0) {

        const y0 = Y(0);

        g.push(<line key="xaxis" x1={pad.l} y1={y0} x2={width - pad.r} y2={y0} className="stroke-gray-300" />);

      }

      // y軸

      if (xmin < 0 && xmax > 0) {

        const x0 = X(0);

        g.push(<line key="yaxis" x1={x0} y1={pad.t} x2={x0} y2={height - pad.b} className="stroke-gray-300" />);

      }

      // 目盛（y）

      for (let i = 0; i <= ticks; i++) {

        const y = ymin + (i * (ymax - ymin)) / ticks;

        const py = Y(y);

        g.push(

          <g key={`yt${i}`}>

            <line x1={pad.l - 4} y1={py} x2={pad.l} y2={py} className="stroke-gray-300" />

            <text x={pad.l - 8} y={py + 4} className="fill-gray-400 text-xs" textAnchor="end">{fmt(y, 2)}</text>

          </g>

        );

      }

      // 目盛（x）

      for (let i = 0; i <= ticks; i++) {

        const x = xmin + (i * (xmax - xmin)) / ticks;

        const px = X(x);

        g.push(

          <g key={`xt${i}`}>

            <line x1={px} y1={height - pad.b} x2={px} y2={height - pad.b + 4} className="stroke-gray-300" />

            <text x={px} y={height - pad.b + 16} className="fill-gray-400 text-xs" textAnchor="middle">{fmt(x, 2)}</text>

          </g>

        );

      }

      return <g>{g}</g>;

    }

    function DemoMVT() {

      const [funcKey, setFuncKey] = useState("sin");

      const F = FUNCS[funcKey];

      const [a, setA] = useState(() => F.domain[0] * 0.3);

      const [b, setB] = useState(() => F.domain[1] * 0.3);


      useEffect(() => {

        // 関数切替時に範囲初期化

        setA(F.domain[0] * 0.3);

        setB(F.domain[1] * 0.3);

      }, [funcKey]);


      const plot = usePlot(F.domain, F.niceY);

      const { X, Y, xmin, xmax } = plot;

      const aClamped = clamp(Math.min(a, b - 1e-6), xmin, xmax);

      const bClamped = clamp(Math.max(b, a + 1e-6), xmin, xmax);


      const mSec = (F.f(bClamped) - F.f(aClamped)) / (bClamped - aClamped);

      const c = findC(F.df, mSec, aClamped, bClamped);


      const path = useMemo(() => samplePath(F.f, X, Y, xmin, xmax, 500), [funcKey, X, Y, xmin, xmax]);


      const secLine = {

        x1: X(aClamped),

        y1: Y(F.f(aClamped)),

        x2: X(bClamped),

        y2: Y(F.f(bClamped)),

      };


      // 接線（c, f(c)）

      const fc = F.f(c);

      const dfc = F.df(c);

      const Lx1 = xmin, Lx2 = xmax;

      const Ly1 = fc + dfc * (Lx1 - c);

      const Ly2 = fc + dfc * (Lx2 - c);


      return (

        <Card

          title="平均値の定理を可視化"

          subtitle="割線の傾き = どこかの接線の傾き。スライダーで区間[a,b]を動かそう。"

          right={

            <select

              value={funcKey}

              onChange={(e) => setFuncKey(e.target.value)}

              className="border rounded-md text-sm px-2 py-1"

            >

              {Object.values(FUNCS).map((f) => (

                <option key={f.key} value={f.key}>{f.name}</option>

              ))}

            </select>

          }

        >

          <div className="flex flex-col gap-4">

            <svg width={plot.width} height={plot.height} className="w-full">

              <rect x={0} y={0} width={plot.width} height={plot.height} rx={14} className="fill-gray-50" />

              <Axes plot={plot} />

              {/* 関数グラフ */}

              <path d={path} className="stroke-gray-700 fill-none" strokeWidth={2} />


              {/* 割線 */}

              <line {...secLine} className="stroke-amber-500" strokeWidth={2} />

              <circle cx={secLine.x1} cy={secLine.y1} r={4} className="fill-amber-500" />

              <circle cx={secLine.x2} cy={secLine.y2} r={4} className="fill-amber-500" />


              {/* 接線 at c */}

              <line x1={plot.X(Lx1)} y1={plot.Y(Ly1)} x2={plot.X(Lx2)} y2={plot.Y(Ly2)} className="stroke-sky-600" strokeWidth={2} />

              <circle cx={plot.X(c)} cy={plot.Y(fc)} r={5} className="fill-sky-600" />

            </svg>


            <div className="grid grid-cols-1 md:grid-cols-2 gap-3">

              <Slider label="a" min={F.domain[0]} max={F.domain[1]} value={aClamped} onChange={setA} />

              <Slider label="b" min={F.domain[0]} max={F.domain[1]} value={bClamped} onChange={setB} />

            </div>


            <div className="text-sm text-gray-700 grid grid-cols-1 md:grid-cols-3 gap-3">

              <div className="p-3 bg-gray-50 rounded-xl">割線の傾き  m = {(fmt(mSec, 4))}</div>

              <div className="p-3 bg-gray-50 rounded-xl">見つかった c ≈ {fmt(c, 4)}</div>

            </div>

            <p className="text-sm text-gray-500">常に m = f'(c) となる c が区間 (a,b) 内に少なくとも1つ存在します（平均値の定理）。</p>

        </Card>

    // 2) リプシッツ上界 |f(b)-f(a)| <= M|b-a|（速度制限の直感つき）

    function DemoLipschitz() {

      const [funcKey, setFuncKey] = useState("sin");

      const [origin, setOrigin] = useState("a"); // 帯の起点 a / b

      const F = FUNCS[funcKey];

      const [a, setA] = useState(() => F.domain[0] * 0.4);

      const [b, setB] = useState(() => F.domain[1] * 0.4);


      useEffect(() => { setA(F.domain[0] * 0.4); setB(F.domain[1] * 0.4); }, [funcKey]);


      const plot = usePlot(F.domain, F.niceY);

      const { X, Y, xmin, xmax, pad } = plot;

      const aC = clamp(Math.min(a, b - 1e-6), xmin, xmax);

      const bC = clamp(Math.max(b, a + 1e-6), xmin, xmax);

      const lo = Math.min(aC, bC), hi = Math.max(aC, bC);


      // M = sup_{[a,b]} |f'|

      const M = useMemo(() => {

        let m = 0; const N = 512;

        for (let i = 0; i <= N; i++) {

          const x = lo + ((hi - lo) * i) / N;

          m = Math.max(m, Math.abs(F.df(x)));

        }

        return m;

      }, [funcKey, lo, hi]);


      const lhs = Math.abs(F.f(bC) - F.f(aC));

      const rhs = M * Math.abs(bC - aC);


      const path = useMemo(() => samplePath(F.f, X, Y, xmin, xmax, 500), [funcKey, X, Y, xmin, xmax]);


      // 帯の起点を選択

      const oX = origin === "a" ? aC : bC; // origin x

      const oY = F.f(oX);

      const otherX = origin === "a" ? bC : aC;

      const upper = (x) => oY + M * (x - oX);

      const lower = (x) => oY - M * (x - oX);


      // 影の帯ポリゴン

      const bandPoly = (() => {

        const pts = [

          [X(lo), Y(upper(lo))],

          [X(hi), Y(upper(hi))],

          [X(hi), Y(lower(hi))],

          [X(lo), Y(lower(lo))],

        ];

        return `M ${pts[0][0]} ${pts[0][1]} L ${pts[1][0]} ${pts[1][1]} L ${pts[2][0]} ${pts[2][1]} L ${pts[3][0]} ${pts[3][1]} Z`;

      })();


      // 他端の“許容範囲”縦バー

      const barX = X(otherX);

      const barTop = Y(upper(otherX));

      const barBot = Y(lower(otherX));


      // 曲線が帯内にいるかのチェック

      const inside = useMemo(() => {

        const N = 200;

        for (let i = 0; i <= N; i++) {

          const x = lo + ((hi - lo) * i) / N;

          const y = F.f(x);

          if (y > upper(x) + 1e-9 || y < lower(x) - 1e-9) return false;

        }

        return true;

      }, [lo, hi, M, funcKey, origin]);


      // ラベル用文字列

      const delta = Math.abs(otherX - oX);

      const labelBand = `帯:  y = f(${origin}) ± M (x − ${origin})`;

      const labelBarTop = `上端 ≈ ${fmt(upper(otherX),3)}`;

      const labelBarBot = `下端 ≈ ${fmt(lower(otherX),3)}`;


      return (

        <Card

          title="増分の上界（リプシッツ）"

          subtitle="M は“1 歩 x に進むごとに y は高々 M だけしか動けない”。影の帯が到達可能域を示す。"

          right={

            <div className="flex items-center gap-3">

              <select value={funcKey} onChange={(e)=>setFuncKey(e.target.value)} className="border rounded-md text-sm px-2 py-1">

                {Object.values(FUNCS).map(f=> <option key={f.key} value={f.key}>{f.name}</option>)}

              </select>

              <div className="flex items-center gap-2 text-sm">

                <span className="text-gray-600">起点</span>

                <label className="flex items-center gap-1"><input type="radio" name="origin" value="a" checked={origin==='a'} onChange={()=>setOrigin('a')} />a</label>

                <label className="flex items-center gap-1"><input type="radio" name="origin" value="b" checked={origin==='b'} onChange={()=>setOrigin('b')} />b</label>

              </div>

            </div>

          }

        >

          <div className="flex flex-col gap-4">

            <svg width={plot.width} height={plot.height} className="w-full">

              <rect x={0} y={0} width={plot.width} height={plot.height} rx={14} className="fill-gray-50" />

              <Axes plot={plot} />


              {/* 速度制限の影の帯（到達可能域） */}

              <path d={bandPoly} className="fill-amber-200 opacity-50" />


              {/* ラベル（左上に帯の式） */}

              <text x={pad.l + 8} y={pad.t + 14} className="fill-amber-700 text-[11px]">{labelBand}</text>

              <text x={pad.l + 8} y={pad.t + 28} className="fill-emerald-700 text-[11px]">M ≈ {fmt(M,3)},  |Δx| = {fmt(delta,3)}</text>


              {/* 曲線と割線 */}

              <path d={path} className="stroke-gray-700 fill-none" strokeWidth={2} />

              <line x1={X(aC)} y1={Y(F.f(aC))} x2={X(bC)} y2={Y(F.f(bC))} className="stroke-amber-500" strokeWidth={2} />


              {/* a, b と許容範囲バー（他端） */}

              <circle cx={X(aC)} cy={Y(F.f(aC))} r={4} className="fill-emerald-600" />

              <circle cx={X(bC)} cy={Y(F.f(bC))} r={4} className="fill-emerald-600" />

              <line x1={barX} y1={barBot} x2={barX} y2={barTop} className="stroke-amber-600" strokeDasharray="4 4" />

              <text x={barX + 6} y={barTop - 4} className="fill-amber-700 text-[10px]">{labelBarTop}</text>

              <text x={barX + 6} y={barBot + 12} className="fill-amber-700 text-[10px]">{labelBarBot}</text>

            </svg>


            <div className="grid grid-cols-1 md:grid-cols-2 gap-3">

    // 2.5) 道予測：Lipschitz に基づく「取りうる道」の可視化（真の関数を隠して考える + ベイズ風アップデート）

    function DemoLipschitzPredict() {

      const [funcKey, setFuncKey] = useState("sin");

      const [origin, setOrigin] = useState("a");

      const [hideTruth, setHideTruth] = useState(true);

      const [useManualM, setUseManualM] = useState(true);

      const [manualM, setManualM] = useState(1);

      const [nPaths, setNPaths] = useState(5);

      const [seed, setSeed] = useState(1);


      // 観測の有無・位置・ノイズ

      const [useObs, setUseObs] = useState(false);

      const [obsX, setObsX] = useState(0); // 初期値は後で更新

      const [eps, setEps] = useState(0);


      const F = FUNCS[funcKey];

      const [a, setA] = useState(() => F.domain[0] * 0.3);

      const [b, setB] = useState(() => F.domain[1] * 0.3);


      useEffect(() => { setA(F.domain[0] * 0.3); setB(F.domain[1] * 0.3); }, [funcKey]);

      useEffect(() => { setObsX((F.domain[0] + F.domain[1]) / 2); setEps(0); }, [funcKey]);


      const plot = usePlot(F.domain, F.niceY);

      const { X, Y, xmin, xmax, pad, ymin, ymax } = plot;

      const aC = clamp(Math.min(a, b - 1e-6), xmin, xmax);

      const bC = clamp(Math.max(b, a + 1e-6), xmin, xmax);

      const lo = Math.min(aC, bC), hi = Math.max(aC, bC);


      // 自動M（参考値）

      const Mauto = useMemo(() => {

        let m = 0; const N = 512;

        for (let i = 0; i <= N; i++) {

          const x = lo + ((hi - lo) * i) / N;

          m = Math.max(m, Math.abs(F.df(x)));

        }

        return m;

      }, [funcKey, lo, hi]);


      const M = useManualM ? manualM : Mauto;


      // 起点と帯

      const oX = origin === "a" ? aC : bC;

      const oY = F.f(oX);

      const otherX = origin === "a" ? bC : aC;

      const upperBase = (x) => oY + M * (x - oX);

      const lowerBase = (x) => oY - M * (x - oX);


      // 観測点と更新帯（交差）

      const xObs = clamp(obsX, lo, hi);

      const yObs = F.f(xObs);

      const upperObs = (x) => yObs + eps + M * Math.abs(x - xObs);

      const lowerObs = (x) => yObs - eps - M * Math.abs(x - xObs);


      const upper = (x) => useObs ? Math.min(upperBase(x), upperObs(x)) : upperBase(x);

      const lower = (x) => useObs ? Math.max(lowerBase(x), lowerObs(x)) : lowerBase(x);


      // 影の帯（基準）

      const bandBase = (() => {

        const L = X(lo), R = X(hi);

        return `M ${L} ${Y(upperBase(lo))} L ${R} ${Y(upperBase(hi))} L ${R} ${Y(lowerBase(hi))} L ${L} ${Y(lowerBase(lo))} Z`;

        const up = ts.map((t)=> upper(t));

        const loArr = ts.map((t)=> lower(t));

        const feasible = up.every((u, i)=> u >= loArr[i] - 1e-9);

        // パス生成（上を左→右、下を右→左）

        const pL = loArr.map((y,i)=> `L ${X(ts[N-1-i])} ${Y(loArr[N-1-i])}`);

        const d = `${pU.join(' ')} ${pL.join(' ')} Z`;

      function rng() {

      // 「取りうる道」を生成：|Δy| ≤ M|Δx| を積み上げる。帯からはみ出さないよう clamp。

          const s = (rand() * 2 - 1) * M; // slope in [-M, M]

          ny = Math.max(loY, Math.min(hiY, ny));

          pts.push([nx, ny]);

          x = nx; y = ny;

        }

        return pts;

      }

    // 3) 不等式：sin x ≤ x（x≥0）と ln x ≤ x-1（x>0）

    function DemoInequalities() {

      const c = findC(F.df, mSec, Math.min(a, x), Math.max(a, x));

        const dfc = F.df(c);

        const Lx1 = xmin, Lx2 = xmax;

        const Ly1 = fc + dfc * (Lx1 - c);

        const Ly2 = fc + dfc * (Lx2 - c);

        return { fc, Lx1, Lx2, Ly1, Ly2 };

      }, [c, xmin, xmax, mode]);


      // sin: cos(c)<=1 → mSec<=1 → sin x <= x

      // ln: 1/c <= 1 → ln x - ln1 <= x-1

      const ok = isSin ? Math.sin(x) <= x + 1e-12 : Math.log(x) <= x - 1 + 1e-12;

              {tangent && (

                <>

                  <line x1={X(tangent.Lx1)} y1={Y(tangent.Ly1)} x2={X(tangent.Lx2)} y2={Y(tangent.Ly2)} className="stroke-sky-600" strokeWidth={2} />

                </>

              )}

            </svg>


            <Slider label={isSin?"x (≥0)":"x (&gt;0)"} min={F.domain[0]} max={F.domain[1]} value={x} onChange={setX} />


            <div className="grid grid-cols-1 md:grid-cols-3 gap-3 text-sm">

              <div className="p-3 bg-gray-50 rounded-xl">割線の傾き m = {fmt(mSec,4)}</div>

              <div className="p-3 bg-gray-50 rounded-xl">c ≈ {fmt(c,4)}</div>

    // 4) テイラー2次近似の誤差上界（誤差チューブ可視化）

        return m;

      }, [a, x, funcKey]);


      // チューブ（上限・下限）を [xmin,xmax] で構成

      const tube = useMemo(() => {

        const N = 220;

        const ts = Array.from({length:N}, (_,i)=> xmin + ((xmax - xmin) * i) / (N-1));


        let supAt = null; // 関数 sup(t)

        if (tubeMode === "local") {

          // a に最も近いインデックス

          const d2abs = ts.map((t)=> Math.abs(d2f(t)));

          const idxA = Math.max(0, Math.min(N-1, Math.round((a - xmin) / (xmax - xmin) * (N-1))));

          const fwd = d2abs.slice();

          for(let i=idxA+1;i<N;i++) fwd[i] = Math.max(fwd[i-1], d2abs[i]);

          const bwd = d2abs.slice();

          for(let i=idxA-1;i>=0;i--) bwd[i] = Math.max(bwd[i+1], d2abs[i]);

          supAt = (t, i) => (i>=idxA ? fwd[i] : bwd[i]);

        } else {

          supAt = () => Kglobal; // 一定K

        }


        const Lvals = ts.map((t)=> L(t));

        const B = ts.map((t, i)=> 0.5 * supAt(t,i) * (t - a) * (t - a));

        const upper = ts.map((t, i)=> Lvals[i] + B[i]);

        const lower = ts.map((t, i)=> Lvals[i] - B[i]);


        // SVG パス（帯 = 上を左→右、下を右→左に辿って閉じる）

        const pU = upper.map((y,i)=> `L ${X(ts[i])} ${Y(y)}`); pU[0] = `M ${X(ts[0])} ${Y(upper[0])}`;

        const pL = lower.map((y,i)=> `L ${X(ts[N-1-i])} ${Y(lower[N-1-i])}`);

        const band = `${pU.join(' ')} ${pL.join(' ')} Z`;


        return { ts, upper, lower, Lvals, band };

      }, [a, xmin, xmax, funcKey, tubeMode, Kglobal]);


      // 直線（補間）

      const Lx1 = xmin, Lx2 = xmax;

      const Ly1 = L(Lx1), Ly2 = L(Lx2);


      // 現在の x での上界（帯の半幅）

      const idxX = useMemo(() => Math.max(0, Math.min(tube.ts.length-1, Math.round((x - xmin)/(xmax - xmin)*(tube.ts.length-1)))), [x, tube, xmin, xmax]);

      const boundAtX = 0.5 * Math.abs(tube.upper[idxX] - tube.lower[idxX]);


      const labelTop = tubeMode === "local" ? "チューブ：|f(t)−L(t)| ≤ ½·sup|f''|(a..t)·(t−a)^2" : "チューブ：|f(t)−L(t)| ≤ ½·K·(t−a)^2";


      return (

        <Card

          title="テイラー近似の誤差（ラグランジュ形）"

          subtitle="接線の周囲に“誤差チューブ”を描く。距離が半分になればズレはおよそ1/4に。"

          right={

            <div className="flex items-center gap-3">

              <select value={funcKey} onChange={(e)=>setFuncKey(e.target.value)} className="border rounded-md text-sm px-2 py-1">

                {Object.values(FUNCS).map(f=> <option key={f.key} value={f.key}>{f.name}</option>)}

              </select>

              <div className="flex items-center gap-2 text-sm">

                <span className="text-gray-600">方式</span>

                <label className="flex items-center gap-1"><input type="radio" name="tube" value="local" checked={tubeMode==='local'} onChange={()=>setTubeMode('local')} />自動</label>

                <label className="flex items-center gap-1"><input type="radio" name="tube" value="global" checked={tubeMode==='global'} onChange={()=>setTubeMode('global')} />一定K</label>

              </div>

            </div>

          }

        >

          <div className="flex flex-col gap-4">

            <svg width={plot.width} height={plot.height} className="w-full">

              <rect x={0} y={0} width={plot.width} height={plot.height} rx={14} className="fill-gray-50" />

              <Axes plot={plot} />


              {/* 誤差チューブ（上限・下限） */}

              <path d={tube.band} className="fill-amber-200 opacity-50" />

              <text x={pad.l + 8} y={pad.t + 14} className="fill-amber-700 text-[11px]">{labelTop}</text>

              <text x={pad.l + 8} y={pad.t + 28} className="fill-emerald-700 text-[11px]">K ≈ {fmt(Kglobal,4)}（区間 [min(a,x), max(a,x)] の sup|f''|）</text>


              {/* 関数と接線 */}

              <path d={path} className="stroke-gray-700 fill-none" strokeWidth={2} />

              <line x1={X(Lx1)} y1={Y(Ly1)} x2={X(Lx2)} y2={Y(Ly2)} className="stroke-sky-600" strokeWidth={2} />


              {/* a, x と縦の誤差 */}

              <circle cx={X(a)} cy={Y(f(a))} r={4} className="fill-sky-600" />

              <circle cx={X(x)} cy={Y(f(x))} r={4} className="fill-emerald-600" />

              <circle cx={X(x)} cy={Y(L(x))} r={4} className="fill-amber-500" />

              <line x1={X(x)} y1={Y(f(x))} x2={X(x)} y2={Y(L(x))} className="stroke-amber-500" strokeDasharray="4 4" />

              <text x={X(x) + 6} y={(Y(f(x))+Y(L(x)))/2} className="fill-amber-700 text-[10px]">|err| ≈ {fmt(Math.abs(errActual),4)} / bound ≈ {fmt(boundAtX,4)}</text>

            </svg>


            <div className="grid grid-cols-1 md:grid-cols-2 gap-3">

              <Slider label="a (展開点)" min={F.domain[0]} max={F.domain[1]} value={a} onChange={setA} />

              <Slider label="x (評価点)" min={F.domain[0]} max={F.domain[1]} value={x} onChange={setXval} />

            </div>


            <div className="grid grid-cols-1 md:grid-cols-4 gap-3 text-sm">

              <div className="p-3 bg-gray-50 rounded-xl">実誤差 |f(x)-L(x)| = {fmt(Math.abs(errActual),6)}</div>

              <div className="p-3 bg-gray-50 rounded-xl">上界 ≈ {fmt(boundAtX,6)}</div>

              <div className="p-3 bg-gray-50 rounded-xl">a = {fmt(a,3)} / x = {fmt(x,3)}</div>

              <div className="p-3 bg-gray-50 rounded-xl">評価: {Math.abs(errActual) <= boundAtX + 1e-12 ? "✅ 内に収まる" : "❌ 超えた"}</div>

            </div>

            <p className="text-sm text-gray-500">「曲率が大きいほどズレやすい」「距離が半分ならズレは四分の一」――その“安全幅”がチューブです。</p>

          </div>

        </Card>

      );

    }


    // 5) Self-check（簡易テストケース）

    function DevTests() {

      // 既存テスト

      const xs1 = [0, 0.5, 1, 2, 3];

      const t1 = xs1.every((x) => Math.sin(x) <= x + 1e-12);


      const xs2 = [0.5, 1, 2, 3];

      const t2 = xs2.every((x) => Math.log(x) <= x - 1 + 1e-12);


      const pairs3 = [

        [-Math.PI / 2, Math.PI / 2],

        [0, Math.PI / 3],

        [-1, 2],

      ];

      const t3 = pairs3.every(([a, b]) => Math.abs(Math.sin(b) - Math.sin(a)) <= Math.abs(b - a) + 1e-12);


      const f4 = (x) => x * x * x - x;

      const df4 = (x) => 3 * x * x - 1;

      const a4 = -1, b4 = 1;

      const m4 = (f4(b4) - f4(a4)) / (b4 - a4);

      const c4 = findC(df4, m4, a4, b4);

      const t4 = near(df4(c4), m4, 1e-2);


      const f5 = (x) => Math.exp(x) / 3;

      const df5 = (x) => Math.exp(x) / 3;

      const d2f5 = (x) => Math.exp(x) / 3;

      const a5 = 0, x5 = 1;

      const L5 = (t) => f5(a5) + df5(a5) * (t - a5);

      const err5 = Math.abs(f5(x5) - L5(x5));

      const bound5 = 0.5 * (Math.exp(1) / 3) * (x5 - a5) * (x5 - a5);

      const t5 = err5 <= bound5 + 1e-12;


      // 追加テスト（UIの直感に対応）

      // 6) リプシッツ：sin、区間 [-1,1]、起点a=0 で帯内

      const t6 = (()=>{

        const a=0, lo=-1, hi=1, M=1; const N=200;

        for(let i=0;i<=N;i++){ const x=lo+(hi-lo)*i/N; const y=Math.sin(x); if(Math.abs(y-Math.sin(a))> M*Math.abs(x-a)+1e-9) return false; }

        return true;

      })();


      // 7) リプシッツ：sin、同区間で起点b側でも帯内

      const t7 = (()=>{

        const b=0, lo=-1, hi=1, M=1; const N=200;

        for(let i=0;i<=N;i++){ const x=lo+(hi-lo)*i/N; const y=Math.sin(x); if(Math.abs(y-Math.sin(b))> M*Math.abs(x-b)+1e-9) return false; }

        return true;

      })();


      // 8) テイラー：x^2 は K=2 で上界＝実誤差に一致

      const t8 = (()=>{

        const f=(x)=>x*x, df=(x)=>2*x, d2f=(x)=>2; const a=0.3, t=0.8;

        const L=f(a)+df(a)*(t-a); const err=Math.abs(f(t)-L); const bound=0.5*Math.abs(d2f(0))*(t-a)*(t-a);

        return Math.abs(err-bound) < 1e-9; // ほぼ一致

      })();


      // 9) テイラー：e^x/3 で 局所境界 ≤ 一定K境界（a=0, x=1）

      const t9 = (()=>{

        const f=(x)=>Math.exp(x)/3, df=(x)=>Math.exp(x)/3, d2f=(x)=>Math.exp(x)/3; const a=0, x=1; const K=Math.exp(1)/3; const N=100;

        for(let i=0;i<=N;i++){ const t=i/N; const local=0.5*(Math.exp(t)/3)*(t-a)*(t-a); const global=0.5*K*(t-a)*(t-a); if(local>global+1e-9) return false; }

        return true;

      })();


      // 10) 道予測：M を小さく取りすぎると真の関数が帯から出る（sin, a=0, M=0.9, x=0.1）

      const t10 = (()=>{

        const a=0, M=0.9, x=0.1; return Math.abs(Math.sin(x)-Math.sin(a)) > M*Math.abs(x-a) - 1e-12; // はみ出すなら true

      })();


      // 11) 道予測：|Δy| ≤ M|Δx| を積み上げた道は常に帯内

      const t11 = (()=>{

        const M=1.2, a=0.0, x1=1.0; const steps=50; let x=a, y=0; // y0=0 とする

        const lower=(t)=> 0 - M*Math.abs(t-a), upper=(t)=> 0 + M*Math.abs(t-a);

        for(let i=0;i<steps;i++){

          const dx=(x1-a)/steps; const s = (i%2===0? 0.6 : -0.6) * M; // 交互に±0.6M

          const nx = x + dx; let ny = y + s*dx; if(ny<lower(nx)) ny=lower(nx); if(ny>upper(nx)) ny=upper(nx);

          if(ny>upper(nx)+1e-9 || ny<lower(nx)-1e-9) return false; x=nx; y=ny;

        }

        return true;

      })();


      // 12) 観測で帯が狭まる（sin, a=0, x_obs=0.5, M=1, ε=0 で x=0.25 の幅が縮む）

      const t12 = (()=>{

        const a=0, M=1, x=0.25, xobs=0.5, eps=0; const y0=Math.sin(a), yobs=Math.sin(xobs);

        const low1=y0 - M*Math.abs(x-a), high1=y0 + M*Math.abs(x-a);

        const low2=yobs - eps - M*Math.abs(x-xobs), high2=yobs + eps + M*Math.abs(x-xobs);

        const w0=high1-low1; const w1=Math.max(0, Math.min(high1, high2) - Math.max(low1, low2));

        return w1 < w0 - 1e-6;

      })();


      // 13) 観測と M が矛盾（sin, a=0, x_obs=0.5, M=0.3, ε=0 で x=0.5 交差なし）

      const t13 = (()=>{

        const a=0, M=0.3, x=0.5, xobs=0.5, eps=0; const y0=Math.sin(a), yobs=Math.sin(xobs);

        const low1=y0 - M*Math.abs(x-a), high1=y0 + M*Math.abs(x-a);

        const low2=yobs - eps - M*Math.abs(x-xobs), high2=yobs + eps + M*Math.abs(x-xobs);

        return Math.min(high1, high2) < Math.max(low1, low2) - 1e-12; // 交差しない

      })();


      const cases = [

        { name: "sin x ≤ x（0,0.5,1,2,3）", ok: t1, detail: xs1.map((x)=>`sin(${fmt(x,2)})≤${fmt(x,2)}`).join(", ") },

        { name: "ln x ≤ x-1（0.5,1,2,3）", ok: t2, detail: xs2.map((x)=>`ln(${fmt(x,2)})≤${fmt(x,2)}-1`).join(", ") },

        { name: "リプシッツ：|sin b-sin a|≤|b-a|", ok: t3, detail: pairs3.map(([a,b])=>`[${fmt(a,2)},${fmt(b,2)}]`).join("; ") },

        { name: "MVT：f(x)=x^3-x（a=-1,b=1）", ok: t4, detail: `m=${fmt(m4,4)}, f'(c)≈${fmt(df4(c4),4)}, c≈${fmt(c4,4)}` },

        { name: "Taylor：e^x/3（一次, a=0→x=1）", ok: t5, detail: `|err|=${fmt(err5,6)} ≤ bound=${fmt(bound5,6)}` },

        { name: "Lipschitz 帯：sin（起点a、区間[-1,1]）", ok: t6, detail: "M=1 で帯内" },

        { name: "Lipschitz 帯：sin（起点b、区間[-1,1]）", ok: t7, detail: "M=1 で帯内" },

        { name: "Taylor：x^2 は上界に一致", ok: t8, detail: "K=2 で誤差＝境界" },

        { name: "Taylor：局所 ≤ 一定K（e^x/3）", ok: t9, detail: "local bound ≤ global bound" },

        { name: "Predict：M 小さすぎると帯から出る（sin, M=0.9）", ok: t10, detail: "はみ出し検知" },

        { name: "Predict：積み上げ道は帯内", ok: t11, detail: "|Δy| ≤ M|Δx|" },

        { name: "Predict+Obs：観測で帯が狭まる", ok: t12, detail: "x_obs=0.5, x=0.25" },

        { name: "Predict+Obs：観測とMが矛盾を検出", ok: t13, detail: "M=0.3, ε=0" },

      ];


      return (

        <Card title="Self-check（簡易テスト）" subtitle="主要な性質が数値的に成立するかのスモークテスト">

          <ul className="text-sm space-y-2">

            {cases.map((c, i) => (

              <li key={i} className="flex items-center gap-2">

                <span className="w-5">{c.ok ? "✅" : "❌"}</span>

                <span className="font-medium">{c.name}</span>

                <span className="text-gray-500">— {c.detail}</span>

              </li>

            ))}

    }
