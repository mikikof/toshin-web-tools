    <script type="importmap">

        {

            "imports": {

                "three": "https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js",

                "three/addons/": "https://cdn.jsdelivr.net/npm/three@0.160.0/examples/jsm/"

            }

        }

    <script type="module">

        import * as THREE from 'three';

        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';


        // --- グローバル変数 ---

        let scene2D, camera2D, renderer2D, circleGroup;

        let sceneCavalieri, cameraCavalieri, rendererCavalieri, controlsCavalieri;

        let sceneSphere, cameraSphere, rendererSphere, controlsSphere, sphereSliceGroup;

        let hemisphere, cylinder, cone, slicePlane, crossSectionHemi, crossSectionCyl;

        

        const R = 1; // 3Dオブジェクトの基準半径

        const cavalieriFrustumSize = 4.5; // カメラの表示領域サイズ


        // UI要素

        const SLICES_2D_SLIDER = document.getElementById('slices-2d-slider');

        const SLICES_2D_LABEL = document.getElementById('slices-2d-label');

        

        const SLICE_HEIGHT_SLIDER = document.getElementById('slice-height-slider');

        const SLICE_HEIGHT_LABEL = document.getElementById('slice-height-label');

        const AREA_A_LABEL = document.getElementById('area-a');

        const AREA_B_LABEL = document.getElementById('area-b');

        

        const SLICES_3D_SPHERE_SLIDER = document.getElementById('slices-3d-sphere-slider');

        const SLICES_3D_SPHERE_LABEL = document.getElementById('slices-3d-sphere-label');


        // --- 初期化処理 ---

        function init() {

            init2D();

            initSphereDemo();

            initCavalieriDemo();

            animate();

            setupEventListeners();

            updateSliceHeight(parseFloat(SLICE_HEIGHT_SLIDER.value) / 100);

        }


        // --- 2Dデモの初期化 ---

        function init2D() {

            const container = document.getElementById('canvas-2d-container');

            scene2D = new THREE.Scene();

            scene2D.background = new THREE.Color(0xffffff);

            camera2D = new THREE.OrthographicCamera(container.clientWidth / -2, container.clientWidth / 2, container.clientHeight / 2, container.clientHeight / -2, 1, 1000);

            camera2D.position.z = 10;

            camera2D.zoom = 2.5;

            camera2D.updateProjectionMatrix();

            renderer2D = new THREE.WebGLRenderer({ antialias: true });

            renderer2D.setSize(container.clientWidth, container.clientHeight);

            container.appendChild(renderer2D.domElement);

            circleGroup = new THREE.Group();

            scene2D.add(circleGroup);

            createCircleSlices(parseInt(SLICES_2D_SLIDER.value));

        }


        // --- 球体構成デモの初期化 ---

        function initSphereDemo() {

            const container = document.getElementById('canvas-3d-sphere-container');

            sceneSphere = new THREE.Scene();

            sceneSphere.background = new THREE.Color(0xffffff);

            

            cameraSphere = new THREE.PerspectiveCamera(75, container.clientWidth / container.clientHeight, 0.1, 1000);

            cameraSphere.position.set(0, 1.5, 3.5);


            const ambientLight = new THREE.AmbientLight(0xffffff, 1.2);

            sceneSphere.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 2.0);

            directionalLight.position.set(1.5, 2.5, 1.5);

            sceneSphere.add(directionalLight);


            rendererSphere = new THREE.WebGLRenderer({ antialias: true });

            rendererSphere.setSize(container.clientWidth, container.clientHeight);

            container.appendChild(rendererSphere.domElement);


            controlsSphere = new OrbitControls(cameraSphere, rendererSphere.domElement);

            controlsSphere.enableDamping = true;


            sphereSliceGroup = new THREE.Group();

            sceneSphere.add(sphereSliceGroup);


            createFullSphereSlices(parseInt(SLICES_3D_SPHERE_SLIDER.value));

        }


        // --- カヴァリエリの原理デモの初期化 ---

        function initCavalieriDemo() {

            const container = document.getElementById('canvas-3d-container');

            sceneCavalieri = new THREE.Scene();

            sceneCavalieri.background = new THREE.Color(0xffffff);


            // ★修正点: PerspectiveCameraからOrthographicCameraに変更

            const aspect = container.clientWidth / container.clientHeight;

            cameraCavalieri = new THREE.OrthographicCamera(cavalieriFrustumSize * aspect / -2, cavalieriFrustumSize * aspect / 2, cavalieriFrustumSize / 2, cavalieriFrustumSize / -2, 0.1, 100);

            cameraCavalieri.position.set(0, 1.5, 10); // 平行投影なのでZの位置はズームに影響しないが、クリッピングのために設定


            const ambientLight = new THREE.AmbientLight(0xffffff, 1.2);

            sceneCavalieri.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 2.2);

            directionalLight.position.set(2, 3, 2);

            sceneCavalieri.add(directionalLight);


            rendererCavalieri = new THREE.WebGLRenderer({ antialias: true });

            rendererCavalieri.setSize(container.clientWidth, container.clientHeight);

            container.appendChild(rendererCavalieri.domElement);


            controlsCavalieri = new OrbitControls(cameraCavalieri, rendererCavalieri.domElement);

            controlsCavalieri.enableDamping = true;

            controlsCavalieri.target.set(0, R / 2, 0);

            // 平行投影カメラではズームが直感的でないため、無効化

            controlsCavalieri.enableZoom = false;


            createCavalieriObjects();

        }

        

        // --- 2D円のスライスを作成 ---

        function createCircleSlices(numSlices) {

            circleGroup.clear();

            const radius = 60;

            const sliceWidth = (radius * 2) / numSlices;

            

            for (let i = 0; i < numSlices; i++) {

                const x = -radius + sliceWidth * (i + 0.5);

                const height = 2 * Math.sqrt(Math.max(0, radius * radius - x * x));

                

                const geometry = new THREE.PlaneGeometry(sliceWidth, height);

                const color = new THREE.Color(`hsl(${(i / numSlices) * 360}, 90%, 70%)`);

                const material = new THREE.MeshBasicMaterial({ color: color, side: THREE.DoubleSide });

                const slice = new THREE.Mesh(geometry, material);

                

                slice.position.x = x;

                circleGroup.add(slice);

            }

        }


        // --- 球のスライスを作成 ---

        function createFullSphereSlices(numSlices) {

            sphereSliceGroup.clear();

            const radius = 1.2;

            const sliceHeight = (radius * 2) / numSlices;


            for (let i = 0; i < numSlices; i++) {

                const y = -radius + sliceHeight * (i + 0.5);

                const sliceRadius = Math.sqrt(radius * radius - y * y);

                

                const geometry = new THREE.CylinderGeometry(sliceRadius, sliceRadius, sliceHeight, 32);

                const color = new THREE.Color(`hsl(${((y + radius) / (radius*2)) * 360}, 90%, 70%)`);

                const material = new THREE.MeshStandardMaterial({ color: color, roughness: 0.5, metalness: 0.1 });

                const slice = new THREE.Mesh(geometry, material);

                

                slice.position.y = y;

                sphereSliceGroup.add(slice);

            }

        }


        // --- カヴァリエリの原理用オブジェクト作成 ---

        function createCavalieriObjects() {

            const group = new THREE.Group();

            const separation = R * 2.5;


            const hemiGeom = new THREE.SphereGeometry(R, 64, 32, 0, Math.PI * 2, 0, Math.PI / 2);

            const hemiMat = new THREE.MeshStandardMaterial({ color: 0x4ade80, transparent: true, opacity: 0.3, side: THREE.DoubleSide });

            hemisphere = new THREE.Mesh(hemiGeom, hemiMat);

            hemisphere.position.x = separation / 2;

            group.add(hemisphere);


            const cylGeom = new THREE.CylinderGeometry(R, R, R, 64, 1, true);

            const cylMat = new THREE.MeshStandardMaterial({ color: 0x60a5fa, transparent: true, opacity: 0.4, side: THREE.DoubleSide });

            cylinder = new THREE.Mesh(cylGeom, cylMat);

            cylinder.position.set(-separation / 2, R / 2, 0);

            group.add(cylinder);


            const coneGeom = new THREE.ConeGeometry(R, R, 64, 1, true);

            const coneMat = new THREE.MeshStandardMaterial({ color: 0x4b5563, side: THREE.BackSide });

            cone = new THREE.Mesh(coneGeom, coneMat);

            cone.position.set(-separation / 2, R / 2, 0);

            cone.rotation.x = Math.PI;

            group.add(cone);


            const planeGeom = new THREE.PlaneGeometry(separation + R * 2, R * 1.5);

            const planeMat = new THREE.MeshBasicMaterial({ color: 0x333333, transparent: true, opacity: 0.08, side: THREE.DoubleSide });

            slicePlane = new THREE.Mesh(planeGeom, planeMat);

            slicePlane.rotation.x = -Math.PI / 2;

            group.add(slicePlane);


            const crossHemiGeom = new THREE.RingGeometry(0, 1, 64);

            const crossHemiMat = new THREE.MeshBasicMaterial({ color: 0x16a34a, side: THREE.DoubleSide, depthTest: false });

            crossSectionHemi = new THREE.Mesh(crossHemiGeom, crossHemiMat);

            crossSectionHemi.rotation.x = -Math.PI / 2;

            crossSectionHemi.position.x = separation / 2;

            crossSectionHemi.renderOrder = 1; 

            group.add(crossSectionHemi);

            

            const crossCylGeom = new THREE.RingGeometry(0, R, 64);

            const crossCylMat = new THREE.MeshBasicMaterial({ color: 0x2563eb, side: THREE.DoubleSide, depthTest: false });

            crossSectionCyl = new THREE.Mesh(crossCylGeom, crossCylMat);

            crossSectionCyl.rotation.x = -Math.PI / 2;

            crossSectionCyl.position.x = -separation / 2;

            crossSectionCyl.renderOrder = 1; 

            group.add(crossSectionCyl);


            sceneCavalieri.add(group);

        }

        

        // --- スライスの高さを更新 ---

        function updateSliceHeight(h_ratio) {

            const h = h_ratio * R;

            SLICE_HEIGHT_LABEL.textContent = h.toFixed(2);

            slicePlane.position.y = h;

            const r_hemi_sq = R * R - h * h;

            const r_hemi = Math.sqrt(Math.max(0, r_hemi_sq));

            crossSectionHemi.scale.set(r_hemi, r_hemi, 1);

            crossSectionHemi.position.y = h + 0.001;

            const r_cone = h; 

            sceneCavalieri.remove(crossSectionCyl);

            crossSectionCyl.geometry.dispose();

            const newCrossCylGeom = new THREE.RingGeometry(r_cone, R, 64);

            crossSectionCyl.geometry = newCrossCylGeom;

            sceneCavalieri.add(crossSectionCyl);

            crossSectionCyl.position.y = h + 0.001; 

            crossSectionCyl.position.x = -(R * 2.5) / 2;

            crossSectionCyl.rotation.x = -Math.PI / 2;

            const area = Math.PI * r_hemi_sq;

            if (area >= 0) {

                AREA_A_LABEL.textContent = area.toFixed(4);

                AREA_B_LABEL.textContent = area.toFixed(4);

            } else {

                AREA_A_LABEL.textContent = "0.0000";

                AREA_B_LABEL.textContent = "0.0000";

            }

        }


        // --- イベントリスナーの設定 ---

        function setupEventListeners() {

            SLICES_2D_SLIDER.addEventListener('input', (e) => {

                const numSlices = parseInt(e.target.value);

                SLICES_2D_LABEL.textContent = numSlices;

                createCircleSlices(numSlices);

            });

            

            SLICE_HEIGHT_SLIDER.addEventListener('input', (e) => {

                updateSliceHeight(parseFloat(e.target.value) / 100);

            });


            SLICES_3D_SPHERE_SLIDER.addEventListener('input', (e) => {

                const numSlices = parseInt(e.target.value);

                SLICES_3D_SPHERE_LABEL.textContent = numSlices;

                createFullSphereSlices(numSlices);

            });


            window.addEventListener('resize', onWindowResize, false);

        }

        

        // --- ウィンドウリサイズ処理 ---

        function onWindowResize() {

            // 2D

            const container2D = document.getElementById('canvas-2d-container');

            if(container2D && renderer2D) {

                camera2D.left = container2D.clientWidth / -2;

                camera2D.right = container2D.clientWidth / 2;

                camera2D.top = container2D.clientHeight / 2;

                camera2D.bottom = container2D.clientHeight / -2;

                camera2D.updateProjectionMatrix();

                renderer2D.setSize(container2D.clientWidth, container2D.clientHeight);

            }


            // Sphere Demo

            const containerSphere = document.getElementById('canvas-3d-sphere-container');

             if(containerSphere && rendererSphere) {

                cameraSphere.aspect = containerSphere.clientWidth / containerSphere.clientHeight;

                cameraSphere.updateProjectionMatrix();

                rendererSphere.setSize(containerSphere.clientWidth, containerSphere.clientHeight);

            }


            // Cavalieri Demo

            const containerCavalieri = document.getElementById('canvas-3d-container');

            if(containerCavalieri && rendererCavalieri) {

                // ★修正点: OrthographicCameraのリサイズ処理

                const aspect = containerCavalieri.clientWidth / containerCavalieri.clientHeight;

                cameraCavalieri.left = cavalieriFrustumSize * aspect / -2;

                cameraCavalieri.right = cavalieriFrustumSize * aspect / 2;

                cameraCavalieri.top = cavalieriFrustumSize / 2;

                cameraCavalieri.bottom = cavalieriFrustumSize / -2;

                cameraCavalieri.updateProjectionMatrix();

                rendererCavalieri.setSize(containerCavalieri.clientWidth, containerCavalieri.clientHeight);

            }

        }


        // --- アニメーションループ ---

        function animate() {

            requestAnimationFrame(animate);

            if(renderer2D) renderer2D.render(scene2D, camera2D);

            if(rendererSphere) {

                controlsSphere.update();

                rendererSphere.render(sceneSphere, cameraSphere);

            }

            if(rendererCavalieri) {

                controlsCavalieri.update();

                rendererCavalieri.render(sceneCavalieri, cameraCavalieri);

            }

        }


        // --- 実行 ---

        init();
