    <script type="module">

        import * as THREE from 'three';

        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';


        // --- グローバル変数と定数 ---

        let scene, camera, renderer, controls;

        let coneGroup, planeMesh, planeObject;

        const coneHeight = 10;

        const coneRadius = 5;

        const coneAngle = Math.atan(coneRadius / coneHeight);


        const angleSlider = document.getElementById('angle-slider');

        const positionSlider = document.getElementById('position-slider');

        const angleValue = document.getElementById('angle-value');

        const positionValue = document.getElementById('position-value');

        const sectionName = document.getElementById('section-name');

        const equationText = document.getElementById('equation-text');

        

        const threeContainer = document.getElementById('three-container');

        const graphCanvas = document.getElementById('graph-canvas');

        const graphCtx = graphCanvas.getContext('2d');


        // --- 初期化 ---

        function init() {

            setupScene();

            createObjects();

            setupEventListeners();

            updatePlane();

            animate();

        }


        function setupScene() {

            scene = new THREE.Scene();

            camera = new THREE.PerspectiveCamera(75, threeContainer.clientWidth / threeContainer.clientHeight, 0.1, 1000);

            camera.position.set(15, 5, 15);

            camera.lookAt(0, 0, 0);


            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

            renderer.setSize(threeContainer.clientWidth, threeContainer.clientHeight);

            renderer.setPixelRatio(window.devicePixelRatio);

            renderer.localClippingEnabled = true;

            threeContainer.appendChild(renderer.domElement);


            const ambientLight = new THREE.AmbientLight(0xffffff, 1.5);

            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 1.5);

            directionalLight.position.set(10, 20, 5);

            scene.add(directionalLight);


            controls = new OrbitControls(camera, renderer.domElement);

            controls.enableDamping = true;

            controls.dampingFactor = 0.05;

        }


        function createObjects() {

            const coneGeometry = new THREE.ConeGeometry(coneRadius, coneHeight, 128, 64, true);

            const coneMaterial = new THREE.MeshStandardMaterial({

                color: 0x9f7aea,

                transparent: true,

                opacity: 0.1,

                side: THREE.DoubleSide,

                clippingPlanes: [],

                clipShadows: true,

                depthWrite: false

            });

            

            const upperCone = new THREE.Mesh(coneGeometry, coneMaterial);

            upperCone.rotation.x = Math.PI;

            upperCone.position.y = coneHeight / 2;


            const lowerCone = new THREE.Mesh(coneGeometry, coneMaterial);

            lowerCone.position.y = -coneHeight / 2;


            coneGroup = new THREE.Group();

            coneGroup.add(upperCone);

            coneGroup.add(lowerCone);

            scene.add(coneGroup);


            const wireframeGeom = new THREE.WireframeGeometry(coneGeometry);

            const wireframeMat = new THREE.LineBasicMaterial({ color: 0xc4b5fd, transparent: true, opacity: 0.4 });

            

            const upperWireframe = new THREE.LineSegments(wireframeGeom, wireframeMat);

            upperWireframe.rotation.x = Math.PI;

            upperWireframe.position.y = coneHeight / 2;


            const lowerWireframe = new THREE.LineSegments(wireframeGeom, wireframeMat);

            lowerWireframe.position.y = -coneHeight / 2;


            scene.add(upperWireframe);

            scene.add(lowerWireframe);


            planeObject = new THREE.Plane(new THREE.Vector3(0, 1, 0), 0);

            coneMaterial.clippingPlanes = [planeObject];


            const planeGeometry = new THREE.PlaneGeometry(30, 30);

            planeGeometry.rotateX(-Math.PI / 2);

            const planeMaterial = new THREE.MeshStandardMaterial({

                color: 0xa0aec0,

                side: THREE.DoubleSide,

                transparent: true,

                opacity: 0.5

            });

            planeMesh = new THREE.Mesh(planeGeometry, planeMaterial);

            scene.add(planeMesh);

        }


        function setupEventListeners() {

            angleSlider.addEventListener('input', updatePlane);

            positionSlider.addEventListener('input', updatePlane);

            

            // ★修正点: プリセットボタンの値を正しいものに修正

            const parabolaAngle = 90 - (coneAngle * 180 / Math.PI);

            document.getElementById('btn-circle').addEventListener('click', () => setPreset(0, 2));

            document.getElementById('btn-ellipse').addEventListener('click', () => setPreset(30, 2.5));

            document.getElementById('btn-parabola').addEventListener('click', () => setPreset(parabolaAngle, 2.5));

            document.getElementById('btn-hyperbola').addEventListener('click', () => setPreset(75, 2));

        }


        function setPreset(angle, position) {

            angleSlider.value = angle;

            positionSlider.value = position;

            updatePlane();

        }


        function updatePlane() {

            const angle = parseFloat(angleSlider.value) * (Math.PI / 180);

            const position = parseFloat(positionSlider.value);


            angleValue.textContent = `${parseFloat(angleSlider.value).toFixed(1)}°`;

            positionValue.textContent = position.toFixed(2);


            planeMesh.position.set(0, position, 0);

            planeMesh.rotation.set(0, 0, 0);

            planeMesh.rotateX(angle);

            planeMesh.updateMatrixWorld();


            const normal = new THREE.Vector3(0, 1, 0);

            normal.applyQuaternion(planeMesh.quaternion);

            

            planeObject.setFromNormalAndCoplanarPoint(normal, planeMesh.position);

            planeObject.negate();


            calculateAndDrawSection();

        }


        function calculateAndDrawSection() {

            const planeAngle = parseFloat(angleSlider.value) * (Math.PI / 180);

            const planePos = parseFloat(positionSlider.value);

            const generatrixAngle = coneAngle;

            const angleToAxis = Math.PI / 2 - planeAngle;

            const epsilon = 1e-3;


            let type = '';

            if (Math.abs(planePos) < epsilon) {

                if (Math.abs(angleToAxis - generatrixAngle) < epsilon) type = '直線';

                else if (angleToAxis > generatrixAngle) type = '点';

                else type = '交わる2直線';

            } else {

                if (Math.abs(angleToAxis - generatrixAngle) < epsilon) type = '放物線';

                else if (angleToAxis > generatrixAngle) {

                    type = '楕円';

                    if (Math.abs(angleToAxis - Math.PI/2) < epsilon) type = '円';

                } else type = '双曲線';

            }

            sectionName.textContent = type;

            

            draw2DViewsFromEquation(type, planeAngle, planePos);

        }


        function draw2DViewsFromEquation(type, planeAngle, planePos) {

            resizeCanvas(graphCanvas, graphCtx);

            graphCtx.clearRect(0, 0, graphCanvas.width, graphCanvas.height);

            

            const viewRange = 10;

            drawGraphAxes(graphCtx, graphCanvas, viewRange);


            const k = Math.tan(coneAngle);

            const cos = Math.cos(planeAngle);

            const sin = Math.sin(planeAngle);

            const epsilon = 1e-3;

            let equationStr = ' ';


            graphCtx.strokeStyle = '#68d391';

            graphCtx.lineWidth = 3;


            let a = 0, b = 0, r = 0, p = 0, a_vert = 0, b_vert = 0;


            switch(type) {

                case '円':

                    r = Math.abs(planePos * k);

                    if (r > 0.01) {

                        drawCircle(graphCtx, graphCanvas, r, viewRange);

                        equationStr = `x² + y² = ${r.toFixed(2)}²`;

                    }

                    break;

                case '楕円': {

                    const term = cos*cos - k*k*sin*sin;

                    if (term > 0) {

                        a = Math.abs(planePos * k * cos / term);

                        b = Math.abs(planePos * k / Math.sqrt(term));

                         if (a > 0.01 && b > 0.01) {

                            drawEllipse(graphCtx, graphCanvas, a, b, viewRange);

                            equationStr = `x²/${a.toFixed(2)}² + y²/${b.toFixed(2)}² = 1`;

                        }

                    }

                    break;

                }

                case '放物線':

                    p = 2 * planePos * k * Math.cos(coneAngle);

                    if (Math.abs(p) > 1e-6) {

                        drawParabolaOnGraph(graphCtx, graphCanvas, p, viewRange);

                        equationStr = `y² = ${p.toFixed(2)}x`;

                    }

                    break;

                case '双曲線':

                    if (Math.abs(planeAngle - Math.PI/2) < epsilon) {

                        const d = Math.abs(planePos);

                        if (d > 0.01 && d < coneRadius) {

                            a_vert = d / k;

                            b_vert = d;

                            drawVerticalHyperbolaOnGraph(graphCtx, graphCanvas, a_vert, b_vert, viewRange);

                            equationStr = `y²/${a_vert.toFixed(2)}² - x²/${b_vert.toFixed(2)}² = 1`;

                        }

                    } else {

                        const term = k*k*sin*sin - cos*cos;

                        if (term > 0) {

                            a = Math.abs(planePos * k * cos / term);

                            b = Math.abs(planePos * k * Math.sqrt(term) / term);

                            if (a > 0.01 && b > 0.01) {

                                drawHyperbolaOnGraph(graphCtx, graphCanvas, a, b, viewRange);

                                equationStr = `x²/${a.toFixed(2)}² - y²/${b.toFixed(2)}² = 1`;

                            }

                        }

                    }

                    break;

                case '点':

                    drawCircle(graphCtx, graphCanvas, 0, viewRange, true);

                    equationStr = `x² + y² = 0`;

                    break;

                case '直線':

                case '交わる2直線':

                    equationStr = type;

                    break;

            }

            equationText.textContent = equationStr;

        }

        

        function resizeCanvas(canvas, ctx) {

            const { width, height } = canvas.getBoundingClientRect();

            if (canvas.width !== width || canvas.height !== height) {

                canvas.width = width;

                canvas.height = height;

                return true;

            }

            return false;

        }


        function drawGraphAxes(ctx, canvas, viewRange) {

            const w = canvas.width;

            const h = canvas.height;

            const cx = w / 2;

            const cy = h / 2;

            ctx.beginPath();

            ctx.strokeStyle = '#4a5568';

            ctx.lineWidth = 1;

            // Grid

            const unit = w / (viewRange * 2);

            for(let i = -viewRange; i <= viewRange; i++) {

                ctx.moveTo(cx + i * unit, 0); ctx.lineTo(cx + i * unit, h);

                ctx.moveTo(0, cy + i * unit); ctx.lineTo(w, cy + i * unit);

            }

            ctx.stroke();


            ctx.beginPath();

            ctx.strokeStyle = '#a0aec0';

            ctx.lineWidth = 1.5;

            // Axes

            ctx.moveTo(0, cy); ctx.lineTo(w, cy);

            ctx.moveTo(cx, 0); ctx.lineTo(cx, h);

            ctx.stroke();

            

            // Arrows

            ctx.fillStyle = '#a0aec0';

            ctx.moveTo(w, cy); ctx.lineTo(w - 10, cy - 5); ctx.lineTo(w - 10, cy + 5); ctx.fill();

            ctx.moveTo(cx, 0); ctx.lineTo(cx - 5, 10); ctx.lineTo(cx + 5, 10); ctx.fill();

            

            ctx.font = '12px Inter';

            ctx.fillStyle = '#a0aec0';

            ctx.fillText('x', w - 15, cy - 10);

            ctx.fillText('y', cx + 10, 15);

        }


        function drawCircle(ctx, canvas, r, viewRange, isPoint = false) {

            const scale = Math.min(canvas.width, canvas.height) / (viewRange * 2);

            const cx = canvas.width / 2;

            const cy = canvas.height / 2;

            if (isPoint) {

                ctx.fillStyle = ctx.strokeStyle;

                ctx.beginPath();

                ctx.arc(cx, cy, 3, 0, 2 * Math.PI);

                ctx.fill();

            } else {

                ctx.beginPath();

                ctx.arc(cx, cy, r * scale, 0, 2 * Math.PI);

                ctx.stroke();

            }

        }


        function drawEllipse(ctx, canvas, a, b, viewRange) {

            const scale = Math.min(canvas.width, canvas.height) / (viewRange * 2);

            const cx = canvas.width / 2;

            const cy = canvas.height / 2;

            ctx.beginPath();

            ctx.ellipse(cx, cy, a * scale, b * scale, 0, 0, 2 * Math.PI);

            ctx.stroke();

        }

        

        function drawParabolaOnGraph(ctx, canvas, p, viewRange) {

            const scale = Math.min(canvas.width, canvas.height) / (viewRange * 2);

            const cx = canvas.width / 2;

            const cy = canvas.height / 2;

            ctx.beginPath();

            let pathStarted = false;

            for(let y_val = -viewRange; y_val <= viewRange; y_val += 0.05) {

                const x_val = y_val * y_val / p;

                if (Math.abs(x_val) > viewRange) {

                    if (pathStarted) ctx.stroke();

                    pathStarted = false;

                    continue;

                }

                const canvasX = cx + x_val * scale;

                const canvasY = cy - y_val * scale;

                if (!pathStarted) {

                    ctx.beginPath();

                    ctx.moveTo(canvasX, canvasY);

                    pathStarted = true;

                } else {

                    ctx.lineTo(canvasX, canvasY);

                }

            }

            if(pathStarted) ctx.stroke();

        }

        

        function drawHyperbolaOnGraph(ctx, canvas, a, b, viewRange) {

            const scale = Math.min(canvas.width, canvas.height) / (viewRange * 2);

            const cx = canvas.width / 2;

            const cy = canvas.height / 2;

            

            ctx.beginPath();

            let pathStarted = false;

            for (let y_val = -viewRange; y_val <= viewRange; y_val += 0.05) {

                const x_val = a * Math.sqrt(1 + (y_val * y_val) / (b * b));

                if (Math.abs(x_val) > viewRange) {

                    if(pathStarted) ctx.stroke();

                    pathStarted = false;

                    continue;

                }

                const canvasX = cx + x_val * scale;

                const canvasY = cy - y_val * scale;

                if (!pathStarted) {

                    ctx.beginPath();

                    ctx.moveTo(canvasX, canvasY);

                    pathStarted = true;

                } else {

                    ctx.lineTo(canvasX, canvasY);

                }

            }

            if(pathStarted) ctx.stroke();

            

            ctx.beginPath();

            pathStarted = false;

            for (let y_val = -viewRange; y_val <= viewRange; y_val += 0.05) {

                const x_val = -a * Math.sqrt(1 + (y_val * y_val) / (b * b));

                 if (Math.abs(x_val) > viewRange) {

                    if(pathStarted) ctx.stroke();

                    pathStarted = false;

                    continue;

                 }

                const canvasX = cx + x_val * scale;

                const canvasY = cy - y_val * scale;

                if (!pathStarted) {

                    ctx.beginPath();

                    ctx.moveTo(canvasX, canvasY);

                    pathStarted = true;

                } else {

                    ctx.lineTo(canvasX, canvasY);

                }

            }

            if(pathStarted) ctx.stroke();

        }
